00164         // Make a bool vector to keep track of which actors to keep when updating.
00165         std::vector<bool> Tracker;
00166         Tracker.resize(OverlappingActors.size());
00167         std::vector<bool>::iterator bit;
00168         for ( bit = Tracker.begin() ; bit != Tracker.end() ; bit++ )
00169         {
00170             (*bit) = false;
00171         }
00172         // Add objects to the necessary lists as needed.  Also track what needs to be removed for later.
00173         for ( int x=0 ; x<OverlappingPairs ; x++ )
00174         {
00175             // Get the non-ghost object from a given pair and cast it to a usable pointer.
00176             btCollisionObject* ColObj = Pair[x].m_pProxy0->m_clientObject != Ghost ? (btCollisionObject*)(Pair[x].m_pProxy0->m_clientObject) : (btCollisionObject*)(Pair[x].m_pProxy1->m_clientObject);
00177             ObjectReference* ActorRef = (ObjectReference*)(ColObj->getUserPointer());
00178             ActorBase* Actor = NULL;
00179             if(phys::WOT_AreaEffect > ActorRef->GetType())
00180                 Actor = (ActorBase*)ActorRef->GetObject();
00181             else
00182                 continue;
00183             // Check list for the actor in the pair.
00184             for( it = OverlappingActors.begin(), bit = Tracker.begin() ; it != OverlappingActors.end() ; it++, bit++ )
00185             {
00186                 if ( Actor == (*it) )
00187                 {
00188                     (*bit) = true;
00189                     break;
00190                 }
00191             }
00192             if ( it == OverlappingActors.end() )
00193             {
00194                 AddActorToList(Actor);
00195                 Tracker.push_back(true);
00196             }
00197         }
00198         // Verify they are the same size.  Then remove items from the list as necessary.
00199         if ( OverlappingActors.size() == Tracker.size() )
00200         {
00201             std::list<ActorBase*>::iterator sit = OverlappingActors.begin();
00202             for ( bit = Tracker.begin() ; bit != Tracker.end() ; )
00203             {
00204                 if ( (*bit) == false )
00205                 {
00206                     bit = Tracker.erase(bit);
00207                     ActorBase* Act = (*sit);
00208                     RemovedActors.push_back(Act);
00209                     sit = OverlappingActors.erase(sit);
00210                 }else{
00211                     sit++;
00212                     bit++;
00213                 }
00214             }
00215         }
00216     }

00254     {

00258         {
00259             delete Shape;

00262 
00263         GraphicsObject = OgreManager->createEntity(Name, Filename, Group);
00264         Ogre::MeshPtr myMesh = GraphicsObject->getMesh();

00266         Ogre::IndexData*  indexData = subMesh->indexData;
00267         Ogre::VertexData* vertexData = subMesh->vertexData;
00268 
00269         const Ogre::VertexElement* posElem = vertexData->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
00270         Ogre::HardwareVertexBufferSharedPtr vBuffer = vertexData->vertexBufferBinding->getBuffer(posElem->getSource());
00271         Ogre::HardwareIndexBufferSharedPtr iBuffer = indexData->indexBuffer;
00272 
00273         unsigned int triCount = indexData->indexCount/3;
00274         Ogre::Vector3* vertices = new Ogre::Vector3[vertexData->vertexCount];
00275         unsigned int* indices = new unsigned int[indexData->indexCount];
00276         unsigned char* vertex = static_cast<unsigned char*>(vBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00277         float* pReal = NULL;
00278         for (size_t j = 0; j < vertexData->vertexCount; j++, vertex += vBuffer->getVertexSize() )
00279         {
00280             posElem->baseVertexPointerToElement(vertex, &pReal);
00281             vertices[j].x = *pReal++;
00282             vertices[j].y = *pReal++;
00283             vertices[j].z = *pReal++;
00284         }
00285         vBuffer->unlock();
00286         size_t index_offset = 0;
00287         bool use32bitindexes = (iBuffer->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
00288 
00289         unsigned long* pLong = static_cast<unsigned long*>(iBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00290         unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
00291 
00292         if (use32bitindexes)
00293         {
00294             for (size_t k = 0; k < triCount*3; ++k)
00295             {
00296                 indices[index_offset++] = pLong[k];
00297             }
00298         }
00299         else
00300         {
00301             for (size_t k = 0; k < triCount*3; ++k)
00302             {
00303                 indices[index_offset++] = static_cast<unsigned long>(pShort[k]);
00304             }
00305         }
00306         iBuffer->unlock();
00307 
00308         btTriangleMesh* trimesh = new btTriangleMesh(use32bitindexes);
00309         btVector3 vert0, vert1, vert2;
00310         int i=0;
00311         for (unsigned int y=0; y<triCount; y++)
00312         {
00313             vert0.setValue(vertices[indices[i]].x, vertices[indices[i]].y, vertices[indices[i]].z);
00314             vert1.setValue(vertices[indices[i+1]].x, vertices[indices[i+1]].y, vertices[indices[i+1]].z);
00315             vert2.setValue(vertices[indices[i+2]].x, vertices[indices[i+2]].y, vertices[indices[i+2]].z);
00316             trimesh->addTriangle(vert0, vert1, vert2);
00317             i+=3;
00318         }
00319         delete[] vertices;

00323         {

00328             OgreManager->destroyEntity(GraphicsObject);
00329             GraphicsObject = NULL;
00330         }
