00146                 if(Comment != String::npos)
00147                     Data = Data.substr(0,Comment);

00151                 // file texturename ~groupname

00158                     size_t GroupSplit = Data.find_first_of('~');

00160                     {
00161                         TextureName = Data.substr(0,GroupSplit);
00162                         GroupName = Data.substr(GroupSplit+1);
00163                         StringTool::Trim(TextureName);
00164                         StringTool::Trim(GroupName);
00165                     }
00166                     this->TAID->TATexture = Ogre::TextureManager::getSingletonPtr()->getByName(Data,GroupName);
00167                     if(this->TAID->TATexture.isNull())

00169                         this->TAID->TATexture = Ogre::TextureManager::getSingletonPtr()->load(TextureName,GroupName,Ogre::TEX_TYPE_2D,0);
00170                     }
00171 
00172                     InverseTextureSize.X = 1.0 / this->TAID->TATexture->getWidth();
00173                     InverseTextureSize.Y = 1.0 / this->TAID->TATexture->getHeight();

00178                 {
00179                     WhitePixel = StringTool::ConvertToVector2(Data);

00187         {
00188             String Name, Data;
00189             Resource::TextSettingsFile::SettingsIterator It;
00190 
00191             std::vector<String> StrValues;
00192             Vector2 Offset;
00193             for(It = Config->begin() ; It != Config->end() ; ++It)
00194             {
00195                 Name = It->first;

00197 
00198                 size_t Comment = Data.find_first_of('#');
00199                 if(Comment != String::npos)
00200                     Data = Data.substr(0, Comment);
00201                 StringTool::ToLowerCase(Name);
00202 
00203                 // lineheight x y
00204                 // >  offset 0 80
00205                 if (Name == "offset")
00206                 {
00207                     Offset = StringTool::ConvertToVector2(Data);
00208                     continue;
00209                 }
00210                 // lineheight lineheight
00211                 // >  lineheight 15
00212                 else if (Name == "lineheight")
00213                 {
00214                     GlyphD->LineHeight = StringTool::ConvertToReal(Data);
00215                     continue;
00216                 }
00217                 // spacelength spacelength
00218                 // >  spacelength 3
00219                 else if (Name == "spacelength")
00220                 {
00221                     GlyphD->SpaceLength = StringTool::ConvertToReal(Data);
00222                     continue;
00223                 }
00224                 // spacelength spacelength
00225                 // >  spacelength 3
00226                 else if (Name == "baseline")
00227                 {
00228                     GlyphD->Baseline = StringTool::ConvertToReal(Data);
00229                     continue;
00230                 }
00231                 // monowidth width
00232                 // >  monowidth 3
00233                 else if (Name == "monowidth")
00234                 {
00235                     GlyphD->MonoWidth = StringTool::ConvertToReal(Data);
00236                     continue;
00237                 }
00238                 // range lower upper
00239                 // >  range 33 126
00240                 else if (Name == "range")
00241                 {
00242                     Vector2 Range = StringTool::ConvertToVector2(Data);
00243                     GlyphD->RangeBegin = (UInt32)Range.X;
00244                     GlyphD->RangeEnd = (UInt32)Range.Y;
00245                 }
00246                 // kerning kerning
00247                 // >  kerning -1
00248                 else if(Name == "letterspacing")
00249                 {
00250                     GlyphD->LetterSpacing = StringTool::ConvertToReal(Data);
00251                 }
00252             }
00253 
00254             for(UInt32 Index = GlyphD->RangeBegin ; Index <= GlyphD->RangeEnd ; Index++)
00255             {
00256                 Glyph* NewGlyph = new Glyph();
00257                 GlyphD->Glyphs.push_back(NewGlyph);
00258 
00259                 std::stringstream GlyphStr;
00260                 GlyphStr << "glyph_" << Index;
00261 
00262                 It = Config->find(GlyphStr.str());
00263                 if(It == Config->end())
00264                     continue;
00265 
00266                 StrValues = StringTool::Split((*It).second," ",5);
00267 
00268                 if(StrValues.size() < 4)
00269                 {
00270                     continue;
00271                 }
00272 
00273                 NewGlyph->UVLeft    = Offset.X + StringTool::ConvertToReal( StrValues[0] );
00274                 NewGlyph->UVTop     = Offset.Y + StringTool::ConvertToReal( StrValues[1] );
00275                 NewGlyph->UVWidth   = StringTool::ConvertToReal( StrValues[2] );
00276                 NewGlyph->UVHeight  = StringTool::ConvertToReal( StrValues[3] );
00277                 NewGlyph->UVRight   = NewGlyph->UVLeft + NewGlyph->UVWidth;
00278                 NewGlyph->UVBottom  = NewGlyph->UVTop + NewGlyph->UVHeight;
00279 
00280                 if(StrValues.size() == 5) NewGlyph->GlyphAdvance = StringTool::ConvertToReal( StrValues[4] );
00281                 else NewGlyph->GlyphAdvance = NewGlyph->UVWidth;
00282 
00283                 NewGlyph->GlyphChar = (GlyphD->Glyphs.size() - 1) + GlyphD->RangeBegin;
00284             }
00285         }

00487             for(std::map<UInt32, GlyphData*>::iterator gd_it = Glyphs.begin(); gd_it != Glyphs.end(); gd_it++)
00488             {
00489                 for(std::vector<Glyph*>::iterator it = (*gd_it).second->Glyphs.begin(); it != (*gd_it).second->Glyphs.end(); it++)
00490                 {
00491                     (*it)->UVLeft        -= TexelX;
00492                     (*it)->UVTop         -= TexelY;
00493                     (*it)->UVRight       += TexelX;
00494                     (*it)->UVBottom      += TexelY;
00495 
00496                     (*it)->UVLeft        *= InverseTextureSize.X;
00497                     (*it)->UVTop         *= InverseTextureSize.Y;
00498                     (*it)->UVRight       *= InverseTextureSize.X;

00502                     (*it)->AtlasCoords[QC_TopLeft].Y = (*it)->UVTop;
