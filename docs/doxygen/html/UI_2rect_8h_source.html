57  class MEZZ_LIB Rect
58  {
59  protected:
60  /// @brief Simple check to see if a point is within a given range.
61  /// @param Value The value to compare against the range.
62  /// @param Min The minimum value of the range.
63  /// @param Max The maximum value of the range.
64  /// @return Returns true if the provided value is within the provided range, false otherwise.
65  inline Boole ValueInRange(const Real& Value, const Real& Min, const Real& Max) const
66  {
67  return (Value >= Min) && (Value <= Max);
68  }
69  public:
70  /// @brief Vector2 representing the top-left position of the rect.
71  Vector2 Position;
72  /// @brief Vector2 representing the width and height of the rect.
73  Vector2 Size;
74  /// @brief Boole representing whether this rect is represented in relative units or absolute units(pixels).
75  Boole Relative;

77  /// @brief Less Detailed Real Constructor.
78  /// @param PosX The position of this rect on the X axis.
79  /// @param PosY The position of this rect on the Y axis.
80  /// @param SizeX The size of this rect on the X axis.
81  /// @param SizeY The size of this rect on the Y axis.
82  Rect(const Real& PosX, const Real& PosY, const Real& SizeX, const Real& SizeY)
83  {
84  this->Position.SetValues(PosX,PosY);
85  this->Size.SetValues(SizeX,SizeY);
86  this->Relative = false;
87  }
88  /// @brief Real Constructor.
89  /// @param PosX The position of this rect on the X axis.
90  /// @param PosY The position of this rect on the Y axis.
91  /// @param SizeX The size of this rect on the X axis.
92  /// @param SizeY The size of this rect on the Y axis.
93  Rect(const Real& PosX, const Real& PosY, const Real& SizeX, const Real& SizeY, Boole Relative)
94  {
95  this->Position.SetValues(PosX,PosY);
96  this->Size.SetValues(SizeX,SizeY);
97  this->Relative = Relative;
98  }
99  /// @brief Less Detailed Vector2 Constructor.
100  /// @details Sets all the data of the class.
101  /// @param Position The position of the Renderable's Rect.
102  /// @param Size The size of the Renderable's Rect.
103  Rect(const Vector2& Position, const Vector2& Size)
104  {
105  this->Position = Position;
106  this->Size = Size;
107  this->Relative = false;
108  }

110  /// @details Sets all the data of the class.
111  /// @param Position The position of the Renderable's Rect.

116  this->Position = Position;

118  this->Relative = Relative;

148  /// @brief Gets the value of the top edge of this rect.
149  /// @return Returns a Real that is the point on the vertical axis this rect's top edge is located.
150  inline Real GetTopEdge() const
151  {
152  return this->Position.Y;
153  }
154  /// @brief Gets the value of the bottom edge of this rect.
155  /// @return Returns a Real that is the point on the vertical axis this rect's bottom edge is located.
156  inline Real GetBottomEdge() const
157  {
158  return this->Position.Y + this->Size.Y;
159  }
160  /// @brief Gets the value of the left edge of this rect.
161  /// @return Returns a Real that is the point on the horizontal axis this rect's left edge is located.
162  inline Real GetLeftEdge() const
163  {
164  return this->Position.X;
165  }
166  /// @brief Gets the value of the right edge of this rect.
167  /// @return Returns a Real that is the point on the horizontal axis this rect's right edge is located.
168  inline Real GetRightEdge() const
169  {
170  return this->Position.X + this->Size.X;
171  }
172 
173  /// @brief Gets the value of the horizontal center of this rect.
174  /// @return Returns a Real that is the point on the horizontal axis that is this rect's center.
175  inline Real GetHorizontalCenter() const
176  {
177  return this->Position.X + (this->Size.X * 0.5);
178  }
179  /// @brief Gets the value of the vertical center of this rect.
180  /// @return Returns a Real that is the point on the vertical axis that is this rect's center.
181  inline Real GetVerticalCenter() const
182  {
183  return this->Position.Y + (this->Size.Y * 0.5);
184  }
185  /// @brief Gets the coordinates to the center of this rect.
186  /// @return Returns a vector2 containing the central point of this rect.
187  inline Vector2 GetRectCenter() const
188  {
189  return Vector2(this->GetHorizontalCenter(),this->GetVerticalCenter());
190  }
191 
192  /// @brief Checks to see if another Rect is overlapping with this one.
193  /// @param OtherRect The other rect to compare against.
194  /// @return Returns true if these rects overlap with each other, false otherwise.
195  inline Boole CheckOverlap(const Rect& OtherRect) const
196  {
197  Boole XOverlap = ValueInRange(this->Position.X, OtherRect.Position.X, OtherRect.Position.X + OtherRect.Size.X) ||
198  ValueInRange(OtherRect.Position.X, this->Position.X, this->Position.X + this->Size.X);
199 
200  Boole YOverlap = ValueInRange(this->Position.Y, OtherRect.Position.Y, OtherRect.Position.Y + OtherRect.Size.Y) ||
201  ValueInRange(OtherRect.Position.Y, this->Position.Y, this->Position.Y + this->Size.Y);
202 
203  return (XOverlap && YOverlap);
204  }
205  /// @brief Checks to see if a point in 2D space is inside this rect.
206  /// @param Point The point in 2D space to check.
207  /// @return Returns true if the provided point is within this rect, false otherwise.
208  inline Boole IsInside(const Vector2& Point) const
209  {
210  return ( ValueInRange(Point.X,this->Position.X,this->Position.X + this->Size.X) &&
211  ValueInRange(Point.Y,this->Position.Y,this->Position.Y + this->Size.Y) );
212  }
213  /// @brief Gets whether or point on the X axis is within the limits of this rect or not.
214  /// @param Position The point on the X axis.
215  /// @return Returns true in the provided position is within this rect's limits, false otherwise.
216  inline Boole IsWithinWidth(const Real& Position) const
217  {
218  return ValueInRange(Position,this->Position.X,this->Position.X + this->Size.X);
219  }
220  /// @brief Gets whether or point on the Y axis is within the limits of this rect or not.
221  /// @param Position The point on the Y axis.
222  /// @return Returns true in the provided position is within this rect's limits, false otherwise.
223  inline Boole IsWithinHeight(const Real& Position) const
224  {
225  return ValueInRange(Position,this->Position.Y,this->Position.Y + this->Size.Y);
226  }

231  /// @brief Scales this rect.
232  /// @note This will scale the rect based on it's center, not it's top-left position. If you want to instead preserve the position of the Rect then apply the scaling manually.
233  /// @param Scaling A Vector2 representing the X and Y scaling to be applied to this rect.
234  /// @return Returns a reference to this.
235  inline Rect& ApplyScaling(const Vector2& Scaling)
236  {
237  Vector2 PrevCenter = this->GetRectCenter();
238  this->Size *= Scaling;
239 
240  this->Position.X = PrevCenter.X - ( this->Size.X * 0.5 );
241  this->Position.Y = PrevCenter.Y - ( this->Size.Y * 0.5 );
242 
243  return *this;
244  }

277  /// @brief Convert this class to an XML::Node ready for serialization.
278  /// @param ParentNode The point in the XML hierarchy that all this renderable should be appended to.
279  void ProtoSerialize(XML::Node& ParentNode) const
280  {
281  XML::Node RectNode = ParentNode.AppendChild( Rect::GetSerializableName() );
282 
283  if( RectNode.AppendAttribute("Version").SetValue("1") &&
284  RectNode.AppendAttribute("PositionX").SetValue(this->Position.X) &&
285  RectNode.AppendAttribute("PositionY").SetValue(this->Position.Y) &&
286  RectNode.AppendAttribute("SizeX").SetValue(this->Size.X) &&
287  RectNode.AppendAttribute("SizeY").SetValue(this->Size.Y) )
288  {
289  return;
290  }else{
291  SerializeError("Create XML Attribute Values",Rect::GetSerializableName(),true);
292  }
293  }
294  /// @brief Take the data stored in an XML Node and overwrite this object with it.
295  /// @param SelfRoot An XML::Node containing the data to populate this class with.
296  void ProtoDeSerialize(const XML::Node& SelfRoot)
297  {
298  XML::Attribute CurrAttrib;
299  XML::Node RectNode = SelfRoot.GetChild( Rect::GetSerializableName() );
300 
301  if( !RectNode.Empty() ) {
302  if(RectNode.GetAttribute("Version").AsInt() == 1) {
303  CurrAttrib = RectNode.GetAttribute("PositionX");
304  if( !CurrAttrib.Empty() )
305  this->Position.X = CurrAttrib.AsReal();
306 
307  CurrAttrib = RectNode.GetAttribute("PositionY");
308  if( !CurrAttrib.Empty() )
309  this->Position.Y = CurrAttrib.AsReal();
310 
311  CurrAttrib = RectNode.GetAttribute("SizeX");
312  if( !CurrAttrib.Empty() )
313  this->Size.X = CurrAttrib.AsReal();
314 
315  CurrAttrib = RectNode.GetAttribute("SizeY");
316  if( !CurrAttrib.Empty() )
317  this->Size.X = CurrAttrib.AsReal();
318  }else{
319  MEZZ_EXCEPTION(ExceptionBase::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + Rect::GetSerializableName() + ": Not Version 1.");
320  }
321  }else{
322  MEZZ_EXCEPTION(ExceptionBase::II_IDENTITY_NOT_FOUND_EXCEPTION,Rect::GetSerializableName() + " was not found in the provided XML node, which was expected.");
323  }
324  }
