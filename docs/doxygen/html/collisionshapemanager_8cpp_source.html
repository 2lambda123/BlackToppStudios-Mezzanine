00086     btTriangleMesh* CollisionShapeManager::CreateBulletTrimesh(Mesh* ObjectMesh, bool UseAllSubmeshes)
00087     {
00088         Ogre::MeshPtr myMesh = ObjectMesh->GetOgreMesh();
00089         Ogre::SubMesh* subMesh = NULL;
00090         Ogre::IndexData*  IndexData = NULL;
00091         Ogre::VertexData* VertexData = NULL;
00092         bool use32bitindexes = false;
00093         unsigned int triCount = 0;
00094         unsigned int vCount = 0;
00095         unsigned int iCount = 0;
00096         Whole VertPrevSize = 0;
00097         Whole IndiPrevSize = 0;
00098         Ogre::Vector3* vertices = NULL;

00100         bool SharedVerts = myMesh->getSubMesh(0)->useSharedVertices;
00101 
00102         if(UseAllSubmeshes)
00103         {
00104             for( Whole X = 0 ; X < myMesh->getNumSubMeshes() ; X++ )
00105             {
00106                 vCount += SharedVerts ? myMesh->sharedVertexData->vertexCount : myMesh->getSubMesh(X)->vertexData->vertexCount;
00107                 iCount += myMesh->getSubMesh(X)->indexData->indexCount;
00108             }
00109         }else{
00110             vCount += SharedVerts ? myMesh->sharedVertexData->vertexCount : myMesh->getSubMesh(0)->vertexData->vertexCount;
00111             iCount += myMesh->getSubMesh(0)->indexData->indexCount;
00112         }
00113 
00114         vertices = new Ogre::Vector3[vCount];
00115         indices  = new unsigned long[iCount];
00116 
00117         // Get the submesh and associated data
00118         for( unsigned short int SubMeshIndex = 0 ; SubMeshIndex < myMesh->getNumSubMeshes() ; SubMeshIndex++ )
00119         {
00120             if(!UseAllSubmeshes && (SubMeshIndex > 0))
00121                 break;
00122             if(SharedVerts && (SubMeshIndex > 0))
00123                 break;
00124 
00125             subMesh = myMesh->getSubMesh(SubMeshIndex);
00126             IndexData = subMesh->indexData;
00127             VertexData = SharedVerts ? myMesh->sharedVertexData : myMesh->getSubMesh(SubMeshIndex)->vertexData;
00128 
00129             // Get the position element
00130             const Ogre::VertexElement* posElem = VertexData->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
00131             // Get a pointer to the vertex buffer
00132             Ogre::HardwareVertexBufferSharedPtr vBuffer = VertexData->vertexBufferBinding->getBuffer(posElem->getSource());
00133             // Get a pointer to the index buffer
00134             Ogre::HardwareIndexBufferSharedPtr iBuffer = IndexData->indexBuffer;
00135             // Get the number of triangles
00136             triCount+=(IndexData->indexCount/3);
00137 
00138             // Lock the vertex buffer (READ ONLY)
00139             unsigned char* vertex = static_cast<unsigned char*>(vBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00140             float* pReal = NULL;
00141             for (size_t j = 0; j < VertexData->vertexCount; ++j, vertex += vBuffer->getVertexSize() )
00142             {
00143                 posElem->baseVertexPointerToElement(vertex, &pReal);
00144                 Ogre::Vector3 pt(pReal[0], pReal[1], pReal[2]);
00145                 vertices[j + VertPrevSize] = pt;
00146             }
00147             vBuffer->unlock();
00148             size_t index_offset = 0;
00149             use32bitindexes = (iBuffer->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
00150 
00151             // Lock the index buffer (READ ONLY)
00152             unsigned long* pLong = static_cast<unsigned long*>(iBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00153             unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
00154 
00155             if (use32bitindexes)
00156             {
00157                 for (size_t k = 0; k < triCount*3; ++k)
00158                 {
00159                     indices[index_offset+IndiPrevSize] = pLong[k];
00160                     index_offset++;
00161                 }
00162             }
00163             else
00164             {
00165                 for (size_t k = 0; k < triCount*3; ++k)
00166                 {
00167                     indices[index_offset+IndiPrevSize] = static_cast<unsigned long>(pShort[k]);
00168                     index_offset++;
00169                 }
00170             }
00171             iBuffer->unlock();
00172 
00173             VertPrevSize+=VertexData->vertexCount;
00174             IndiPrevSize+=IndexData->indexCount;
00175         }
00176 
00177         // We now have vertices and indices ready to go
00178 
00179         // The Bullet triangle mesh
00180         btTriangleMesh* trimesh = new btTriangleMesh(use32bitindexes);
00181 
00182         // Setup the tri mesh
00183         btVector3 vert0, vert1, vert2;
00184         int i=0;
00185 
00186         // For every triangle
00187         for (unsigned int y=0; y<triCount; y++)
00188         {
00189             // Set each vertex
00190             vert0.setValue(vertices[indices[i]].x, vertices[indices[i]].y, vertices[indices[i]].z);
00191             vert1.setValue(vertices[indices[i+1]].x, vertices[indices[i+1]].y, vertices[indices[i+1]].z);
00192             vert2.setValue(vertices[indices[i+2]].x, vertices[indices[i+2]].y, vertices[indices[i+2]].z);
00193 
00194             // Add it into the trimesh
00195             trimesh->addTriangle(vert0, vert1, vert2);
00196 
00197             // Increase index count
00198             i+=3;
00199         }
00200         delete[] vertices;
00201         delete[] indices;
00202 
00203         return trimesh;
00204     }
