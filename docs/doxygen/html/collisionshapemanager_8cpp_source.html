00084     btTriangleMesh* CollisionShapeManager::CreateBulletTrimesh(Mesh* ObjectMesh, bool UseAllSubmeshes)
00085     {
00086         Ogre::MeshPtr myMesh = ObjectMesh->GetOgreMesh();
00087         Ogre::SubMesh* subMesh = NULL;
00088         Ogre::IndexData*  IndexData = NULL;
00089         Ogre::VertexData* VertexData = NULL;
00090         bool use32bitindexes = false;
00091         unsigned int triCount = 0;
00092         unsigned int vCount = 0;
00093         unsigned int iCount = 0;
00094         Whole VertPrevSize = 0;
00095         Whole IndiPrevSize = 0;
00096         Ogre::Vector3* vertices = NULL;
00097         unsigned long* indices = NULL;
00098         bool SharedVerts = myMesh->getSubMesh(0)->useSharedVertices;
00099 
00100         if(UseAllSubmeshes)
00101         {
00102             for( Whole X = 0 ; X < myMesh->getNumSubMeshes() ; X++ )
00103             {
00104                 vCount += SharedVerts ? myMesh->sharedVertexData->vertexCount : myMesh->getSubMesh(X)->vertexData->vertexCount;
00105                 iCount += myMesh->getSubMesh(X)->indexData->indexCount;
00106             }
00107         }else{
00108             vCount += SharedVerts ? myMesh->sharedVertexData->vertexCount : myMesh->getSubMesh(0)->vertexData->vertexCount;
00109             iCount += myMesh->getSubMesh(0)->indexData->indexCount;
00110         }
00111 
00112         vertices = new Ogre::Vector3[vCount];
00113         indices  = new unsigned long[iCount];
00114 
00115         // Get the submesh and associated data
00116         for( unsigned short int SubMeshIndex = 0 ; SubMeshIndex < myMesh->getNumSubMeshes() ; SubMeshIndex++ )
00117         {
00118             if(!UseAllSubmeshes && (SubMeshIndex > 0))
00119                 break;
00120             if(SharedVerts && (SubMeshIndex > 0))
00121                 break;
00122 
00123             subMesh = myMesh->getSubMesh(SubMeshIndex);
00124             IndexData = subMesh->indexData;
00125             VertexData = SharedVerts ? myMesh->sharedVertexData : myMesh->getSubMesh(SubMeshIndex)->vertexData;
00126 
00127             // Get the position element
00128             const Ogre::VertexElement* posElem = VertexData->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
00129             // Get a pointer to the vertex buffer
00130             Ogre::HardwareVertexBufferSharedPtr vBuffer = VertexData->vertexBufferBinding->getBuffer(posElem->getSource());
00131             // Get a pointer to the index buffer
00132             Ogre::HardwareIndexBufferSharedPtr iBuffer = IndexData->indexBuffer;
00133             // Get the number of triangles
00134             triCount+=(IndexData->indexCount/3);
00135 
00136             // Lock the vertex buffer (READ ONLY)
00137             unsigned char* vertex = static_cast<unsigned char*>(vBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00138             float* pReal = NULL;
00139             for (size_t j = 0; j < VertexData->vertexCount; ++j, vertex += vBuffer->getVertexSize() )
00140             {
00141                 posElem->baseVertexPointerToElement(vertex, &pReal);
00142                 Ogre::Vector3 pt(pReal[0], pReal[1], pReal[2]);
00143                 vertices[j + VertPrevSize] = pt;
00144             }
00145             vBuffer->unlock();
00146             size_t index_offset = 0;
00147             use32bitindexes = (iBuffer->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
00148 
00149             // Lock the index buffer (READ ONLY)
00150             unsigned long* pLong = static_cast<unsigned long*>(iBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00151             unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
00152 
00153             if (use32bitindexes)
00154             {
00155                 for (size_t k = 0; k < triCount*3; ++k)
00156                 {
00157                     indices[index_offset+IndiPrevSize] = pLong[k];
00158                     index_offset++;
00159                 }
00160             }
00161             else
00162             {
00163                 for (size_t k = 0; k < triCount*3; ++k)
00164                 {
00165                     indices[index_offset+IndiPrevSize] = static_cast<unsigned long>(pShort[k]);
00166                     index_offset++;
00167                 }
00168             }
00169             iBuffer->unlock();
00170 
00171             VertPrevSize+=VertexData->vertexCount;
00172             IndiPrevSize+=IndexData->indexCount;
00173         }
00174 
00175         // We now have vertices and indices ready to go
00176 
00177         // The Bullet triangle mesh
00178         btTriangleMesh* trimesh = new btTriangleMesh(use32bitindexes);
00179 
00180         // Setup the tri mesh
00181         btVector3 vert0, vert1, vert2;
00182         int i=0;
00183 
00184         // For every triangle
00185         for (unsigned int y=0; y<triCount; y++)
00186         {
00187             // Set each vertex
00188             vert0.setValue(vertices[indices[i]].x, vertices[indices[i]].y, vertices[indices[i]].z);
00189             vert1.setValue(vertices[indices[i+1]].x, vertices[indices[i+1]].y, vertices[indices[i+1]].z);
00190             vert2.setValue(vertices[indices[i+2]].x, vertices[indices[i+2]].y, vertices[indices[i+2]].z);
00191 
00192             // Add it into the trimesh
00193             trimesh->addTriangle(vert0, vert1, vert2);
00194 
00195             // Increase index count
00196             i+=3;
00197         }
00198         delete[] vertices;
00199         delete[] indices;
00200 
00201         return trimesh;
00202     }
