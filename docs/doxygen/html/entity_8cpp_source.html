00068             SceneManager* Manager;

00070             /// @internal
00071             /// @brief Creates all the necessary data structures for the Entity.
00072             EntityInternalData(const String& Name, const String& MeshName, const String& Group, SceneManager* manager)
00073             {
00074                 this->Manager = manager;
00075                 Ogre::SceneManager* OgreManager = Manager->GetGraphicsWorldPointer();
00076                 OgreEntity = OgreManager->createEntity(Name,MeshName,Group);
00077                 OgreNode = OgreManager->createSceneNode();
00078                 OgreManager->getRootSceneNode()->addChild(OgreNode);
00079             }
00080 
00081             /// @internal
00082             /// @brief Cleans up after the Entity.
00083             ~EntityInternalData()
00084             {
00085                 Manager->GetGraphicsWorldPointer()->destroyEntity(OgreEntity);
00086                 Manager->GetGraphicsWorldPointer()->destroySceneNode(OgreNode);
00087             }
00088         };//EntityInternalData
