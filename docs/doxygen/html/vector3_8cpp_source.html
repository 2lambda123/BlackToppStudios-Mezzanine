00165     StandardAxis Vector3::IsStandardUnitAxis() const
00166     {
00167         if (1.0==this->X && 0.0==this->Y && 0.0==this->Z)
00168         {
00169             return Axis_X;
00170         } else if (0.0==this->X) {                                         // Not Unit_X
00171             if (1.0==this->Y && 0.0==this->Z)
00172             {
00173                 return Axis_Y;
00174             } else if (0.0==this->Y && 1.0==this->Z) {                                     // Not Unit_Y so hopefully it is Z
00175                 return Axis_Y;
00176             }
00177         }
00178         throw(Exception("Cannot convert Vector3 to StandardAxis in Vector3::IsStandardUnitAxis, Vector3 may not be Axis Aligned or may not be Unit Length."));

00432     Real Vector3::Distance(const Vector3 &OtherVec) const
00433     {
00434         float Xdelta = Ogre::Math::Abs(this->X - OtherVec.X);
00435         float Ydelta = Ogre::Math::Abs(this->Y - OtherVec.Y);
00436         float Zdelta = Ogre::Math::Abs(this->Z - OtherVec.Z);
00437         return Ogre::Math::Sqrt( Xdelta*Xdelta + Ydelta*Ydelta + Zdelta*Zdelta );
00438     }
