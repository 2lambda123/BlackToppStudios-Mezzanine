55  void ButtonDevice::UpdateButtonTransitions()
56  {
57  for( std::vector<Whole>::iterator It = this->TransitioningIndexes.begin() ; It != this->TransitioningIndexes.end() ; ++It )
58  {
59  Input::ButtonState State = this->Buttons.at( (*It) );
60  if( Input::BUTTON_PRESSING == State )
61  this->Buttons.at( (*It) ) = Input::BUTTON_DOWN;
62  else if( Input::BUTTON_LIFTING == State )
63  this->Buttons.at( (*It) ) = Input::BUTTON_UP;
64  }
65  this->TransitioningIndexes.clear();
66  }
