00110         void LineList::_Redraw()
00111         {
00112             if(Dirty == false)
00113                 return;
00114             RenderVertices.clear();
00115             if(!Visible)
00116             {
00117                 Dirty = false;
00118                 return;
00119             }
00120             if(Positions.size() < 2)
00121                 return;
00122 
00123             VertexData Temp;
00124             Real HalfThickness = Thickness * 0.5;
00125             Vector2 PerpNorm, LastLeft, LastRight, ThisLeft, ThisRight, UV = Parent->GetSolidUV(PriAtlas);
00126             size_t Index = 1;
00127             for(  ; Index < Positions.size() ; Index++ )
00128             {
00129                 PerpNorm  = (Positions[Index] - Positions[Index - 1]).Perpendicular().Normalize();
00130                 LastLeft  = Positions[Index -1 ] - PerpNorm * HalfThickness;
00131                 LastRight = Positions[Index -1 ] + PerpNorm * HalfThickness;
00132                 ThisLeft  = Positions[Index] - PerpNorm * HalfThickness;
00133                 ThisRight = Positions[Index] + PerpNorm * HalfThickness;
00134 
00135                 // Triangle A
00136                 PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas);      // Left/Bottom
00137                 PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas);        // Right/Top
00138                 PushVertex(RenderVertices,Temp,LastLeft.X,LastLeft.Y,UV,Colour,PriAtlas);        // Left/Top
00139                 // Triangle B
00140                 PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas);      // Left/Bottom
00141                 PushVertex(RenderVertices,Temp,ThisRight.X,ThisRight.Y,UV,Colour,PriAtlas);      // Right/Bottom
00142                 PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas);        // Right/Top
00143             }
00144 
00145             if(IsClosed)
00146             {
00147                 Index = Positions.size() - 1;
00148                 PerpNorm  = (Positions[0] - Positions[Index]).Perpendicular().Normalize();
00149                 LastLeft  = Positions[Index] - PerpNorm * HalfThickness;
00150                 LastRight = Positions[Index] + PerpNorm * HalfThickness;
00151                 ThisLeft  = Positions[0] - PerpNorm * HalfThickness;
00152                 ThisRight = Positions[0] + PerpNorm * HalfThickness;
00153 
00154                 // Triangle A
00155                 PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas);       // Left/Bottom
00156                 PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas);         // Right/Top
00157                 PushVertex(RenderVertices,Temp,LastLeft.X,LastLeft.Y,UV,Colour,PriAtlas);         // Left/Top
00158                 // Triangle B
00159                 PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas);       // Left/Bottom
00160                 PushVertex(RenderVertices,Temp,ThisRight.X,ThisRight.Y,UV,Colour,PriAtlas);       // Right/Bottom
00161                 PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas);         // Right/Top
00162             }
00163         }
