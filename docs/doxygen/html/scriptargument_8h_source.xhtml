54  /// @brief The interface for a script argument
55  /// @details All the members that all script for all languages must implement.
56  /// @n @n
57  /// These are created to provide data to scripts. This is intended to primarily
58  /// to act as an abstraction layer between game data and data inside scripts,
59  /// but currently just transports primitives into and out of scripts without needing to care
60  /// about any underlying types. This limits what operations can be done with
61  /// this, because it is intended to provide an abstraction that can be used
62  /// to implement specific algorithms for languages that require them.
63  /// @todo Define operators for the iScriptArgument at least for assignment
64  ///////////////////////////////////////

66  {
67  public:
68  /// @brief Overidable Deconstructor

70  {}
71

74  virtual String GetString() const = 0;
75 
76  /// @brief Get the Argument as a Whole
77  /// @return The argument value lexographically converted as a @ref Whole
78  virtual Whole GetWhole() const = 0;

109  /// @brief A generic implementation of a ScriptArgument that is suitable for primitive types in most situations
110  template <class T>
111  class MEZZ_LIB ScriptArgumentGeneric : public iScriptArgument
112  {
113  protected:
114  /// @brief The actual data.
115  /// @warning To prevent Slicing don't store the parent class in a container.
116  /// This practically guarantees that derived class will have different sizes
117  /// and overflow container boundaries.
118  T Datum;
119 
120  public:
121  /// @brief To make working with this easier.
122  typedef T Type;
123 
124  /// @brief Create an initialized Argument
125  /// @param InitialValue The value to initialize the Argument with.
126  /// @note Intentionally not explicit, this allow for passing convertable types directly to functions.
127  ScriptArgumentGeneric(T InitialValue) :
128  Datum(InitialValue)
129  {}
130 
131  /// @brief Overloadable Deconstructor
132  virtual ~ScriptArgumentGeneric()
133  {}
134 
135  /// @brief Get the Argument as a String, slow default implementation.
136  /// @return The argument value lexographically converted as a @ref String
137  virtual String GetString() const
138  { return ToString(Datum); }
139 
140  /// @brief Get the Argument as a Whole, slow default implementation.
141  /// @return The argument value lexographically converted as a @ref Whole

144

146  /// @return The argument value lexographically converted as an @ref Integer
147  virtual Integer GetInteger() const

149 
150  /// @brief Get the Argument as a Real, slow default implementation.
151  /// @return The argument value lexographically converted as an @ref Real
152  virtual Real GetReal() const
153  { return ToReal(Datum); }
154 
155  /// @brief Get the Argument as a Bool
156  /// @return The argument value lexographically converted as an @ref Bool
157  virtual Bool GetBool() const
158  { return ToBool(Datum); }
159 
160  /// @brief Provide an overload point to change assignment that operators will use.
161  /// @param NewValue The new value for this.
162  virtual void SetValue(T NewValue)
163  { Datum=NewValue; }
164 
165  /// @brief Get the raw primitive to value.
166  /// @return The internal value that meaningful operation can be performed on.
167  virtual T GetValue() const
168  { return Datum; }
169 
170  /// @brief Get data about this having an unknown type
171  /// @return This will return an Integer containing GenericUnknown. If needed proper specialization of this can be made for any class, see the ScriptArgumentGeneric<Mezzanine::Scripting::Lua::Lua51Nil>
172  virtual Integer GetTypeData() const
173  { return GenericUnknown; }
