00075         void Menu::Update(bool Force)
00076         {
00077             if(!Force)
00078                 SubWidgetUpdate();
00079             MetaCode::ButtonState State = InputQueryTool::GetMouseButtonState(1);
00080             if(HoveredSubWidget)
00081             {
00082                 Button* button = HoveredSubWidget->GetHoveredButton();
00083                 if(button && State == MetaCode::BUTTON_LIFTING)
00084                 {
00085                     UI::MenuWindow* MenWin = static_cast<UI::MenuWindow*>(HoveredSubWidget);
00086                     UI::MenuWindow* ChildMenWin = MenWin->GetWindowOfAccessButton(button);
00087                     if(ChildMenWin)
00088                     {
00089                         if(!ChildMenWin->IsVisible())
00090                         {
00091                             if(MenuStack.back()->GetAutoHide())
00092                                 MenuStack.back()->Hide();
00093                             MenuStack.push_back(ChildMenWin);
00094                             ChildMenWin->Show();
00095                             return;
00096                         }else{
00097                             RollMenuBackToWindow(MenWin);
00098                         }
00099                     }
00100                     else if(button == MenWin->GetBackButton())
00101                     {
00102                         MenWin->Hide();
00103                         MenuStack.pop_back();
00104                         if(!MenuStack.back()->IsVisible())
00105                             MenuStack.back()->Show();
00106                         return;
00107                     }
00108                 }
00109                 if(MetaCode::BUTTON_PRESSING == State)
00110                 {
00111                     SubWidgetFocus = HoveredSubWidget;
00112                 }
00113             }
00114             if(SubWidgetFocus && (SubWidgetFocus != HoveredSubWidget))
00115             {
00116                 SubWidgetFocusUpdate(true);
00117             }
00118             else if(MetaCode::BUTTON_DOWN == State && Force)
00119             {
00120                 SubWidgetFocusUpdate(Force);
00121             }
00122             if(MetaCode::BUTTON_LIFTING == State)
00123             {
00124                 SubWidgetFocus = NULL;
00125             }
00126         }
