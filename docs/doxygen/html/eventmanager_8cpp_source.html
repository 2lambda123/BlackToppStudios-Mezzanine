00078         /// This will always drop quit events, and store that information for later use.
00079         /// @param event This is the event SDL expects use to filters, To get real data from this we setup it up so that if the event is a null pointer the function will return data about quit messages
00080         /// @param userdata Nothing, a dummy argument that could be used by SDL
00081         /// @warning Do not use this. It can only cause problems. This is for SDL, the user input subsystem, to filter certain events.
00082         /// @return This will always return either 0 or 1 to SDL. 0 if it should drop the event, which it does to all SDL_quit events, 1 if the event should be allowed, which it does to all events which are not SDL_quit events. If a null pointer was passed, then this will return 4 if it dropped an SDL_Quit, and 2 if it has not droppped an SDL_quit.
00083         //int MezzSDLFilter(void *userdata, const SDL_Event *event,  );
00084         //If this is passed an event that points to 0 it will function as a method to tell us if an SDL_QUIT message has been thrown
00085         //this will return 2 if it has not seen an SDL_quit, and a 4 if it has
00086         int MezzSDLFilter(void *userdata, SDL_Event *event )
00087         {
00088             //We need to protect this thing with a Mutex, SDL 1.3 says this filter could run in different thread.
00089             static bool DroppedQuit=false;
00090 
00091             if(event!=0)                //if this is a real event
00092             {
00093                 if ( event->type == SDL_QUIT )
00094                 {
00095                     DroppedQuit=true;   //Drop all quit events, and track that we dropped them
00096                     return 0;
00097                 }
00098                 return 1;
00099             }else{
00100                 if(DroppedQuit)         //4 if we need to make a quit event
00101                 {
00102                     DroppedQuit=false;  //Reset this so we don't add more in the future by accident
00103                     return 4;
00104                 }
00105                 return 2;
00106             }
