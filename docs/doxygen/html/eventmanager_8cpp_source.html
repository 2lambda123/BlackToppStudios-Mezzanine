00076         /// @internal
00077         /// @brief SDL uses this to filter events it presents to applications
00078         /// @details This is used used to filter out SQL_quit messages, and generate appropriate messages for the game developer to use.
00079         /// This will always drop quit events, and store that information for later use.
00080         /// \n
00081         /// If this is passed an event that points to 0 it will function as a method to tell us if an SDL_QUIT message has been thrown
00082         /// this will return 2 if it has not seen an SDL_quit, and a 4 if it has
00083         /// @param event This is the event SDL expects use to filters, To get real data from this we setup it up so that if the event is a null pointer the function will return data about quit messages
00084         /// @param userdata Nothing, a dummy argument that could be used by SDL
00085         /// @warning Do not use this. It can only cause problems. This is for SDL, the user input subsystem, to filter certain events.
00086         /// @return This will always return either 0 or 1 to SDL. 0 if it should drop the event, which it does to all SDL_quit events, 1 if the event should be allowed, which it does to all events which are not SDL_quit events. If a null pointer was passed, then this will return 4 if it dropped an SDL_Quit, and 2 if it has not droppped an SDL_quit.
00087         int MezzSDLFilter(void *userdata, SDL_Event *event )
00088         {
00089             //We need to protect this thing with a Mutex, SDL 1.3 says this filter could run in different thread.
00090             static bool DroppedQuit=false;
00091 
00092             if(event!=0)                //if this is a real event
00093             {
00094                 if ( event->type == SDL_QUIT )
00095                 {
00096                     DroppedQuit=true;   //Drop all quit events, and track that we dropped them
00097                     return 0;
00098                 }
00099                 return 1;
00100             }else{
00101                 if(DroppedQuit)         //4 if we need to make a quit event
00102                 {
00103                     DroppedQuit=false;  //Reset this so we don't add more in the future by accident
00104                     return 4;
00105                 }
00106                 return 2;
00107             }
00108         }
