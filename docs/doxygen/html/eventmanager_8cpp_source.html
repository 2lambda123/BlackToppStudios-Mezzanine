00076         /// @brief SDL uses this to filter events it presents to applications
00077         /// @details This is used used to filter out SQL_quit messages, and generate appropriate messages for the game developer to use.
00078         /// This will always drop quit events, and store that information for later use.
00079         /// @param event This is the event SDL expects use to filters, To get real data from this we setup it up so that if the event is a null pointer the function will return data about quit messages
00080         /// @param userdata Nothing, a dummy argument that could be used by SDL
00081         /// @warning Do not use this. It can only cause problems. This is for SDL, the user input subsystem, to filter certain events.
00082         /// @return This will always return either 0 or 1 to SDL. 0 if it should drop the event, which it does to all SDL_quit events, 1 if the event should be allowed, which it does to all events which are not SDL_quit events. If a null pointer was passed, then this will return 4 if it dropped an SDL_Quit, and 2 if it has not droppped an SDL_quit.
00083         //int PhysSDLFilter(void *userdata, const SDL_Event *event,  );
00084         //If this is passed an event that points to 0 it will function as a method to tell us if an SDL_QUIT message has been thrown
00085         //this will return 2 if it has not seen an SDL_quit, and a 4 if it has
00086         int PhysSDLFilter(void *userdata, SDL_Event *event )
00087         {
00088             //We need to protect this thing with a Mutex, SDL 1.3 says this filter could run in different thread.
00089             static bool DroppedQuit=false;
00090 
00091             if(event!=0)                //if this is a real event
00092             {
00093                 if ( event->type == SDL_QUIT )
00094                 {
00095                     DroppedQuit=true;   //Drop all quit events, and track that we dropped them
00096                     return 0;
00097                 }
00098                 return 1;
00099             }else{
00100                 if(DroppedQuit)         //4 if we need to make a quit event
00101                 {
00102                     DroppedQuit=false;  //Reset this so we don't add more in the future by accident
00103                     return 4;
00104                 }
00105                 return 2;
00106             }
00107         }

00308     void EventManager::UpdateEvents()
00309     {
00310         UpdateQuitEvents(); //quit events skips the preprocessing step and goes straight into the the main Queue, becuase of how we need to get them from sdl
00311 
00312         RawEvent FromSDLRaw;                                    //used to hold data as we go through loop
00313         EventUserInput* FromSDLEvent = new EventUserInput();    //Used to build up all of our userinput data into one event
00314         bool ClearKeyPresses=false;                             //if true All the keypresses will be dropped and all keys will be assumed to be up
00315 
00316         // Here we iterate through manual check to insert any requested polling checks and perpetuate button and key down events
00317         for(internal::EventManagerInternalData::ManualCheckIterator Iter=_Data->ManualCheck.begin(); _Data->ManualCheck.end()!=Iter; ++Iter)
00318         {
00319             if(internal::EventManagerInternalData::Keypress & Iter->second)     //if the keypress event is in there, then the key must be down
00320                 { FromSDLEvent->AddCode(MetaCode::BUTTON_DOWN, Iter->first); }
00321             else
00322                 { FromSDLEvent->AddCode(MetaCode::BUTTON_UP, Iter->first); }    //It must be just a polling check
00323         }
00324 
00325         /* Here is a list of SDL event which aren't coded yet.
00326         //event types
00327         Tabbed items are fully working
00328         items with one space are partially implemented of known to not work
00329         other items are unimplemented
00330             SDL_FIRSTEVENT                              unused (do not remove)          Application events
00331             SDL_QUIT                            user-requested quit             Window events
00332             SDL_WINDOWEVENT                             window state change
00333          SDL_SYSWMEVENT                         system specific event           Keyboard events
00334             SDL_KEYDOWN                         key pressed
00335             SDL_KEYUP                           key released
00336         SDL_TEXTEDITING                         keyboard text editing (composition)
00337         SDL_TEXTINPUT                           keyboard text input             Mouse events
00338             SDL_MOUSEMOTION                             mouse moved
00339             SDL_MOUSEBUTTONDOWN                         mouse button pressed
00340             SDL_MOUSEBUTTONUP                           mouse button released
00341         SDL_MOUSEWHEEL                          mouse wheel motion              Tablet or multiple mice input device events
00342         SDL_INPUTMOTION                         input moved
00343         SDL_INPUTBUTTONDOWN                             input button pressed
00344         SDL_INPUTBUTTONUP                               input button released
00345         SDL_INPUTWHEEL                          input wheel motion
00346         SDL_INPUTPROXIMITYIN                            input pen entered proximity
00347         SDL_INPUTPROXIMITYOUT                           input pen left proximity                Joystick events
00348             SDL_JOYAXISMOTION                           joystick axis motion
00349          SDL_JOYBALLMOTION                              joystick trackball motion
00350             SDL_JOYHATMOTION                            joystick hat position change
00351             SDL_JOYBUTTONDOWN                           joystick button pressed
00352             SDL_JOYBUTTONUP                             joystick button released                Touch events
00353         SDL_FINGERDOWN
00354         SDL_FINGERUP
00355         SDL_FINGERMOTION
00356         SDL_TOUCHBUTTONDOWN
00357         SDL_TOUCHBUTTONUP                               Gesture events
00358         SDL_DOLLARGESTURE
00359         SDL_DOLLARRECORD
00360         SDL_MULTIGESTURE                                Clipboard events
00361         SDL_CLIPBOARDUPDATE                             the clipboard changed           Obsolete events
00362         SDL_EVENT_COMPAT1                               SDL 1.2 events for compatibility

00365         SDL_RegisterEvents()
00366         SDL_USEREVENT                           a user-specified event
00367         SDL_LASTEVENT                           only for bounding internal arrays
00368         */
00369 
00370         while(SDL_PollEvent(&FromSDLRaw))
00371         {
00372             switch(FromSDLRaw.type)
00373             {
00374         //Events and User input sorted by estimate frequency
00375                 case SDL_MOUSEBUTTONUP:     case SDL_KEYUP:             case SDL_JOYBUTTONUP:{
00376                     MetaCode ResultCode(FromSDLRaw);
00377                     _Data->RemoveMetaCodesToManualCheck( FromSDLEvent->AddCode(ResultCode), internal::EventManagerInternalData::Keypress);
00378                     break;}
00379 
00380                 case SDL_KEYDOWN: {
00381                     MetaCode ResultCode(FromSDLRaw);
00382                     if ( !(_Data->ManualCheck[ResultCode.GetCode()]) )              //This checks for operating system level key repititions and skips adding them
00383                         { _Data->AddMetaCodesToManualCheck( FromSDLEvent->AddCode(ResultCode), internal::EventManagerInternalData::Keypress); }
00384                     break; }
00385 
00386                 case SDL_MOUSEBUTTONDOWN:   case SDL_JOYBUTTONDOWN:
00387                     _Data->AddMetaCodesToManualCheck( FromSDLEvent->AddCode(FromSDLRaw), internal::EventManagerInternalData::Keypress);
00388                     break;
00389 
00390                 case SDL_MOUSEMOTION:       case SDL_JOYAXISMOTION:     case SDL_JOYHATMOTION:      case SDL_JOYBALLMOTION: // What is a joyball, like the bowling games maybe, or the ORBit controller
00391                     FromSDLEvent->AddCodesFromRawEvent(FromSDLRaw);
00392                     break;
00393 
00394                 case SDL_WINDOWEVENT: {
00395                     EventGameWindow* React = new EventGameWindow(FromSDLRaw);
00396                     if(EventGameWindow::GAME_WINDOW_FOCUS_LOST==React->GetEventID())        //we dropp all keypresses when windows are switched
00397                         { ClearKeyPresses = true; }
00398                     this->AddEvent(React);
00399                     break; }
00400 
00401                 case SDL_SYSWMEVENT:
00402                         // call a function with ifdefs here
00403                     break;

00419         }

00432 
00433         //Check to see if we should add a User input event or not. We wouldn't want to pass an empty event
00434         if(FromSDLEvent->GetMetaCodeCount()==0)
00435         {
00436             delete FromSDLEvent;
00437         }else{
00438             _Data->EventQ.push_back(FromSDLEvent);
00439         }
