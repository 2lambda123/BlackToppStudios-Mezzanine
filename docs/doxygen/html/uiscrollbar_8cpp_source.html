00421         void Scrollbar::ScrollBackScroll()
00422         {
00423             if(!ScrollBackLock)
00424             {
00425                 Vector2 MousePos = InputQueryTool::GetMouseCoordinates();
00426                 Vector2 ScPos = Scroller->GetActualPosition();
00427                 Vector2 ScSize = Scroller->GetActualSize();
00428                 if(Horizontal)
00429                 {
00430                     if(MousePos.X < ScPos.X)//left
00431                     {
00432                         ScPos.X = ScPos.X - ScSize.X <= ScrollerLowerLimit ? ScrollerLowerLimit : ScPos.X - ScSize.X ;
00433                         Scroller->SetActualPosition(ScPos);
00434                     }
00435                     else if(MousePos.X > ScPos.X + ScSize.X)//right
00436                     {
00437                         ScPos.X = ScPos.X + (ScSize.X * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.X : ScPos.X + (ScSize.X * 2) ;
00438                         Scroller->SetActualPosition(ScPos);
00439                     }
00440                 }else{
00441                     if(MousePos.Y < ScPos.Y)//above
00442                     {
00443                         ScPos.Y = ScPos.Y - ScSize.Y <= ScrollerLowerLimit ? ScrollerLowerLimit : ScPos.Y - ScSize.Y ;
00444                         Scroller->SetActualPosition(ScPos);
00445                     }
00446                     else if(MousePos.Y > ScPos.Y + ScSize.Y)//below
00447                     {
00448                         ScPos.Y = ScPos.Y + (ScSize.Y * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.Y : ScPos.Y + ScSize.Y ;
00449                         Scroller->SetActualPosition(ScPos);
00450                     }
00451                 }
00452             }

00456         {
00457             MetaCode::ButtonState State = InputQueryTool::GetMouseButtonState(1);
00458             if(HoveredButton)

00476                         if(MetaCode::BUTTON_PRESSING == State)

00478                             UpLeftLock = false;

00481                         {
00482                             ButtonScroll(true);
00483                         }
00484                     }
00485                     else if(HoveredButton == DownRightButton)
00486                     {
00487                         if(MetaCode::BUTTON_PRESSING == State)
00488                         {
00489                             DownRightLock = false;
00490                         }
00491                         else if(MetaCode::BUTTON_DOWN == State)
00492                         {
00493                             ButtonScroll(false);
00494                         }
00495                     }
00496                 }
00497             }
00498             else if(HoveredBack)
00499             {
00500                 if(MetaCode::BUTTON_PRESSING == State)
00501                 {
00502                     ScrollBackLock = false;
00503                     ScrollBackScroll();
00504                 }
00505                 else if(MetaCode::BUTTON_DOWN == State)
00506                 {
00507                     if(ScrollBackLock && !ScrollerLock)
00508                     {
00509                         Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();
00510                         MouseScroll(Offset);
00511                     }else{
00512                         ScrollBackScroll();
00513                     }
00514                 }
00515             }
00516             else if(Force && MetaCode::BUTTON_DOWN == State)
00517             {
00518                 Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();
00519                 MouseScroll(Offset);
00520             }
00521             else if(MetaCode::BUTTON_LIFTING == State)
00522             {
00523                 ScrollerLock = true;
00524                 ScrollBackLock = true;
00525                 UpLeftLock = true;
