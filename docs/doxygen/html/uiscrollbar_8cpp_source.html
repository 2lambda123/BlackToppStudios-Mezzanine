00537                             DownRightLock = false;
00538                         }
00539                         else if(MetaCode::BUTTON_DOWN == State)
00540                         {
00541                             ButtonScroll(false);
00542                         }
00543                     }
00544                 }
00545             }
00546             else if(HoveredBack)
00547             {
00548                 if(MetaCode::BUTTON_PRESSING == State)

00554                 {
00555                     if(ScrollBackLock && !ScrollerLock)
00556                     {
00557                         Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();

00563             }
00564             else if(Force && MetaCode::BUTTON_DOWN == State)

00566                 Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();
00567                 MouseScroll(Offset);

00571                 ScrollerLock = true;
