00467         void Scrollbar::ScrollBackScroll()
00468         {
00469             if(!ScrollBackLock)
00470             {
00471                 Vector2 MousePos = InputQueryTool::GetMouseCoordinates();
00472                 Vector2 ScPos = Scroller->GetActualPosition();
00473                 Vector2 ScSize = Scroller->GetActualSize();
00474                 if(Horizontal)
00475                 {
00476                     if(MousePos.X < ScPos.X)//left

00479                         Scroller->SetActualPosition(ScPos);

00481                     else if(MousePos.X > ScPos.X + ScSize.X)//right
00482                     {
00483                         ScPos.X = ScPos.X + (ScSize.X * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.X : ScPos.X + (ScSize.X * 2) ;
00484                         Scroller->SetActualPosition(ScPos);
00485                     }
00486                 }else{
00487                     if(MousePos.Y < ScPos.Y)//above
00488                     {
00489                         ScPos.Y = ScPos.Y - ScSize.Y <= ScrollerLowerLimit ? ScrollerLowerLimit : ScPos.Y - ScSize.Y ;
00490                         Scroller->SetActualPosition(ScPos);
00491                     }
00492                     else if(MousePos.Y > ScPos.Y + ScSize.Y)//below
00493                     {
00494                         ScPos.Y = ScPos.Y + (ScSize.Y * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.Y : ScPos.Y + ScSize.Y ;

00532                     {
00533                         if(MetaCode::BUTTON_PRESSING == State)
00534                         {
00535                             DownRightLock = false;
00536                         }
00537                         else if(MetaCode::BUTTON_DOWN == State)
00538                         {
00539                             ButtonScroll(false);
00540                         }
00541                     }
00542                 }
00543             }
00544             else if(HoveredBack)
00545             {
00546                 if(MetaCode::BUTTON_PRESSING == State)
00547                 {
00548                     ScrollBackLock = false;
00549                     ScrollBackScroll();
00550                 }
00551                 else if(MetaCode::BUTTON_DOWN == State)
00552                 {
00553                     if(ScrollBackLock && !ScrollerLock)
00554                     {
00555                         Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();
00556                         MouseScroll(Offset);
00557                     }else{
00558                         ScrollBackScroll();
00559                     }
00560                 }
00561             }
00562             else if(Force && MetaCode::BUTTON_DOWN == State)
00563             {
00564                 Vector2 Offset = InputQueryTool::GetMousePrevFrameOffset();
00565                 MouseScroll(Offset);
00566             }
00567             else if(MetaCode::BUTTON_LIFTING == State)
00568             {
00569                 ScrollerLock = true;
00570                 ScrollBackLock = true;
00571                 UpLeftLock = true;
00572                 DownRightLock = true;

00575         }
