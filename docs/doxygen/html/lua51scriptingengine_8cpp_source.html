90  /// @internal
91  /// @brief Used in the LuaScriptingEngine to get data from lua_dump during script compilation
92  /// @param State The Lua state as provide by lua_dump
93  /// @param Buffer A pointer to the compiled Lua chunk.
94  /// @param Size The Size of the Lua chunk in bytes
95  /// @param BinBuff A pointer to a BinaryTools::BinaryBuffer that will serve as the real output
96  /// @note This is a lua_Writer as per http://www.lua.org/manual/5.1/manual.html#lua_Writer
97  int LuaBytecodeDumper(lua_State *State, const void* Buffer, size_t Size, void* BinBuff)
98  {
99  BinaryTools::BinaryBuffer* CompilingScript = reinterpret_cast<BinaryTools::BinaryBuffer*>(BinBuff);
100  CompilingScript->Concatenate(
101  (BinaryTools::BinaryBuffer::Byte*) Buffer,
102  Size
103  );
104  return 0;
105  }

107  /// @internal
108  /// @param State The Lua state as provided by lua_load()
109  /// @param Buffer A BinaryBuffer containing the bytecode to load into Lua
110  /// @param Size an output parameter to convey the size of the return to Lua.
111  /// @return A pointer to a binary buffer suitable for Lua's use and the size of that buffer in the output parameter Size
112  /// @warning The Lua documentation clearly indicates sec ond parameter should be const and the third parameter should be a size_t* but the compiler says long unsigned int*
113  /// @note This is a lua_Reader as per http://www.lua.org/manual/5.1/manual.html#lua_Reader
114  const char* LuaBytecodeLoader(lua_State* State, void* BinBuff, size_t* Size)
115  {
116  FlaggedBuffer* LoadingBuffer = reinterpret_cast<FlaggedBuffer*>(BinBuff);
117  *Size = LoadingBuffer->Size;
118 
119  return (const char*)LoadingBuffer->Binary;
120  }

122  /// @internal
123  /// @param State The Lua state as provided by lua_load()
124  /// @param Buffer A Lua51Script* containing the source to load into Lua
125  /// @param Size an output parameter to convey the size of the return to Lua.
126  /// @return A pointer to a binary buffer suitable for Lua's use and the size of that buffer in the output parameter Size
127  /// @warning The Lua documentation clearly indicates second parameter should be const and the third parameter should be a size_t* but the compiler says unsigned int*
128  /// @note This is a lua_Reader as per http://www.lua.org/manual/5.1/manual.html#lua_Reader
129  const char* LuaSourceLoader(lua_State* State, void* BinBuff, size_t* Size)
130  {
131  Lua51Script* LoadingScript = reinterpret_cast<Lua51Script*>(BinBuff);
132  if(LoadingScript->GetByteCodeReference().Loaded)
133  {
134  LoadingScript->GetByteCodeReference().Loaded = false;
135  return 0;
136  }else{
137  *Size = LoadingScript->GetSourceCode().size();
138  LoadingScript->GetByteCodeReference().Loaded = true;
139  return LoadingScript->GetSourceCode().c_str();
140  }
141  }

148  void Lua51ScriptingEngine::ThrowFromLuaErrorCode(int LuaReturn)
149  {
150  if(0==LuaReturn)
151  { return; }
152 
153  String ErrorMessage;
154  size_t Length;
155  const char* ErrorCString = lua_tolstring(this->State, -1, &Length);
156  ErrorMessage.reserve(Length+2);
157  ErrorMessage = String(ErrorCString);
158  lua_pop(State,1);
159  if(ErrorMessage.length()!=Length)
160  { MEZZ_EXCEPTION(Exception::SCRIPT_EXCEPTION_LUA, "Lua is putting odd things in error messages:\n"+ErrorMessage); }
161  ErrorMessage += "\n";
162 
163  switch(LuaReturn)
164  {
165  case LUA_YIELD:
166  MEZZ_EXCEPTION(Exception::SCRIPT_EXCEPTION_LUA_YIELD, "Lua returned a LUA_YIELD instead of completing:\n"+ErrorMessage);
167  case LUA_ERRRUN:
168  MEZZ_EXCEPTION(Exception::SCRIPT_EXCEPTION_LUA_RUNTIME, "There was a runtime Error handling the Lua script:\n"+ErrorMessage);
169  case LUA_ERRSYNTAX:
170  MEZZ_EXCEPTION(Exception::SYNTAX_ERROR_EXCEPTION_LUA, "There was an error with the syntax of the Lua script:\n"+ErrorMessage);
171  case LUA_ERRERR:
172  MEZZ_EXCEPTION(Exception::SCRIPT_EXCEPTION_LUA_ERRERR, "There was an error when Lua attempted to handle an error:\n"+ErrorMessage);
173  case LUA_ERRMEM:
174  MEZZ_EXCEPTION(Exception::MM_OUT_OF_MEMORY_EXCEPTION, "Lua could not allocate memory:\n"+ErrorMessage);
175  case LUA_ERRFILE:
176  MEZZ_EXCEPTION(Exception::IO_FILE_EXCEPTION, "Lua had an error with file IO:\n"+ErrorMessage);
177  default:
178  MEZZ_EXCEPTION(Exception::SCRIPT_EXCEPTION_LUA, "Lua had an error and we are not sure what it was:\n"+ErrorMessage);
179  }
180  }

246  CountedPtr<iScriptArgument> Lua51ScriptingEngine::ScriptArgFromStack()
247  {
248  iScriptArgument* Results = NULL;
249  Integer StackLocation = lua_gettop(State);
250  if(LUA_TNIL==lua_type(State,StackLocation))
251  { Results = new Lua51NilArgument; }
252  if(LUA_TBOOLEAN==lua_type(State,StackLocation))
253  { Results = new Lua51BoolArgument; }
254  if(LUA_TLIGHTUSERDATA==lua_type(State,StackLocation))
255  { MEZZ_EXCEPTION(Exception::PARAMETERS_CAST_EXCEPTION, "A LuaArgument could not be converted as one for return value purposes. Found a LightUserData."); }
256  if(LUA_TNUMBER==lua_type(State,StackLocation))
257  { Results = new Lua51RealArgument; }
258  if(LUA_TSTRING==lua_type(State,StackLocation))
259  { Results = new Lua51StringArgument; }
260  if(LUA_TTABLE==lua_type(State,StackLocation))
261  { MEZZ_EXCEPTION(Exception::PARAMETERS_CAST_EXCEPTION, "A LuaArgument could not be converted as one for return value purposes. Found a Table."); }
262  if(LUA_TFUNCTION==lua_type(State,StackLocation))
263  { MEZZ_EXCEPTION(Exception::PARAMETERS_CAST_EXCEPTION, "A LuaArgument could not be converted as one for return value purposes. Found a Function."); }
264  if(LUA_TUSERDATA==lua_type(State,StackLocation))
265  { MEZZ_EXCEPTION(Exception::PARAMETERS_CAST_EXCEPTION, "A LuaArgument could not be converted as one for return value purposes. Found a UserData."); }
266  if(LUA_TTHREAD==lua_type(State,StackLocation))
267  { MEZZ_EXCEPTION(Exception::PARAMETERS_CAST_EXCEPTION, "A LuaArgument could not be converted as one for return value purposes. Found a Thread."); }
268 
269  dynamic_cast<LuaArgument*>(Results)->Pop(State);
270  return CountedPtr<iScriptArgument>(Results);
271  }

375  Lua51ScriptingEngine::Lua51Libraries Lua51ScriptingEngine::GetLibFromName(String Name)
376  {
377  if(!Name.size())
378  { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Cannot convert zero length name to valid lua library name"); }
379  using namespace Mezzanine::StringTools;
380  ToLowerCase(Name);
381 
382  switch(Name[0])
383  {
384  case 'n': if(Name==LowerCaseCopy(NoLibName)) { return NoLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with n: " + Name); }
385  case 'b': if(Name==LowerCaseCopy(BaseLibName)) { return BaseLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with b: " + Name); }
386  case 's': if(Name==LowerCaseCopy(StringLibName)) { return StringLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with s: " + Name); }
387  case 't': if(Name==LowerCaseCopy(TableLibName)) { return TableLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with t: " + Name); }
388  case 'i': if(Name==LowerCaseCopy(IOLibName)) { return IOLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with i: " + Name); }
389  case 'o': if(Name==LowerCaseCopy(OSLibName)) { return OSLib; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with o: " + Name); }
390  case 'a': if(Name==LowerCaseCopy(AllLibsName)) { return AllLibs; } else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with a: " + Name); }
391  case 'p':
392  if (Name==LowerCaseCopy(PackageLibName)) { return PackageLib; }
393  else if(Name==LowerCaseCopy(MezzPhysicsLibName)) { return MezzPhysicsLib; }
394  else if(Name==LowerCaseCopy(MezzPhysicsSafeLibName)) { return MezzPhysicsSafeLib; }
395  else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with p: " + Name); }
396  case 'm':
397  if (Name==LowerCaseCopy(MathLibName)) { return MathLib; }
398  else if(Name==LowerCaseCopy(MezzLibName)) { return MezzLib; }
399  else if(Name==LowerCaseCopy(MezzSafeLibName)) { return MezzSafeLib; }
400  else if(Name==LowerCaseCopy(MezzXMLLibName)) { return MezzXMLLib; }
401  else if(Name==LowerCaseCopy(MezzXMLSafeLibName)) { return MezzXMLSafeLib; }
402  else if(Name==LowerCaseCopy(MezzThreadingLibName)) { return MezzThreadingLib; }
403  else if(Name==LowerCaseCopy(MezzThreadingSafeLibName)) { return MezzThreadingSafeLib; }
404  else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with m: " + Name); }
405  case 'd':
406  if (Name==LowerCaseCopy(DebugLibName)) { return DebugLib; }
407  else if(Name==LowerCaseCopy(DefaultLibsName)) { return DefaultLibs; }
408  else { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert name starting with d: " + Name); }
409  default: MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Could not convert: " + Name);
410  }
411  }

421  Lua51ScriptingEngine::Lua51ScriptingEngine(NameValuePairList& Params) : State(luaL_newstate())
422  {
423  CheckLuaStateAfterConstruction();
424  Integer ToLoad = NoLib;
425  if(Params.empty())
426  { ToLoad = DefaultLibs; }
427  else
428  {
429  for(NameValuePairList::iterator Iter = Params.begin(); Params.end() != Iter; Iter++ )
430  {
431  Lua51Libraries Lib( GetLibFromName(Iter->first) );
432  String LoadState(Iter->second);
433  if(String("Load")==LoadState)
434  { ToLoad |= Lib; }
435  else{
436  if(String("Unload")==LoadState)
437  { ToLoad &= ~Lib; }
438  else
439  { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Unknown loadstate parameter, during name value pair construction."); }
440  }
441  }
442  }
443  OpenLibraries( (Lua51Libraries)ToLoad );
444  }

480  void Lua51ScriptingEngine::Execute(Lua51Script* ScriptToRun)
481  {
482  Integer StackSize = lua_gettop(this->State);
483  ScriptOntoStack(ScriptToRun);
484  assert(lua_gettop(this->State) == StackSize + 1);
485  ScriptArgsOntoStack(ScriptToRun);
486  assert(lua_gettop(this->State) == StackSize + 1 + Integer(ScriptToRun->GetArgumentCount()));
487  StackExecute(ScriptToRun->GetArgumentCount());
488  ScriptArgsFromStack(ScriptToRun, StackSize);
489  assert(lua_gettop(this->State) == StackSize);
490  }

764  const String& Lua51ScriptingEngine::GetLuaTypeString(int StackLocation)
765  {
766  if(std::abs(StackLocation)>GetStackCount())
767  { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Attempting to inspect beyond Lua stack while getting typestring."); }
768 
769  if(LUA_TNIL==lua_type(State,StackLocation))
770  { return TypeNameNil; }
771  if(LUA_TBOOLEAN==lua_type(State,StackLocation))
772  { return TypeNameBoolean; }
773  if(LUA_TLIGHTUSERDATA==lua_type(State,StackLocation))
774  { return TypeNameLightUserData; }
775  if(LUA_TNUMBER==lua_type(State,StackLocation))
776  { return TypeNameNumber; }
777  if(LUA_TSTRING==lua_type(State,StackLocation))
778  { return TypeNameString; }
779  if(LUA_TTABLE==lua_type(State,StackLocation))
780  { return TypeNameTable; }
781  if(LUA_TFUNCTION==lua_type(State,StackLocation))
782  { return TypeNameFunction; }
783  if(LUA_TUSERDATA==lua_type(State,StackLocation))
784  { return TypeNameUserData; }
785  if(LUA_TTHREAD==lua_type(State,StackLocation))
786  { return TypeNameThread; }
787  MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "The thing on the Lua stack match no known types.");
788  return NoLibName;

797  void Lua51ScriptingEngine::PopulateTabCompletionTrie(CommandTrie& CommandGroup, const String& TableName, std::vector<String> AlreadyDidTables)
798  {
799  // If no table is based start at the basemost global table
800  int Top = GetStackCount();
801  if(TableName=="")
802  {
803  lua_getglobal(State, GlobalTableName.c_str());
804  AlreadyDidTables.push_back(GlobalTableName);
805  }
806  else
807  {
808  lua_getglobal(State, TableName.c_str());
809  AlreadyDidTables.push_back(TableName);
810  }
811 
812  // Handle errors
813  if(GetStackCount()==Top) // if lua_getglobal puts nothing on the stack
814  { MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Lua51 Engine needs a table name to read for tab completion data instead an invalid identifier was passed."); }
815  if(!(LUA_TTABLE==lua_type(State,-1)))
816  {
817  lua_pop(State, 1);
818  MEZZ_EXCEPTION(Exception::PARAMETERS_RANGE_EXCEPTION, "Lua51 Engine needs a able name to read for tab completion data instead something else was passed.");
819  }
820 
821  // iterate of each entry, gather its name and type and recurse into subtables
822  lua_pushnil(State);
823  while (lua_next(State, -2) != 0)
824  {
825  // Capture Strings(names) and their associated types
826  if(LUA_TSTRING==lua_type(State,-2))
827  {
828  String TablePrefix;
829  if(TableName!=String(""))
830  { TablePrefix = TableName + "."; }
831  CommandGroup.insert((TablePrefix + lua_tostring(State,-2)).c_str(), &GetLuaTypeString(-1));
832  }
833 
834  // If it is a table and we have no done it yet, recurse
835  if(LUA_TTABLE==lua_type(State,-1))
836  {
837  lua_pop(State, 1);
838  bool AlreadyParsed = false;
839 
840  for(std::vector<String>::const_iterator Iter = AlreadyDidTables.begin();
841  AlreadyDidTables.end() != Iter;
842  Iter++)
843  {
844  if(*Iter==TableName)
845  {
846  AlreadyParsed = true;
847  break;
848  }
849  }
850  if(!AlreadyParsed)
851  { PopulateTabCompletionTrie(CommandGroup, lua_tostring(State,-1), AlreadyDidTables); }
852  }else{
853  lua_pop(State, 1);
854  }
855  }
856  lua_pop(State, 1);
857  }
