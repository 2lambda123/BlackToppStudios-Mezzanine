00063         {
00064             /// @todo Once the resource system has been updated to be independant of Ogre's, this function needs to be updated to access our methods instead of theirs.
00065             Clear();

00069             String CurrentSectionName("");

00073             String Line, SetName, SetVal;

00075             {
00076                 Line = Stream->getLine();

00078                 {
00079                     if (Line.at(0) == '[' && Line.at(Line.length()-1) == ']')
00080                     {
00081                         CurrentSectionName = Line.substr(1, Line.length() - 2);
00082                         SectionIterator SecIt = Sections.find(CurrentSectionName);
00083                         if (SecIt == Sections.end())
00084                         {
00085                             CurrentSection = new SettingsMap();
00086                             Sections[CurrentSectionName] = CurrentSection;
00087                         }else{
00088                             CurrentSection = SecIt->second;
00089                         }
00090                     }else{
00091                         String::size_type SeparatorPos = Line.find_first_of(Separators,0);
00092                         if(SeparatorPos != String::npos)
00093                         {
00094                             SetName = Line.substr(0,SeparatorPos);
00095                             String::size_type NonSeparatorPos = Line.find_first_not_of(Separators,SeparatorPos);
00096                             SetVal = (NonSeparatorPos == String::npos) ? "" : Line.substr(NonSeparatorPos);
00097                             if (TrimWhitespace)
00098                             {
00099                                 StringTool::Trim(SetVal);
00100                                 StringTool::Trim(SetName);
00101                             }
00102                             CurrentSection->insert(SettingsMap::value_type(SetName,SetVal));
00103                         }
00104                     }
00105                 }
00106             }
00107         }
