284  void Screen::CreateVertexBuffer(const Whole& InitialSize)
285  {
286  this->SID->RenderOp.vertexData = OGRE_NEW Ogre::VertexData;
287  this->SID->RenderOp.vertexData->vertexStart = 0;
288 
289  Ogre::VertexDeclaration* VertexDecl = this->SID->RenderOp.vertexData->vertexDeclaration;
290  size_t Offset = 0;
291 
292  // Position.
293  VertexDecl->addElement(0,0,Ogre::VET_FLOAT3,Ogre::VES_POSITION);
294  Offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
295 
296  // Colour
297  VertexDecl->addElement(0,Offset,Ogre::VET_FLOAT4,Ogre::VES_DIFFUSE);
298  Offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT4);
299 
300  // Texture Coordinates
301  VertexDecl->addElement(0,Offset,Ogre::VET_FLOAT2, Ogre::VES_TEXTURE_COORDINATES);
302 
303  this->SID->VertexBuffer = Ogre::HardwareBufferManager::getSingletonPtr()->createVertexBuffer(
304  VertexDecl->getVertexSize(0),
305  InitialSize,
306  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY_DISCARDABLE,
307  false);
308 
309  this->SID->RenderOp.vertexData->vertexBufferBinding->setBinding(0,this->SID->VertexBuffer);
310  this->SID->RenderOp.operationType = Ogre::RenderOperation::OT_TRIANGLE_LIST;
311  this->SID->RenderOp.useIndexes = false;
312  }

321  void Screen::ResizeVertexBuffer(const Whole& RequestedSize)
322  {
323  if( this->SID->VertexBuffer.isNull() )
324  this->CreateVertexBuffer(RequestedSize);
325 
326  if(RequestedSize > this->SID->VertexBuffer->getNumVertices() ) {
327  // Update only by powers of 2
328  Whole NewVertexBufferSize = 1;
329  while(NewVertexBufferSize < RequestedSize)
330  NewVertexBufferSize <<= 1;
331 
332  this->SID->VertexBuffer = Ogre::HardwareBufferManager::getSingletonPtr()->createVertexBuffer(
333  this->SID->RenderOp.vertexData->vertexDeclaration->getVertexSize(0),
334  NewVertexBufferSize,
335  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY_DISCARDABLE,
336  false);
337 
338  this->SID->RenderOp.vertexData->vertexStart = 0;
339  this->SID->RenderOp.vertexData->vertexBufferBinding->setBinding(0,this->SID->VertexBuffer);
340  }
341  }
