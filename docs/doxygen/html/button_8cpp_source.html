86  Boole Button::HandleInputImpl(const Input::MetaCode& Code)
87  {
88  // Check to see if this is an activation code
89  if( Code.IsDeviceButton() && this->ActivationCodes.count( Input::MetaCodeKey(Code) ) ) {
90  // Are we changing our activation state from a mouse code?
91  if( Code.IsMouseButton() ) {
92  if( this->IsHovered() && Input::BUTTON_PRESSING == Code.GetMetaValue() ) {
93  Boole Result = this->Activate();
94  if(Result)
95  this->MouseActivated = true;
96  return true;
97  }else if( Input::BUTTON_LIFTING == Code.GetMetaValue() ) {
98  this->Deactivate();
99  this->MouseActivated = false;
100  return true;
101  }
102  }else{
103  if( Input::BUTTON_PRESSING == Code.GetMetaValue() ) {
104  this->Activate();
105  return true;
106  }else if( Input::BUTTON_LIFTING == Code.GetMetaValue() ) {
107  this->Deactivate();
108  return true;
109  }
110  }
111  }
112  return false;
113  }

115  void Button::ConstructButton()
116  {
117  // Add our button events
118  this->AddEvent(Button::EventActivated);
119  this->AddEvent(Button::EventStandby);
120  this->AddEvent(Button::EventDeactivated);
121  // Handle auto activation registration
122  UIManager* Manager = this->ParentScreen->GetManager();
123  if( Manager->ButtonAutoRegisterEnabled() )
124  {
125  const UIManager::InputContainer& Codes = Manager->GetAutoRegisteredCodes();
126  for( Whole X = 0 ; X < Codes.size() ; X++ )
127  this->BindActivationKeyOrButton(Codes.at(X));
128  }
129  }

131  Boole Button::VertifyActivationCode(const Input::InputCode Code)
132  {
133  return (Input::KEY_FIRST < Code && Input::KEY_LAST > Code) ||
134  (Input::MOUSEBUTTON_FIRST <= Code && Input::MOUSEBUTTON_LAST >= Code) ||
135  (Input::CONTROLLERBUTTON_FIRST <= Code && Input::CONTROLLERBUTTON_LAST >= Code);

138  Boole Button::Activate()
139  {
140  if( this->Activation == AS_Activated )
141  return false;
142  if( this->LockoutTimer.GetInitialTime() != 0 && !(this->LockoutTimer.IsStopped()) )
143  return false;
144 
145  this->Activation = AS_Activated;
146  this->_OnActivate();
147  return true;
148  }
149 
150  Boole Button::Deactivate()
151  {
152  if( this->Activation == AS_Deactivated )
153  return false;
154  if( this->LockoutTimer.GetInitialTime() != 0 )
155  {
156  this->LockoutTimer.Reset();
157  this->LockoutTimer.Start();
158  }
159 
160  this->Activation = AS_Deactivated;
161  this->_OnDeactivate();
162  return true;
163  }
164 
165  Boole Button::Standby()
166  {
167  if(this->Activation == AS_Activation_Standby)
168  return false;
169 
170  this->Activation = AS_Activation_Standby;
171  this->_OnStandby();
172  return true;
173  }

265  void Button::ProtoSerializeProperties(XML::Node& SelfRoot) const
266  {
267  this->Widget::ProtoSerializeProperties(SelfRoot);
268 
269  XML::Node PropertiesNode = SelfRoot.AppendChild( Button::GetSerializableName() + "Properties" );
270 
271  if( PropertiesNode.AppendAttribute("Version").SetValue("1") )
272  {
273  // Only if we have a valid timer
274  if( this->LockoutTimer.GetInitialTime() != 0 ) {
275  if( PropertiesNode.AppendAttribute("LockoutTime").SetValue( this->LockoutTimer.GetInitialTime() ) == false ) {
276  SerializeError("Create XML Attribute Values",Button::GetSerializableName() + "Properties",true);
277  }
278  }
279 
280  XML::Node CodesNode = PropertiesNode.AppendChild( "ActivationCodes" );
281  for( ActivationCodeIterator CodeIt = this->ActivationCodes.begin() ; CodeIt != this->ActivationCodes.end() ; ++CodeIt )
282  {
283  (*CodeIt).GetKeyData().ProtoSerialize(CodesNode);
284  }
285 
286  return;
287  }else{
288  SerializeError("Create XML Attribute Values",Button::GetSerializableName() + "Properties",true);
289  }
290  }

292  void Button::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
293  {
294  this->Widget::ProtoDeSerializeProperties(SelfRoot);
295 
296  XML::Attribute CurrAttrib;
297  XML::Node PropertiesNode = SelfRoot.GetChild( Button::GetSerializableName() + "Properties" );
298 
299  if( !PropertiesNode.Empty() ) {
300  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
301  CurrAttrib = PropertiesNode.GetAttribute("LockoutTime");
302  if( !CurrAttrib.Empty() )
303  this->SetLockoutTime( CurrAttrib.AsUint() );
304 
305  XML::Node CodesNode = PropertiesNode.GetChild( "ActivationCodes" );
306  if( !CodesNode.Empty() ) {
307  for( XML::NodeIterator CodeNodeIt = CodesNode.begin() ; CodeNodeIt != CodesNode.end() ; ++CodeNodeIt )
308  {
309  Input::MetaCode ActivationCode;
310  ActivationCode.ProtoDeSerialize( (*CodeNodeIt) );
311  this->BindActivationKeyOrButton(ActivationCode);
312  }
313  }
314  }else{
315  MEZZ_EXCEPTION(ExceptionBase::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (Button::GetSerializableName() + "Properties") + ": Not Version 1.");
316  }
317  }else{
318  MEZZ_EXCEPTION(ExceptionBase::II_IDENTITY_NOT_FOUND_EXCEPTION,Button::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
319  }
320  }
