59  /// @enum HTTPRequestMethod
60  /// @brief This enum is a listing of commonly supported HTTP request methods.
61  /// @details GET and HEAD are expected to be implemented by all HTTP servers, and generally can be used without much issue. OPTIONS is also
62  /// expected to be available but is a little less often. Other requests may or may not be available. OPTIONS can be used to see what is
63  /// available on a given server, or you can just make a request and see if it returns "HSC_BadRequest" or "HSC_NotImplemented".
64  enum HTTPRequestMethod
65  {
66  HRM_Invalid = 0, ///< Use for error conditions, not an actual request method.
67  // HTTP 1.0 Methods
68  HRM_GET = 1, ///< Requests a representation of the specified resource.
69  HRM_HEAD = 2, ///< Similar to GET, but only requests the header without the body.
70  HRM_POST = 3, ///< Requests the server accept enclosed data as a new subordinate of the specified resource/URI. Such as a blog or forum post.
71  // HTTP 1.1 Methods
72  HRM_OPTIONS = 4, ///< Retrieves a list of the supported HTTP methods on the server.
73  HRM_PUT = 5, ///< Places or replaces a supplied resource at the specified URI.
74  HRM_DELETE = 6, ///< Deletes the specified resource.
75  HRM_TRACE = 7, ///< Requests a list of changes or additions done by other sessions.
76  HRM_CONNECT = 8, ///< Converts the request connection to a tunnel, useful for sending traffic through a proxy.
77  // RFC 5789 Method Extensions
78  HRM_PATCH = 9, ///< Applies partial modifications to a resource.
79  // WebDAV Method Extensions
80  HRM_PROPFIND = 10, ///< Retrieves a property, stored as XML, from a resource. Can alternatively be overloaded to retrieve the directory hierarchy of a remote system.
81  HRM_PROPPATCH = 11, ///< Can change and/or delete multiple properties of a resource in a single atomic act.
82  HRM_MKCOL = 12, ///< Can be used to create a new directory. "COL" is short for Collection, which is the term commonly used for directories in this context.
83  HRM_COPY = 13, ///< Copies a resource from one URI to another.
84  HRM_MOVE = 14, ///< Moves a resource from one URI to another.
85  HRM_LOCK = 15, ///< Locks a resource to prevent alterations being made to it by another session. This is similar in function to mutexes.
86  HRM_UNLOCK = 16 ///< Unlocks a resource, freeing it for other purposes.
87  };

89  /// @enum HTTPStatusCode
90  /// @brief This enum is a listing of HTTP status codes that can be returned in response to a request.
91  enum HTTPStatusCode
92  {
93  HSC_Invalid = 0, ///< Used for internal error conditions, not an actual status code.
94  // 1xx Status Codes - Informational - Only available in HTTP 1.1
95  HSC_Continue = 100, ///< Available in HTTP 1.1. Server has received the request header and client should sent the request body.
96  HSC_SwitchingProtocols = 101, ///< Available in HTTP 1.1. Server is acknowledging the requesters desire to switch protocols and is doing so.
97  HSC_Processing = 102, ///< Available in WebDAV/RFC 2518. Server acknowledges the request, but it will take some time.
98  // 2xx Status Codes - Success
99  HSC_Ok = 200, ///< Available in HTTP 1.0. Standard successful response to a request.
100  HSC_Created = 201, ///< Available in HTTP 1.0. The request to create a resource was successful.
101  HSC_Accepted = 202, ///< Available in HTTP 1.0. The request was successful, but will take time to process.
102  HSC_NonAuthoritativeInfo = 203, ///< Available in HTTP 1.1. The request was successful, but the returned result may be from another source/server.
103  HSC_NoContent = 204, ///< Available in HTTP 1.0. The request was successful, but there is no content to be returned by it.
104  HSC_ResetContent = 205, ///< Available in HTTP 1.0. Similar to MSC_NoContent, but additionally the server is requiring the requester reset the document view.
105  HSC_PartialContent = 206, ///< Available in RFC 7233. The request was successful, but only part of the content is being delivered due to a range header sent by the client.
106  HSC_MultiStatus = 207, ///< Available in WebDAV/RFC 4918. A WebDAV response that means there are multiple statuses to be reported due to multiple sub-requests being sent by the client.
107  HSC_AlreadyReported = 208, ///< Available in WebDAV/RFC 5842. A WebDAV response that means members of a DAV binding have already been reported in a previous reply (to the same request) and won't be reported again.
108  HSC_IMUsed = 226, ///< Available in RFC 3229. The request was successful, and the response is a representation of the result of one or more instance manipulations applied to the current instance.
109  // 3xx Status Codes - Redirection
110  HSC_MultipleChoices = 300, ///< Available in HTTP 1.0. Indicates multiple options for the resource that the client may follow.
111  HSC_MovedPermanently = 301, ///< Available in HTTP 1.0. The resource has been moved and following requests should be directed to following URI.
112  HSC_Found = 302, ///< Available in HTTP 1.0. This code is a mess. Implementations often used it similarly to HSC_SeeOther, despite the standard saying otherwise. The standard meant for this to be a temporary redirect.
113  HSC_SeeOther = 303, ///< Available in HTTP 1.1. The response to the request can be found under another URI using the GET HTTP method.
114  HSC_NotModified = 304, ///< Available in RFC 7232. The requested resource has not been modified since the client last downloaded a copy.
115  HSC_UseProxy = 305, ///< Available in HTTP 1.1. The requested resource is only available through the specified proxy.
116  HSC_SwitchProxy = 306, ///< Available in HTTP 1.0, deprecated. The request should be repeated through the provided proxy.
117  HSC_TemporaryRedirect = 307, ///< Available in HTTP 1.1. The request should be repeated with the provided URI, but future calls may use the original URI.
118  HSC_PermanentRedirect = 308, ///< Available in RFC 7538. The request, and all future requests, should be repeated with the provided URI
119  // 4xx Status Codes - Client Error
120  HSC_BadRequest = 400, ///< Available in HTTP 1.0. The request couldn't be fulfilled because of an error with the request that is believed to be client-side.
121  HSC_Unauthorized = 401, ///< Available in RFC 7235. Authentication is required to access the resource. Sometimes a challenge (for authentication) will be provided along with this response.
122  HSC_PaymentRequired = 402, ///< Available in HTTP 1.0. Mostly unused, intended to express that a micropayment is necessary to access the requested resource.
123  HSC_Forbidden = 403, ///< Available in HTTP 1.0. The request was received but the server refuses to respond to it. Similar to HSC_Unauthorized, but authenticating will do nothing.
124  HSC_NotFound = 404, ///< Available in HTTP 1.0. The requested resource does not exist, but may be available in the future.
125  HSC_MethodNotAllowed = 405, ///< Available in HTTP 1.0. The request method used was made on a resource that doesn't support that method.
126  HSC_NotAcceptable = 406, ///< Available in HTTP 1.0. The requested resource can only generate content that was specified as not acceptable according to the Accept headers specified in the request.
127  HSC_ProxyAuthenticationRequired = 407, ///< Available in RFC 7235. The requesting client must authenticate itself with a proxy.
128  HSC_RequestTimeout = 408, ///< Available in HTTP 1.0. The server timed out while waiting for the clients request.
129  HSC_Conflict = 409, ///< Available in HTTP 1.0. The server detected a conflict in the clients request.
130  HSC_Gone = 410, ///< Available in HTTP 1.0. The requested resource does not exist, and will not be available in the future.
131  HSC_LengthRequired = 411, ///< Available in HTTP 1.0. The requested resource requires the request specify it's length, which it requires.
132  HSC_PreconditionFailed = 412, ///< Available in RFC 7232. The server does not meet one of the preconditions specified in the request.
133  HSC_PayloadTooLarge = 413, ///< Available in RFC 7231. The request is too large for the server to be able or willing to process.
134  HSC_RequestURITooLong = 414, ///< Available in HTTP 1.0. The URI specified in the request was too long for the server to be able or willing to process.
135  HSC_UnsupportedMediaType = 415, ///< Available in HTTP 1.0. The media type specified in the request is a type that the server does not support.
136  HSC_RequestRangeNotSatisfiable = 416, ///< Available in RFC 7233. The request has asked for a portion of a resource, which the server cannot provide.
137  HSC_ExpectationFailed = 417, ///< Available in HTTP 1.0. The server cannot meet the requirements set by the "Expect" request header.
138  HSC_ImATeapot = 418, ///< Available in RFC 2324. A joke status code, not intended for any real-world use.
139  HSC_AuthenticationTimeout = 419, ///< Source of definition unknown. The authentication that was used for a session has timed out, preventing the request from being processed.
140  HSC_MisdirectedRequest = 421, ///< Available in HTTP 2.0. The request was directed at a server that is not able to produce a response.
141  HSC_UnprocessableEntity = 422, ///< Available in WebDAV/RFC 4918. The request is well formed, but contains semantic errors preventing it from being processed.
142  HSC_Locked = 423, ///< Available in WebDAV/RFC 4918. The requested resource is currently locked.
143  HSC_FailedDependency = 424, ///< Available in WebDAV/RFC 4918. The request failed because of a previous request failure.
144  HSC_UpgradeRequired = 426, ///< Available in HTTP 1.0. The server wants the client to change to the protocol specified in the response.
145  HSC_PreconditionRequired = 428, ///< Available in RFC 6585. The server requires a request to a resource to be conditional, in order to avoid conflicts.
146  HSC_TooManyRequests = 429, ///< Available in RFC 6585. The client has sent too many requests recently to process the current request.
147  HSC_RequestHeaderFieldsTooLarge = 431, ///< Available in RFC 6585. The server will not process the request because one or more of the header fields are too large.
148  // 5xx Status Codes - Server Error
149  HSC_InternalServerError = 500, ///< Available in HTTP 1.0. Generic error message for when something goes wrong but the cause is unknown or no suitable error code for it exists.
150  HSC_NotImplemented = 501, ///< Available in HTTP 1.0. The request method is not supported or the server otherwise lacks the ability to process the request.
151  HSC_BadGateway = 502, ///< Available in HTTP 1.0. The server is currently acting as a proxy or gateway, and received an invalid response from the upstream server.
152  HSC_ServiceUnavailable = 503, ///< Available in HTTP 1.0. The server is temporarily down.
153  HSC_GatewayTimeout = 504, ///< Available in HTTP 1.0. The server is currently acting as a proxy or gateway, and didn't receive a timely response from the upstream server.
154  HSC_HTTPVersionNotSupported = 505, ///< Available in HTTP 1.0. The server does not support the version of HTTP used in the request.
155  HSC_VariantAlsoNegotiates = 506, ///< Available in RFC 2295. A transparent content negotiation for the request would result in a circular reference.
156  HSC_InsufficientStorage = 507, ///< Available in WebDAV/RFC 4918. The server does not have enough space to store the representation for the request from the client.
157  HSC_LoopDetected = 508, ///< Available in WebDAV/RFC 5842. An infinite loop was detected while processing a request.
158  HSC_NotExtended = 510, ///< Available in RFC 2774. Further extensions are required by this server to fulfill the request.
159  HSC_NetAuthenticationRequired = 511 ///< Available in RFC 6585. The client needs to authenticate in order to the connected network. Intended to be used by proxies.
160  };
