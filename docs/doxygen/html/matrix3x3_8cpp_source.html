00144     void Matrix3x3::SetFromAxisAngle(const Vector3& Axis, const Real& Angle)
00145     {
00146         Real Cos = MathTool::Cos(Angle);
00147         Real Sin = MathTool::Sin(Angle);
00148         Real OneMinusCos = 1.0 - Cos;
00149         Real X2 = Axis.X * Axis.X;
00150         Real Y2 = Axis.Y * Axis.Y;
00151         Real Z2 = Axis.Z * Axis.Z;
00152         Real XYM = Axis.X * Axis.Y * OneMinusCos;
00153         Real XZM = Axis.X * Axis.Z * OneMinusCos;
00154         Real YZM = Axis.Y * Axis.Z * OneMinusCos;
00155         Real XSin = Axis.X * Sin;
00156         Real YSin = Axis.Y * Sin;
00157         Real ZSin = Axis.Z * Sin;
00158 
00159         Matrix[0][0] = X2 * OneMinusCos + Cos;
00160         Matrix[0][1] = XYM - ZSin;
00161         Matrix[0][2] = XZM + YSin;
00162         Matrix[1][0] = XYM + ZSin;
00163         Matrix[1][1] = Y2 * OneMinusCos + Cos;
00164         Matrix[1][2] = YZM - XSin;
00165         Matrix[2][0] = XZM - YSin;
00166         Matrix[2][1] = YZM + XSin;
00167         Matrix[2][2] = Z2 * OneMinusCos + Cos;

00198     Quaternion Matrix3x3::GetAsQuaternion() const
00199     {
00200         Real Trace = Matrix[0][0] + Matrix[1][1] + Matrix[2][2];
00201         Quaternion Ret;
00202 
00203         if (Trace > Real(0.0))
00204         {
00205             Real Sc = MathTool::Sqrt(Trace + Real(1.0));
00206             Ret.W = (Sc * Real(0.5));
00207             Sc = Real(0.5) / Sc;
00208 
00209             Ret.X = ( (Matrix[2][1] - Matrix[1][2]) * Sc );
00210             Ret.Y = ( (Matrix[0][2] - Matrix[2][0]) * Sc );
00211             Ret.Z = ( (Matrix[1][0] - Matrix[0][1]) * Sc );
00212         }
00213         else
00214         {
00215             int I = Matrix[0][0] < Matrix[1][1] ?
00216                   ( Matrix[1][1] < Matrix[2][2] ? 2 : 1 ) :
00217                   ( Matrix[0][0] < Matrix[2][2] ? 2 : 0 );
00218             int J = (I + 1) % 3;
00219             int K = (I + 2) % 3;
00220 
00221             Real Sc = MathTool::Sqrt(Matrix[I][I] - Matrix[J][J] - Matrix[K][K] + Real(1.0));
00222             Ret[I] = Sc * Real(0.5);
00223             Sc = Real(0.5) / Sc;
00224 
00225             Ret[J] = ( Matrix[J][I] + Matrix[I][J] ) * Sc;
00226             Ret[K] = ( Matrix[K][I] + Matrix[I][K] ) * Sc;
00227             Ret[3] = ( Matrix[K][J] - Matrix[J][K] ) * Sc;
00228         }
00229         return Ret;
