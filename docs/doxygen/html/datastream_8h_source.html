66  /// @enum StreamFlags
67  /// @brief This enum describes the flags that control certain behaviors of a stream.
68  /// @details It is important to note that not all of these flags are used by all streams.
69  enum StreamFlags
70  {
71  SF_None = 0, ///< Error/no special initialization.
72  SF_Read = std::ios_base::in, ///< Permit read operations on the stream.
73  SF_Write = std::ios_base::out, ///< Permit write operations on the stream.
74  SF_Append = std::ios_base::app, ///< All write operations on the stream are done at the end of the stream.
75  SF_AtEnd = std::ios_base::ate, ///< Moves the starting position of the stream to the end upon initialization.
76  SF_Binary = std::ios_base::binary, ///< Tell the stream that the file in question is Binary.
77  SF_Truncate = std::ios_base::trunc ///< Clear the contents of the file when opening. Note that this will also create the file if it's not found.
78  };

90  /// @brief Base class interface for resource streams.
91  /// @details
92  ///////////////////////////////////////
93  class iStreamBase
94  {
95  public:
96  /// @brief Class constructor.
97  iStreamBase() { }
98  /// @brief Class destructor.
99  virtual ~iStreamBase() { }
100 
101  /// @brief Gets the size of the stream.
102  /// @return Returns the size of this stream in bytes.
103  virtual StreamSize GetSize() const = 0;
104  /// @brief Gets whether or not the current position is at the end of the stream.
105  /// @return Returns true if the current position has reached the end of the stream, false otherwise.
106  virtual Boole EoF() const = 0;
107  /// @brief Gets whether or not a critical error was detected in a previous operation in the stream.
108  /// @return Returns true if a critical error has occurred and the stream integrity may be invalid, false otherwise.
109  virtual Boole Bad() const = 0;
110  /// @brief Gets whether or not an otherwise silent and recoverable error was detected in a previous operation in the stream.
111  /// @return Returns true if a non-critical operation failed (such as seek), false otherwise.
112  virtual Boole Fail() const = 0;
113  /// @brief Gets whether or not this stream is intact and ready for operations.
114  /// @return Returns true if no failures have been detected, false otherwise.
115  virtual Boole IsValid() const = 0;
116  /// @brief Clears any stored error state on the stream.
117  /// @remarks This is useful for non-critical errors such as ones that cause "EoF()" or "Fail()" to return true but not
118  /// "Bad()". Using this to clear critical errors is not advised.
119  virtual void ClearErrors() = 0;
120  };//iStreamBase
