54  Sound::Sound(ConstString& SoundName, ConstString& FileName, ConstString& Group, Audio::SoundType SType)
55  : Type(SType),
56  BaseVolume(1.0)
57  {
58  Manager = AudioManager::GetSingletonPtr();
59  String Extension = FileName.substr(FileName.find_last_of(".")+1);
60  Ogre::DataStreamPtr Stream = Ogre::ResourceGroupManager::getSingleton().openResource(FileName,Group);
61  char* buffer = new char[Stream->size()];
62  Stream->read((void*)buffer, Stream->size());
63 
64  SoundSource = Manager->GetcAudioManager()->createFromMemory(SoundName.c_str(),buffer,Stream->size(),Extension.c_str());
65  if (!SoundSource)
66  { MEZZ_EXCEPTION(Exception::INTERNAL_EXCEPTION,"Failed to create sound from file."); }
67 
68  delete[] buffer;
69  UpdateVolume();
70  }
