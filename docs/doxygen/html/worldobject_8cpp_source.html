87  void WorldObject::ProtoSerializeProperties(XML::Node& SelfRoot) const
88  {
89  XML::Node PropertiesNode = SelfRoot.AppendChild( WorldObject::GetSerializableName() + "Properties" );
90 
91  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
92  PropertiesNode.AppendAttribute("Name").SetValue( this->GetName() ) )
93  {
94  XML::Node LocationNode = PropertiesNode.AppendChild("Location");
95  this->GetLocation().ProtoSerialize( LocationNode );
96  XML::Node OrientationNode = PropertiesNode.AppendChild("Orientation");
97  this->GetOrientation().ProtoSerialize( OrientationNode );
98  XML::Node ScaleNode = PropertiesNode.AppendChild("Scale");
99  this->GetScale().ProtoSerialize( ScaleNode );
100 
101  return;
102  }else{
103  SerializeError("Create XML Attribute Values",WorldObject::GetSerializableName() + "Properties",true);
104  }

113  void WorldObject::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
114  {
115  XML::Attribute CurrAttrib;
116  XML::Node PropertiesNode = SelfRoot.GetChild( WorldObject::GetSerializableName() + "Properties" );
117 
118  if( !PropertiesNode.Empty() ) {
119  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
120  CurrAttrib = PropertiesNode.GetAttribute("Name");
121  if( !CurrAttrib.Empty() )
122  this->ObjectName = CurrAttrib.AsString();
123 
124  XML::Node PositionNode = PropertiesNode.GetChild("Location").GetFirstChild();
125  if( !PositionNode.Empty() ) {
126  Vector3 Loc(PositionNode);
127  this->SetLocation(Loc);
128  }
129 
130  XML::Node OrientationNode = PropertiesNode.GetChild("Orientation").GetFirstChild();
131  if( !OrientationNode.Empty() ) {
132  Quaternion Rot(OrientationNode);
133  this->SetOrientation(Rot);
134  }
135 
136  XML::Node ScaleNode = PropertiesNode.GetChild("Scale").GetFirstChild();
137  if( !ScaleNode.Empty() ) {
138  Vector3 Scale(ScaleNode);
139  this->SetScale(Scale);
140  }
141  }else{
142  MEZZ_EXCEPTION(ExceptionBase::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (WorldObject::GetSerializableName() + "Properties" ) + ": Not Version 1.");
143  }
144  }else{
145  MEZZ_EXCEPTION(ExceptionBase::II_IDENTITY_NOT_FOUND_EXCEPTION,WorldObject::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
146  }
147  }

158  void WorldObject::_NotifyCollisionState(Physics::Collision* Col, const Physics::CollisionState State)
159  {
160  if(Physics::Col_Begin == State) {
161  this->CurrentCollisions.insert(Col);
162  }else if(Physics::Col_End == State) {
163  std::set<Physics::Collision*>::iterator ColIt = this->CurrentCollisions.find(Col);
164  if( ColIt != this->CurrentCollisions.end() )

166  }
