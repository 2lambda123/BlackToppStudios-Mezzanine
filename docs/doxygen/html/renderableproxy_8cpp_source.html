218  void RenderableProxy::ProtoSerializeProperties(XML::Node& SelfRoot) const
219  {
220  this->WorldProxy::ProtoSerializeProperties(SelfRoot);
221 
222  XML::Node PropertiesNode = SelfRoot.AppendChild( RenderableProxy::GetSerializableName() + "Properties" );
223 
224  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
225  PropertiesNode.AppendAttribute("Visible").SetValue( this->GetVisible() ? "true" : "false" ) &&
226  PropertiesNode.AppendAttribute("CastShadows").SetValue( this->GetCastShadows() ? "true" : "false" ) &&
227  PropertiesNode.AppendAttribute("LightMask").SetValue( this->GetLightMask() ) &&
228  PropertiesNode.AppendAttribute("VisibilityMask").SetValue( this->GetVisibilityMask() ) &&
229  PropertiesNode.AppendAttribute("QueryMask").SetValue( this->GetQueryMask() ) &&
230  PropertiesNode.AppendAttribute("RenderDistance").SetValue( this->GetRenderDistance() ) )
231  {
232  return;
233  }else{
234  SerializeError("Create XML Attribute Values",RenderableProxy::GetSerializableName() + "Properties",true);
235  }
236  }
237 
238  void RenderableProxy::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
239  {
240  this->WorldProxy::ProtoDeSerializeProperties(SelfRoot);
241 
242  XML::Attribute CurrAttrib;
243  XML::Node PropertiesNode = SelfRoot.GetChild( RenderableProxy::GetSerializableName() + "Properties" );
244 
245  if( !PropertiesNode.Empty() ) {
246  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
247  CurrAttrib = PropertiesNode.GetAttribute("Visible");
248  if( !CurrAttrib.Empty() )
249  this->SetVisible( StringTools::ConvertToBool( CurrAttrib.AsString() ) );
250 
251  CurrAttrib = PropertiesNode.GetAttribute("CastShadows");
252  if( !CurrAttrib.Empty() )
253  this->SetCastShadows( StringTools::ConvertToBool( CurrAttrib.AsString() ) );
254 
255  CurrAttrib = PropertiesNode.GetAttribute("LightMask");
256  if( !CurrAttrib.Empty() )
257  this->SetLightMask( CurrAttrib.AsWhole() );
258 
259  CurrAttrib = PropertiesNode.GetAttribute("VisibilityMask");
260  if( !CurrAttrib.Empty() )
261  this->SetVisibilityMask( CurrAttrib.AsWhole() );
262 
263  CurrAttrib = PropertiesNode.GetAttribute("QueryMask");
264  if( !CurrAttrib.Empty() )
265  this->SetQueryMask( CurrAttrib.AsWhole() );
266 
267  CurrAttrib = PropertiesNode.GetAttribute("RenderDistance");
268  if( !CurrAttrib.Empty() )
269  this->SetRenderDistance( CurrAttrib.AsReal() );
270  }else{
271  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (RenderableProxy::GetSerializableName() + "Properties" ) + ": Not Version 1.");
272  }
273  }else{
274  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,RenderableProxy::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
275  }
