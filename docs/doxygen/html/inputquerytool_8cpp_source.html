00103     void InputQueryTool::GatherEvents(bool ClearEventsFromEventMgr)
00104     {
00105         int H = 0;
00106         int V = 0;
00107         if(!CodeCache.empty())
00108         {
00109             H = CodeCache[MetaCode::MOUSEABSOLUTEHORIZONTAL];
00110             V = CodeCache[MetaCode::MOUSEABSOLUTEVERTICAL];
00111         }
00112         CodeCache.clear();
00113         CodeCache.insert(CodeCache.end(), MetaCode::INPUTEVENT_LAST, 0);
00114         CodeCache[MetaCode::MOUSEABSOLUTEHORIZONTAL] = H;
00115         CodeCache[MetaCode::MOUSEABSOLUTEVERTICAL] = V;
00116 
00117         std::list<EventUserInput*>* UserInput = EventManager::GetSingletonPtr()->GetAllUserInputEvents();   // Get the updated list of events
00118 
00119         if( ClearEventsFromEventMgr )
00120             { EventManager::GetSingletonPtr()->RemoveAllSpecificEvents(EventBase::UserInput); }
00121 
00122         MetaCode::InputCode ThisCode;
00123         for(std::list<EventUserInput*>::iterator Iter = UserInput->begin(); Iter!=UserInput->end(); Iter++) //for each event
00124         {
00125             for(unsigned int c = 0; c<(*Iter)->GetMetaCodeCount(); c++) //For each metacode in the event
00126             {
00127                 ThisCode = (*Iter)->GetMetaCode(c).GetCode() ;
00128                 if( ThisCode==MetaCode::MOUSEVERTICAL || ThisCode==MetaCode::MOUSEHORIZONTAL )
00129                 {
00130                     CodeCache[ThisCode] += (*Iter)->GetMetaCode(c).GetMetaValue();
00131                 }else{
00132                     CodeCache[ThisCode] = (*Iter)->GetMetaCode(c).GetMetaValue();
00133                 }
00134             }
00135         }
00136 
00137         if( ClearEventsFromEventMgr )//Erase everything if we were asked to.
00138         {
00139             for(std::list<EventUserInput*>::iterator Iter = UserInput->begin(); !UserInput->empty(); Iter = UserInput->begin())
00140             {
00141                 delete(*Iter);
00142                 UserInput->remove(*Iter);
00143             }
00144         }
00145     }

00147 }
