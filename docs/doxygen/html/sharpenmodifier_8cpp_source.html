93  void SharpenModifier::Modify(TextureBuffer& Buffer)
94  {
95  ConvolutionModifier Filter;
96  switch(this->Type)
97  {
98  default:
99  case ST_Normal:
100  {
101  Filter.SetKernel( Matrix3x3(0.0f,-1.0f,0.0f,-1.0f,5.0f,-1.0f,0.0f,-1.0f,0.0f) ).CalculateDivisor();
102  break;
103  }
104  case ST_Gaussian:
105  {
106  Real fSigma = 0.5f + ( (5.0f - 0.5f) / 255.0f ) * static_cast<Real>(this->SharpenSigma);
107  Integer Radius = static_cast<Integer>(this->SharpenBlockSize) / 2;
108  Real Min = MathTools::Exp( static_cast<Real>(2 * Radius * Radius) / (-2.0 * fSigma * fSigma) ) / (MathTools::GetTwoPi() * fSigma * fSigma);
109  Integer* Kernel = new Integer[ this->SharpenBlockSize * this->SharpenBlockSize ];
110  Integer Sum = 0;
111  Integer Y = -Radius;
112  Integer X = -Radius;
113  for( Integer i = 0 ; i < this->SharpenBlockSize ; i++ )
114  {
115  for( Integer j = 0 ; j < this->SharpenBlockSize ; j++ )
116  {
117  Kernel[ i * this->SharpenBlockSize + j ] = static_cast<Integer>( ( MathTools::Exp( static_cast<Real>( X * X + Y * Y ) / ( -2.0 * fSigma * fSigma ) ) / ( MathTools::GetTwoPi() * fSigma * fSigma ) ) / Min );
118  Sum += Kernel[ i * this->SharpenBlockSize + j ];
119  X++;
120  }
121  Y++;
122  }
123  Integer c = static_cast<Integer>(this->SharpenBlockSize) >> 1;
124  Integer Divisor = 0;
125  for( Integer i = 0 ; i < this->SharpenBlockSize ; i++ )
126  {
127  for( Integer j = 0 ; j < this->SharpenBlockSize ; j++ )
128  {
129  if( (i == c) && (j == c) ) {
130  Kernel[ i * this->SharpenBlockSize + j ] = 2 * Sum - Kernel[ i * this->SharpenBlockSize + j ];
131  }else{
132  Kernel[ i * this->SharpenBlockSize + j ] = -Kernel[ i * this->SharpenBlockSize + j ];
133  }
134  Divisor += Kernel[ i * this->SharpenBlockSize + j ];
135  }
136  }
137  Filter.SetKernel(this->SharpenBlockSize,Kernel).SetDivisor( static_cast<Real>(Divisor) );
138  delete[] Kernel;
139  break;
140  }
141  }
142  Filter.SetIncludeAlphaChannel(true).Modify(Buffer);
143  }
