00080             bool DoNotBreak = true;

00082             {
00083                 if(XCursor + (FixedCellSize.X + CellSpacing.X) < XSpace)
00084                 {
00085                     XCursor+=(FixedCellSize.X + CellSpacing.X);
00086                     XCellsPerPage++;

00088                 else if(XCursor + FixedCellSize.X < XSpace)
00089                 {
00090                     XCursor+=FixedCellSize.X;
