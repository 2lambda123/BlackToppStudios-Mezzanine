00265         Vector2 RenderableContainerWidget::CalculateOffset(const Vector2& NewSize, const Vector2& OldSize, const Vector2& EleOffset, const UI::ResizeableAnchor& Anchor)
00266         {
00267             Vector2 NewOffset;
00268             switch (Anchor)
00269             {
00270                 case RA_AnchorTopLeft:
00271                 {
00272                     NewOffset = EleOffset;
00273                     break;
00274                 }
00275                 case RA_AnchorTop:
00276                 {
00277                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00278                     NewOffset.Y = EleOffset.Y;
00279                     break;
00280                 }
00281                 case RA_AnchorTopRight:
00282                 {
00283                     NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
00284                     NewOffset.Y = EleOffset.Y;
00285                     break;
00286                 }
00287                 case RA_AnchorLeft:
00288                 {
00289                     NewOffset.X = EleOffset.X;
00290                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00291                     break;
00292                 }
00293                 case RA_AnchorMiddle:
00294                 {
00295                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00296                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00297                     break;
00298                 }
00299                 case RA_AnchorRight:
00300                 {
00301                     NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
00302                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00303                     break;
00304                 }
00305                 case RA_AnchorBottomLeft:
00306                 {
00307                     NewOffset.X = EleOffset.X;
00308                     NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
00309                     break;
00310                 }
00311                 case RA_AnchorBottom:
00312                 {
00313                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00314                     NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
00315                     break;
00316                 }
00317                 case RA_AnchorBottomRight:
00318                 {
00319                     NewOffset = EleOffset + (NewSize - OldSize);
00320                     break;
00321                 }
00322                 default:
00323                     return NewOffset;
00324             }
00325             return NewOffset;
00326         }
