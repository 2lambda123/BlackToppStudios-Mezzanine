00112                 RelSize = Rect.Size / Parent->GetParent()->GetViewportDimensions();
00113 
00114                 SelectionRect.Position = Rect.Position;
00115                 SelectionRect.Size.X = Rect.Size.X - Rect.Size.Y;
00116                 SelectionRect.Size.Y = Rect.Size.Y;

00120                 ListToggleRect.Position.Y = Rect.Position.Y;
