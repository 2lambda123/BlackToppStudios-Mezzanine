93  void RandomPixelsModifier::Modify(TextureBuffer& Buffer)
94  {
95  IntVector2 Point;
96  std::vector<IntVector2> PixelList;
97 
98  MathTools::MersenneTwisterGenerator32 NumGen(this->GeneratorSeed);
99  Whole Area = Buffer.GetWidth() * Buffer.GetHeight();
100  Whole GenerateCount = this->PixelCount;
101  if( GenerateCount == 0 ) {
102  GenerateCount = ( static_cast<Whole>( MathTools::Sqrt( static_cast<Real>( Buffer.GetWidth() ) ) ) + static_cast<Whole>( MathTools::Sqrt( static_cast<Real>( Buffer.GetHeight() ) ) ) ) * 10;
103  }
104 
105  if( GenerateCount == Area ) {
106  RectangleModifier().SetColour(this->PixelColour).Modify(Buffer);
107  return;
108  }
109 
110  while( PixelList.size() != GenerateCount )
111  {
112  Point.X = NumGen.GenerateUInt() % Buffer.GetWidth();
113  Point.Y = NumGen.GenerateUInt() % Buffer.GetHeight();
114 
115  Boole InList = false;
116  for( std::vector<IntVector2>::iterator iter = PixelList.begin() ; iter != PixelList.end() ; iter++ )
117  {
118  if( iter->X == Point.X && iter->Y == Point.Y ) {
119  InList = true;
120  break;
121  }
122  }
123 
124  if( !InList ) {
125  PixelList.push_back(Point);
126  Buffer.SetPixel(Point.X, Point.Y, this->PixelColour);
127  }
128  }
129  }
