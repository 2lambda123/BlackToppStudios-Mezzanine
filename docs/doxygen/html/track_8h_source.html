56  /// @brief A base type that provides container features for different tracks
57  /// @details Tracks are containers of a Discrete set of points, that are presented as
58  /// a continuous range from 0 to 1. Interpolators are used to generate the data between
59  /// the points as it is needed. For example, lets say there is an interpolator provided
60  /// for integers and a track contains only two data points 0 and 100, requesting 0.5
61  /// might return 50.
62  /// @n @n
63  /// This uses std::vector underneath for its performance characteristics.

225  /// @brief Convert this to a node on an XML graph
226  /// @param CurrentRoot the node that will become the parent for the one this creates.
227  /// @details This function is also used to serialize Looped Tracks.
228  /// This serializes the interpoator in use and each data point.
229  virtual void ProtoSerialize(XML::Node& CurrentRoot) const
230  {
231  Mezzanine::XML::Node TrackNode = CurrentRoot.AppendChild(DerivedSerializableName());
232 
233  if(TrackNode)
234  {
235  Mezzanine::XML::Attribute VersionAttr = TrackNode.AppendAttribute("Version");
236  if( VersionAttr )
237  {
238  if( !VersionAttr.SetValue("1") )
239  { SerializeError("Create XML Version Attribute Values", DerivedSerializableName()); }
240  }else{
241  SerializeError("Create XML Version Attributes", DerivedSerializableName());
242  }
243 
244 
245  Mezzanine::XML::Attribute NameAttr = TrackNode.AppendAttribute("Name");
246  if( NameAttr )
247  {
248  if( !NameAttr.SetValue(GetTrackName()) )
249  { SerializeError("Create XML Name Values", DerivedSerializableName()); }
250  }else{
251  SerializeError("Create XML Name Attributes", DerivedSerializableName());
252  }
253 
254  Mezzanine::XML::Node InterpolatorNode = TrackNode.AppendChild("Interpolator");
255  if(InterpolatorNode)
256  {
257  InterpolatorType::ProtoSerialize(InterpolatorNode);
258  }else{
259  SerializeError("Create XML Interpolator Node", DerivedSerializableName());
260  }
261 
262  Mezzanine::XML::Node DataNode = TrackNode.AppendChild("DataPoints");
263  if(DataNode)
264  {
265  for(typename DataContainerType::const_iterator Iter = DataPoints.begin();
266  DataPoints.end()!=Iter;
267  ++Iter)
268  { Iter->ProtoSerialize(DataNode);}
269  }else{
270  SerializeError("Create XML DataPoints Node", DerivedSerializableName());
271  }
272 
273  }else{
274  SerializeError("Create XML Serialization Node", DerivedSerializableName());
275  }
276  }

278  /// @brief Convert a node on an XML into a track.
279  /// @details This will convert and XML into Track or LoopedTrack templated
280  /// on the type that matches this instance of this class.
281  virtual void ProtoDeSerialize(const XML::Node& OneNode)
282  {
283  if ( String(OneNode.Name())==String(DerivedSerializableName()) )
284  {
285  if(OneNode.GetAttribute("Version").AsInt() == 1)
286  {
287  Mezzanine::XML::Node InterpolatorNode = OneNode.GetChild("Interpolator").GetFirstChild();
288  if(InterpolatorNode.Name() == InterpolatorType::SerializableName())
289  {
290  InterpolatorType::ProtoDeSerialize(InterpolatorNode);
291  }else{
292  //DeSerializeError(); // Maybe use this instead?
293  MEZZ_EXCEPTION(Exception::II_IDENTITY_INVALID_EXCEPTION,"Incompatible Interpolator Type Version for " + DerivedSerializableName() + ": Not " + InterpolatorType::SerializableName());
294  }
295 
296  SetTrackNameUnique(OneNode.GetAttribute("Name").AsString());
297  Mezzanine::XML::Node DataNode = OneNode.GetChild("DataPoints");
298  if(DataNode)
299  {
300  XML::Node Iter=DataNode.GetFirstChild();
301  while(Iter)
302  {
303  InterpolatableType Scratch;
304  Scratch.ProtoDeSerialize(Iter);
305  DataPoints.push_back(Scratch);
306  Iter = Iter.GetNextSibling();
307  }
308  }// No else the track could be empty
309  }else{
310  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + DerivedSerializableName() + ": Not Version 1.");
311  }
312  }else{
313  MEZZ_EXCEPTION(Exception::II_IDENTITY_INVALID_EXCEPTION,"Attempting to deserialize a " + DerivedSerializableName() + ", found a " + String(OneNode.Name()) + ".");
314  }
315 
316  }

331  /// @brief A track that keeps an extra data point in the track to make sure it loops.
332  /// @details when Add or push_back is called and there are 2 or more points the track,
333  /// this seamlessly adds a copy of the first data point to the end of the track. When
334  /// Iterating with a Smooth iterator this creates the impression of a loop inside the simulation.
335  template <typename InterpolatorType>
336  class TrackLooped : public Track<InterpolatorType>
337  {
338  protected:
339  /// @brief The base most class of this type.
340  typedef Track<InterpolatorType> BaseType;
341 
342  public:
343  /// @brief The type this class and the interpolator it uses works with.
344  typedef typename InterpolatorType::InterpolatableType InterpolatableType;
345  /// @brief The type of the Container storing the interpolatable data. This is a single point to change all the tracks
346  typedef std::vector<InterpolatableType> DataContainerType;
347  /// @brief An iterator than can take an arbitrary amount of steps by interpolation.
348  typedef SmoothTrackIterator<InterpolatableType> SmoothIteratorType;
349 
350  /// @brief Create a Track from a range of data points, and enforce its being a loop.
351  /// @param Begin An iterator pointing to the beginning of a range to copy.
352  /// @param End an iterator pointing to one past the rang to copy.
353  TrackLooped(typename DataContainerType::iterator Begin,
354  typename DataContainerType::iterator End)
355  : BaseType(Begin,End)
356  { EnforceLoop(); }
357  /// @brief Create a track from a DataContainerType instance, likely a vector and copy the data from it, and enforce it being a loop
358  /// @param DataSet A collection of data points to copy.
359  /// @details if the last item does not match the first this fixes that.
360  TrackLooped(const DataContainerType& DataSet)
361  : BaseType(DataSet)
362  { EnforceLoop(); }
363  /// @brief Create a default empty looped track.
364  TrackLooped()
365  {}
366  /// @brief Virtual Deconstructor.
367  virtual ~TrackLooped()
368  {}
369 
370  /// @brief If this track has more than one datapoint, does the first match the last, if not fix it.
371  void EnforceLoop()
372  {
373  if(BaseType::DataPoints.size()>1 && *(BaseType::DataPoints.end()-1)!=*(BaseType::DataPoints.begin()))
374  { BaseType::DataPoints.push_back(*(BaseType::DataPoints.begin())); }
375  }
376 
377  /// @brief Add another data point, and preserve the track's loopwhile doing so.
378  /// @param AddedValue The data point to add.
379  virtual void push_back(const InterpolatableType& AddedValue)
380  {
381  if(BaseType::DataPoints.size()>1)
382  {
383  *(BaseType::DataPoints.end()-1) = AddedValue;
384  BaseType::DataPoints.push_back(*BaseType::DataPoints.begin());
385  }else if(BaseType::DataPoints.size()==1){
386  BaseType::DataPoints.push_back(AddedValue);
387  BaseType::DataPoints.push_back(*BaseType::DataPoints.begin());
388  }else{
389  BaseType::DataPoints.push_back(AddedValue);
390  }
391  }

400  static String SerializableName()
401  { return String("TrackLooped"); }
