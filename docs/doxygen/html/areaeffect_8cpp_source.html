00399     {
00400         Ogre::SceneManager* OgreManager = World::GetWorldPointer()->GetSceneManager()->GetGraphicsWorldPointer();
00401         PreGraphicsMeshCreate();

00403         {
00404             delete Shape;

00408         GraphicsObject = OgreManager->createEntity(Name, Filename, Group);
00409         Ogre::MeshPtr myMesh = GraphicsObject->getMesh();

00411         Ogre::IndexData*  indexData = subMesh->indexData;
00412         Ogre::VertexData* vertexData = subMesh->vertexData;
00413 
00414         const Ogre::VertexElement* posElem = vertexData->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
00415         Ogre::HardwareVertexBufferSharedPtr vBuffer = vertexData->vertexBufferBinding->getBuffer(posElem->getSource());
00416         Ogre::HardwareIndexBufferSharedPtr iBuffer = indexData->indexBuffer;
00417 
00418         unsigned int triCount = indexData->indexCount/3;
00419         Ogre::Vector3* vertices = new Ogre::Vector3[vertexData->vertexCount];
00420         unsigned int* indices = new unsigned int[indexData->indexCount];
00421         unsigned char* vertex = static_cast<unsigned char*>(vBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00422         float* pReal = NULL;
00423         for (size_t j = 0; j < vertexData->vertexCount; j++, vertex += vBuffer->getVertexSize() )
00424         {
00425             posElem->baseVertexPointerToElement(vertex, &pReal);
00426             vertices[j].x = *pReal++;
00427             vertices[j].y = *pReal++;
00428             vertices[j].z = *pReal++;
00429         }
00430         vBuffer->unlock();
00431         size_t index_offset = 0;
00432         bool use32bitindexes = (iBuffer->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
00433 
00434         unsigned long* pLong = static_cast<unsigned long*>(iBuffer->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
00435         unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
00436 
00437         if (use32bitindexes)
00438         {
00439             for (size_t k = 0; k < triCount*3; ++k)
00440             {
00441                 indices[index_offset++] = pLong[k];
00442             }
00443         }
00444         else
00445         {
00446             for (size_t k = 0; k < triCount*3; ++k)

00767     void GravityWell::ApplyEffect()
00768     {
00769         if(0 == Strength)
00770             return;
00771         ActorBase* Act = NULL;
00772         ActorRigid* ActRig = NULL;
00773         if(!AllowWorldGrav && !AddedActors.empty())
00774         {
00775             for ( std::vector<ActorBase*>::iterator AA = AddedActors.begin() ; AA != AddedActors.end() ; AA++ )
00776             {
00777                 if(ActorBase::Actorrigid != (*AA)->GetType())
00778                     continue;
00779                 ActRig = dynamic_cast<ActorRigid*>(*AA);
00780                 ActRig->GetPhysicsSettings()->SetIndividualGravity(Vector3());
00781             }
00782         }
00783         if(!OverlappingActors.empty())
00784         {
00785             Vector3 ActorLoc, Direction;
00786             Real Distance, AppliedStrength, InvMass;
00787             Vector3 GhostLoc = this->GetLocation();
00788             for ( std::list<ActorBase*>::iterator OA = OverlappingActors.begin() ; OA != OverlappingActors.end() ; OA++ )
00789             {
00790                 if(ActorBase::Actorrigid != (*OA)->GetType())
00791                     continue;
00792                 //Collect necessary data
00793                 ActorLoc = (*OA)->GetLocation();
00794                 Distance = ActorLoc.Distance(GhostLoc);
00795                 Direction = (GhostLoc - ActorLoc) / Distance;
00796                 switch(AttenStyle)
00797                 {
00798                     case phys::Att_Linear:
00799                         AppliedStrength = Strength - (AttenAmount * Distance);
00800                         break;
00801                     case phys::Att_Quadratic:
00802                         AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
00803                         break;
00804                     default:
00805                         AppliedStrength = Strength;
00806                         break;
00807                 }
00808                 ActRig = static_cast<ActorRigid*>(*OA);
00809                 InvMass = ActRig->GetBulletObject()->getInvMass();
00810                 if(0 != InvMass)
00811                     AppliedStrength *= (1 / ActRig->GetBulletObject()->getInvMass());
00812                 else
00813                     AppliedStrength = 0;
00814                 if(0 > AppliedStrength)
00815                     AppliedStrength = 0;
00816                 //Apply the Force
00817                 ActRig->GetBulletObject()->applyCentralForce((Direction * AppliedStrength).GetBulletVector3());
00818             }
00819         }
00820         if(!AllowWorldGrav && !RemovedActors.empty())
00821         {
00822             Vector3 WorldGrav = TheWorld->GetPhysicsManager()->GetGravity();
00823             for ( std::vector<ActorBase*>::iterator RA = RemovedActors.begin() ; RA != RemovedActors.end() ; RA++ )
00824             {
