00566     void GravityWell::ApplyEffect()
00567     {
00568         if(0 == Strength)
00569             return;
00570         ActorBase* Act = NULL;
00571         ActorRigid* ActRig = NULL;
00572         if(!AllowWorldGrav && !AddedActors.empty())
00573         {
00574             for ( std::vector<ActorBase*>::iterator AA = AddedActors.begin() ; AA != AddedActors.end() ; AA++ )
00575             {
00576                 if(ActorBase::Actorrigid != (*AA)->GetType())
00577                     continue;
00578                 ActRig = dynamic_cast<ActorRigid*>(*AA);
00579                 ActRig->GetPhysicsSettings()->SetIndividualGravity(Vector3());
00580             }
00581         }
00582         if(!OverlappingActors.empty())
00583         {
00584             Vector3 ActorLoc, Direction;
00585             Real Distance, AppliedStrength, InvMass;
00586             Vector3 GhostLoc = this->GetLocation();
00587             for ( std::list<ActorBase*>::iterator OA = OverlappingActors.begin() ; OA != OverlappingActors.end() ; OA++ )
00588             {
00589                 if(ActorBase::Actorrigid != (*OA)->GetType())
00590                     continue;
00591                 //Collect necessary data
00592                 ActorLoc = (*OA)->GetLocation();
00593                 Distance = ActorLoc.Distance(GhostLoc);
00594                 Direction = (GhostLoc - ActorLoc) / Distance;
00595                 switch(AttenStyle)
00596                 {
00597                     case Mezzanine::Att_Linear:
00598                         AppliedStrength = Strength - (AttenAmount * Distance);
00599                         break;
00600                     case Mezzanine::Att_Quadratic:
00601                         AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
00602                         break;
00603                     default:
00604                         AppliedStrength = Strength;
00605                         break;
00606                 }
00607                 ActRig = static_cast<ActorRigid*>(*OA);
00608                 InvMass = ActRig->GetBulletObject()->getInvMass();
00609                 if(0 != InvMass)
00610                     AppliedStrength *= (1 / ActRig->GetBulletObject()->getInvMass());
00611                 else
00612                     AppliedStrength = 0;
00613                 if(0 > AppliedStrength)
00614                     AppliedStrength = 0;
00615                 //Apply the Force
00616                 ActRig->GetBulletObject()->applyCentralForce((Direction * AppliedStrength).GetBulletVector3());
00617             }
00618         }
00619         if(!AllowWorldGrav && !RemovedActors.empty())
00620         {
00621             Vector3 WorldGrav = PhysicsManager::GetSingletonPtr()->GetGravity();
00622             for ( std::vector<ActorBase*>::iterator RA = RemovedActors.begin() ; RA != RemovedActors.end() ; RA++ )
00623             {
00624                 if(ActorBase::Actorrigid != (*RA)->GetType())
00625                     continue;
00626                 ActRig = dynamic_cast<ActorRigid*>(*RA);
00627                 ActRig->GetPhysicsSettings()->SetIndividualGravity(WorldGrav);
00628             }
00629         }
00630     }
