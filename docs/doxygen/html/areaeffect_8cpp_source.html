00562     void GravityWell::ApplyEffect()
00563     {
00564         if(0 == Strength)
00565             return;
00566         ActorBase* Act = NULL;
00567         ActorRigid* ActRig = NULL;
00568         if(!AllowWorldGrav && !AddedActors.empty())
00569         {
00570             for ( std::vector<ActorBase*>::iterator AA = AddedActors.begin() ; AA != AddedActors.end() ; AA++ )
00571             {
00572                 if(ActorBase::Actorrigid != (*AA)->GetType())
00573                     continue;
00574                 ActRig = dynamic_cast<ActorRigid*>(*AA);
00575                 ActRig->GetPhysicsSettings()->SetIndividualGravity(Vector3());
00576             }
00577         }
00578         if(!OverlappingActors.empty())
00579         {
00580             Vector3 ActorLoc, Direction;
00581             Real Distance, AppliedStrength, InvMass;
00582             Vector3 GhostLoc = this->GetLocation();
00583             for ( std::list<ActorBase*>::iterator OA = OverlappingActors.begin() ; OA != OverlappingActors.end() ; OA++ )
00584             {
00585                 if(ActorBase::Actorrigid != (*OA)->GetType())
00586                     continue;
00587                 //Collect necessary data
00588                 ActorLoc = (*OA)->GetLocation();
00589                 Distance = ActorLoc.Distance(GhostLoc);
00590                 Direction = (GhostLoc - ActorLoc) / Distance;
00591                 switch(AttenStyle)
00592                 {
00593                     case phys::Att_Linear:
00594                         AppliedStrength = Strength - (AttenAmount * Distance);
00595                         break;
00596                     case phys::Att_Quadratic:
00597                         AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
00598                         break;
00599                     default:
00600                         AppliedStrength = Strength;
00601                         break;
00602                 }
00603                 ActRig = static_cast<ActorRigid*>(*OA);
00604                 InvMass = ActRig->GetBulletObject()->getInvMass();
00605                 if(0 != InvMass)
00606                     AppliedStrength *= (1 / ActRig->GetBulletObject()->getInvMass());
00607                 else
00608                     AppliedStrength = 0;
00609                 if(0 > AppliedStrength)
00610                     AppliedStrength = 0;
00611                 //Apply the Force
00612                 ActRig->GetBulletObject()->applyCentralForce((Direction * AppliedStrength).GetBulletVector3());
00613             }
00614         }
00615         if(!AllowWorldGrav && !RemovedActors.empty())
00616         {
00617             Vector3 WorldGrav = TheWorld->GetPhysicsManager()->GetGravity();
00618             for ( std::vector<ActorBase*>::iterator RA = RemovedActors.begin() ; RA != RemovedActors.end() ; RA++ )
00619             {
00620                 if(ActorBase::Actorrigid != (*RA)->GetType())
00621                     continue;
00622                 ActRig = dynamic_cast<ActorRigid*>(*RA);
00623                 ActRig->GetPhysicsSettings()->SetIndividualGravity(WorldGrav);
00624             }
00625         }
00626     }
