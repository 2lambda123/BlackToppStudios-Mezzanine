00053         Sound::Sound(ConstString& SoundName, ConstString& FileName, ConstString& Group, Audio::SoundType SType)
00054             : Type(SType),
00055               BaseVolume(1.0)
00056         {
00057             Manager = AudioManager::GetSingletonPtr();
00058             String Extension = FileName.substr(FileName.find_last_of(".")+1);
00059             Ogre::DataStreamPtr Stream = Ogre::ResourceGroupManager::getSingleton().openResource(FileName,Group);
00060             char* buffer = new char[Stream->size()];
00061             Stream->read((void*)buffer, Stream->size());
00062 
00063             SoundSource = Manager->GetcAudioManager()->createFromMemory(SoundName.c_str(),buffer,Stream->size(),Extension.c_str());
00064             delete[] buffer;
00065             UpdateVolume();
00066         }
