77  Boole HTTPMessage::ParseHTTPVersion(const String& ToParse)
78  {
79  if( ToParse.length() == 8 &&
80  ToParse.at(6) == '.' &&
81  StringTools::StartsWith( ToParse, "http/", false ) )

83  this->MessageVersion.Major = static_cast<UInt16>( ToParse.at(5) - '0' );

90  Boole HTTPMessage::ParseHTTPFields(StringIterator& CurrIt, const StringIterator EndIt)
91  {
92  this->MessageFields.clear();
93  Boole PreColon = true;
94  String FieldName;
95  String FieldValue;
96  while( CurrIt != EndIt && this->MessageFields.size() < this->MaxHeaderFields )
97  {
98  switch( *CurrIt )
99  {
100  case '\r':
101  {
102  // Just skip
103  break;
104  }
105  case '\n':
106  {
107  PreColon = true;
108  StringTools::ToLowerCase(FieldName);
109  StringTools::Trim(FieldName,true,true);
110  StringTools::Trim(FieldValue,true,true);
111  this->MessageFields.insert(std::pair<String,String>(FieldName,FieldValue));
112  FieldName.clear();
113  FieldValue.clear();
114  if( *(CurrIt + 1) == '\r' && *(CurrIt + 2) == '\n' ) {
115  CurrIt += 3;
116  return true;
117  }
118  break;
119  }
120  case ':':
121  {
122  PreColon = false;
123  break;
124  }
125  default:
126  {
127  if( PreColon ) FieldName.push_back( *CurrIt );
128  else FieldValue.push_back( *CurrIt );
129  break;
130  }
131  }
132  ++CurrIt;
133  }
134  return false;
135  }

137  Boole HTTPMessage::GetMessageComponent(StringIterator& CurrIt, const StringIterator EndIt, String& Component)
138  {
139  while( (*CurrIt) != ' ' && (*CurrIt) != '\r' && CurrIt != EndIt )
140  {
141  Component.push_back( *CurrIt );
142  ++CurrIt;
143  }
144  Boole EoL = false;
145  if( CurrIt != EndIt ) {
146  // Increment passed the end of the line if we're there. Or just increment into the next space valid character if we're on a space.
147  if( (*CurrIt) == '\r' && *(CurrIt + 1) == '\n' ) {
148  EoL = true;
149  CurrIt += 2;
150  }else if( (*CurrIt) == ' ' ) {
151  ++CurrIt;
152  }
153  }
154  return EoL;
155  }
