00141     void ActorRigid::_NotifyCollisionState(Collision* Col, const Collision::CollisionState& State)
00142     {
00143         WorldObject::_NotifyCollisionState(Col,State);
00144         StickyData* StickyD = GetPhysicsSettings()->GetStickyData();
00145         bool UseA = Col->GetObjectA() != this;
00146         // We don't care if sticky behavior isn't set or if the collision has ended.
00147         // If it's ended, then we've probably already done our logic.
00148         if(0 == StickyD->MaxNumContacts || Collision::Col_End == State)
00149             return;
00150         // Constraints only work well with other rigid bodies, so for now just other actorrigids.
00151         /// @todo Update this to be workable with other objects that have rigid bodies internally, and maybe soft bodies.
00152         if( (UseA ? (Col->GetObjectA()->GetType() != Mezzanine::WOT_ActorRigid) : (Col->GetObjectB()->GetType() != Mezzanine::WOT_ActorRigid)) )
00153             return;
00154         // We need a contact point to be present for this to work, since a collision without contact points is an AABB overlap.
00155         // So confirm there are contact points.
00156         if(Col->GetNumContactPoints() < 1)
00157             return;
00158         // Do we have room for additional sticky constraints?
00159         if(StickyD->StickyConstraints.size() < StickyD->MaxNumContacts)
00160         {
00161             // Does this sticky constraint already exist for this pair?
00162             for( Whole X = 0 ; X < StickyD->StickyConstraints.size() ; ++X )
00163             {
00164                 SliderConstraint* StickyCon = StickyD->StickyConstraints.at(X);
00165                 if( Col->PairsMatch(StickyCon->GetActorA(),StickyCon->GetActorB()) )
00166                     return;
00167             }
00168             // Does this collision have a contact that is actually penetrating?
00169             // Does this collision have actual force?
00170             bool NegativeDistFound = false;
00171             Whole FoundIndex = 0;
00172             for( Whole X = 0 ; X < Col->GetNumContactPoints() ; ++X )
00173             {
00174                 if( Col->GetDistance(X) <= FoundIndex && 0 != Col->GetAppliedImpulse(X) )
00175                 {
00176                     FoundIndex = X;
00177                     NegativeDistFound = true;
00178                     break;
00179                 }
00180             }
00181             if(!NegativeDistFound)
00182                 return;
00183             // Get the actor and it's sticky data.
00184             ActorRigid* ActorA = dynamic_cast<ActorRigid*>(UseA ? Col->GetObjectA() : Col->GetObjectB());
00185             StickyData* ActorAStickyData = ActorA->GetPhysicsSettings()->GetStickyData();
00186             // Calculate the transforms.
00187             Vector3 ALoc = UseA ? Col->GetLocalALocation(FoundIndex) : Col->GetLocalBLocation(FoundIndex);
00188             Vector3 BLoc = UseA ? Col->GetLocalBLocation(FoundIndex) : Col->GetLocalALocation(FoundIndex);
00189             Transform TransA(ALoc,ActorA->GetOrientation());
00190             Transform TransB(BLoc,this->GetOrientation());
00191             // Log
00192             std::stringstream logstream;
00193             logstream << "Sticky constraint being constructed with parameters:" << endl;
00194             logstream << "ActorA offset: " << TransA.Location << endl;
00195             logstream << "ActorB offset: " << TransB.Location << endl;
00196             logstream << "ActorA rotation: " << TransA.Rotation << endl;
00197             logstream << "ActorB rotation: " << TransB.Rotation << endl;
00198             World::GetWorldPointer()->Log(logstream.str());
00199             World::GetWorldPointer()->DoMainLoopLogging();
00200             // Create and configure the constraint.
00201             StickyConstraint* NewSticky = new StickyConstraint(ActorA,this,TransA,TransB);
00202             NewSticky->SetUpperLinLimit(0.0);
00203             NewSticky->SetUpperAngLimit(0.0);
00204             NewSticky->SetLowerLinLimit(0.0);
00205             NewSticky->SetLowerAngLimit(0.0);
00206             // Add the constraint to the world and other necessary structures.
00207             PhysicsManager::GetSingletonPtr()->GetPhysicsWorldPointer()->addConstraint(NewSticky->GetConstraintBase(),true);
00208             StickyD->StickyConstraints.push_back(NewSticky);
00209             ActorAStickyData->StickyConstraints.push_back(NewSticky);
00210         }
00211     }
