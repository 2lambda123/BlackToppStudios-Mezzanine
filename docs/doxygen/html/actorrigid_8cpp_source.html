00176     void ActorRigid::_NotifyCollisionState(Collision* Col, const Collision::CollisionState& State)
00177     {
00178         WorldObject::_NotifyCollisionState(Col,State);
00179         StickyData* StickyD = GetPhysicsSettings()->GetStickyData();
00180         bool UseA = Col->GetObjectA() != this;
00181         // We don't care if sticky behavior isn't set or if the collision has ended.
00182         // If it's ended, then we've probably already done our logic.
00183         if(0 == StickyD->MaxNumContacts || Collision::Col_End == State)
00184             return;
00185         // Constraints only work well with other rigid bodies, so for now just other actorrigids.
00186         /// @todo Update this to be workable with other objects that have rigid bodies internally, and maybe soft bodies.
00187         if( (UseA ? (Col->GetObjectA()->GetType() != Mezzanine::WSO_ActorRigid) : (Col->GetObjectB()->GetType() != Mezzanine::WSO_ActorRigid)) )
00188             return;
00189         // We need a contact point to be present for this to work, since a collision without contact points is an AABB overlap.
00190         // So confirm there are contact points.
00191         if(Col->GetNumContactPoints() < 1)
00192             return;
00193         // Do we have room for additional sticky constraints?
00194         if(StickyD->StickyConstraints.size() < StickyD->MaxNumContacts)
00195         {
00196             // Does this sticky constraint already exist for this pair?
00197             for( Whole X = 0 ; X < StickyD->StickyConstraints.size() ; ++X )
00198             {
00199                 StickyConstraint* StickyCon = StickyD->StickyConstraints.at(X);
00200                 if( Col->PairsMatch(StickyCon->GetActorA(),StickyCon->GetActorB()) )
00201                     return;
00202             }
00203             // Alternatively, is this sticky constraint about to be constructed?
00204             for( Whole Y = 0 ; Y < StickyD->CreationQueue.size() ; ++Y )
00205             {
00206                 StickyConstraintConstructionInfo& CurrInfo = StickyD->CreationQueue.at(Y);
00207                 if( Col->PairsMatch(CurrInfo.ActA,CurrInfo.ActB) )
00208                     return;
00209             }
00210             // Does this collision have a contact that is actually penetrating?
00211             // Does this collision have actual force?
00212             bool NegativeDistFound = false;
00213             Whole FoundIndex = 0;
00214             Real BestMatch = 123000;
00215             for( Whole X = 0 ; X < Col->GetNumContactPoints() ; ++X )
00216             {
00217                 Real Dist = Col->GetDistance(X);
00218                 if( MathTool::Fabs(Dist) < BestMatch && Dist <= 0 &&
00219                     0 != Col->GetAppliedImpulse(X) )
00220                 {
00221                     FoundIndex = X;
00222                     BestMatch = MathTool::Fabs(Dist);
00223                     NegativeDistFound = true;
00224                     break;
00225                 }
00226             }
00227             if(!NegativeDistFound)
00228                 return;
00229             // Ok, it's passed all the tests, the constraint is being made.
00230             // Get the actor and it's sticky data.
00231             StickyConstraintConstructionInfo NewInfo;
00232             NewInfo.ActA = dynamic_cast<ActorRigid*>(UseA ? Col->GetObjectA() : Col->GetObjectB());
00233             NewInfo.ActB = this;
00234             //StickyData* ActorAStickyData = ActorA->GetPhysicsSettings()->GetStickyData();
00235             // Calculate the transforms.
00236             Vector3 ALoc = UseA ? Col->GetLocalALocation(FoundIndex) : Col->GetLocalBLocation(FoundIndex);
00237             Vector3 BLoc = UseA ? Col->GetLocalBLocation(FoundIndex) : Col->GetLocalALocation(FoundIndex);
00238             NewInfo.TransA = Transform(ALoc,NewInfo.ActA->GetOrientation());
00239             NewInfo.TransB = Transform(BLoc,this->GetOrientation());
00240             StickyD->CreationQueue.push_back(NewInfo);
00241 
00242             /*ObjectPair CollisionPair(ActorA,this);
00243             PhysicsManager::GetSingletonPtr()->RemoveCollision(PhysicsManager::GetSingletonPtr()->GetCollision(&CollisionPair));
00244 
00245             // Create and configure the constraint.
00246             StickyConstraint* NewSticky = new StickyConstraint(ActorA,this,TransA,TransB);
00247             NewSticky->SetUpperLinLimit(0.0);
00248             NewSticky->SetUpperAngLimit(0.0);
00249             NewSticky->SetLowerLinLimit(0.0);
00250             NewSticky->SetLowerAngLimit(0.0);
