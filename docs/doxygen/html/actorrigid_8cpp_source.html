00140     void ActorRigid::_NotifyCollisionState(Collision* Col, const Collision::CollisionState& State)
00141     {
00142         ActorBase::_NotifyCollisionState(Col,State);
00143         StickyData* StickyD = GetPhysicsSettings()->GetStickyData();
00144         // We don't care if sticky behavior isn't set or if the collision has ended.
00145         // If it's ended, then we've probably already done our logic.
00146         if(NULL == StickyD || Collision::Col_End == State)
00147             return;
00148         // We need a contact point to be present for this to work, since a collision without contact points is an AABB overlap.
00149         // So confirm there are contact points.
00150         if(Col->GetNumContactPoints() == 0)
00151             return;
00152 
00153         if(StickyD->StickyConstraints.size() < StickyD->MaxNumContacts)
00154         {
00155             for( Whole X = 0 ; X < StickyD->StickyConstraints.size() ; ++X )
00156             {
00157                 Generic6DofConstraint* StickyCon = StickyD->StickyConstraints.at(X);
00158                 if( Col->PairsMatch(StickyCon->GetActorA(),StickyCon->GetActorB()) )
00159                     return;
00160             }
00161             bool UseA = Col->GetActorA() != this;
00162             ActorRigid* ActorA = dynamic_cast<ActorRigid*>(UseA ? Col->GetActorA() : Col->GetActorB());
00163             Transform TransA(UseA ? Col->GetLocalALocation(1) : Col->GetLocalBLocation(1));
00164             Transform TransB(UseA ? Col->GetLocalBLocation(1) : Col->GetLocalALocation(1));
00165             Generic6DofConstraint* NewSticky = new Generic6DofConstraint(ActorA,this,TransA,TransB);
00166             NewSticky->SetAngularLimitLower(Vector3());
00167             NewSticky->SetAngularLimitUpper(Vector3());
00168             NewSticky->SetLinearLimitLower(Vector3());
00169             NewSticky->SetLinearLimitUpper(Vector3());
00170             PhysicsManager::GetSingletonPtr()->GetPhysicsWorldPointer()->addConstraint(NewSticky->GetConstraintBase(),true);
00171             StickyD->StickyConstraints.push_back(NewSticky);
00172         }
00173     }
