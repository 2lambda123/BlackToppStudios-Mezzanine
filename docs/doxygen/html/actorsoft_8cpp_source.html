00106     void ActorSoft::CreateManualMesh (internal::MeshInfo &TheMesh)
00107     {
00108         Ogre::ManualObject* ManualEntity = new Ogre::ManualObject(TheMesh.Name);
00109         ManualEntity->setDynamic(true);
00110         ManualEntity->estimateVertexCount(TheMesh.VCount);
00111         ManualEntity->estimateIndexCount(TheMesh.ICount);
00112 
00113         ManualEntity->begin(TheMesh.Material, TheMesh.RendOp, TheMesh.Group);
00114         for( int x=0 ; x < TheMesh.VCount ; x++ )
00115         {
00116             ManualEntity->position(TheMesh.Verticies[x]);
00117             ManualEntity->normal(TheMesh.Normals[x]);
00118             ManualEntity->textureCoord(TheMesh.Textures[x]);
00119         }
00120         for( int y=0 ; y < TheMesh.ICount ; y++ )
00121         {
00122             ManualEntity->index(TheMesh.Indicies[y]);
00123         }
00124         ManualEntity->end();
00125         ManualEntity->convertToMesh(TheMesh.Name + "M", TheMesh.Group);
00126         delete ManualEntity;
