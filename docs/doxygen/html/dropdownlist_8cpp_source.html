98  void DropDownList::ConstructDropDownList(const UI::ScrollbarStyle& Style)
99  {
100  this->SelectionDisplay = this->ParentScreen->CreateWidget(this->Name+".Display",UnifiedRect(0,0,1,1,0,0,0,0));
101  this->SelectionDisplay->SetHorizontalSizingRules(UI::SR_Fill_Available);
102  this->SelectionDisplay->SetVerticalSizingRules(UI::SR_Unified_Dims);
103  this->ListToggle = this->ParentScreen->CreateCheckBox(this->Name+".Toggle",UnifiedRect(0,0,1,1,0,0,0,0));
104  this->ListToggle->SetHorizontalSizingRules(UI::SR_Match_Other_Axis);
105  this->ListToggle->SetVerticalSizingRules(UI::SR_Unified_Dims);
106  this->SelectionList = this->ParentScreen->CreateListBox(this->Name+".List",UnifiedRect(0,1,1,5,0,1,0,0),Style);
107  this->SelectionList->Hide();
108 
109  SingleLineTextLayer* DisplayText = this->SelectionDisplay->CreateSingleLineTextLayer();
110  this->SelectionDisplay->AddLayerToGroup(DisplayText,5,"Normal");
111  this->SelectionDisplay->AddLayerToGroup(DisplayText,5,"Hovered");
112 
113  this->ListToggle->Subscribe(CheckBox::EventSelected,this);
114  this->ListToggle->Subscribe(CheckBox::EventDeselected,this);
115  this->SelectionList->GetListContainer()->Subscribe(PagedContainer::EventChildFocusGained,this);
116  }

118  void DropDownList::UpdateCurrentSelection(Widget* NewSelection)
119  {
120  RenderLayerGroup* NewSelectionActive = NewSelection->GetActiveGroup();
121  RenderLayerGroup* DisSelectionActive = this->SelectionDisplay->GetActiveGroup();
122  if( NewSelectionActive != NULL && DisSelectionActive != NULL ) {
123  /// @todo This currently assumes the default ZOrder assigned to text layers in list items. If that should change or be more conveniently
124  /// configurable, this should be updated.
125  RenderLayer* UncastedNewText = NewSelectionActive->GetLayerByZOrder(5);
126  RenderLayer* UncastedDisText = DisSelectionActive->GetLayerByZOrder(5);
127  if( ( UncastedNewText != NULL && UncastedNewText->IsTextLayer() ) &&
128  ( UncastedDisText != NULL && UncastedDisText->IsTextLayer() ) )
129  {
130  TextLayer* CastedNewText = static_cast<TextLayer*>( UncastedNewText );
131  TextLayer* CastedDisText = static_cast<TextLayer*>( UncastedDisText );
132  CastedDisText->SetDefaultFont( CastedNewText->GetDefaultFont() );
133  CastedDisText->SetText( CastedNewText->GetText() );
134  }
135  }

141  void DropDownList::UpdateDimensions(const Rect& OldSelfRect, const Rect& NewSelfRect)
142  {
143  // Update the personal data first
144  this->ActDims = NewSelfRect;
145 
146  /// @todo Fix the need to remove and re-add this child.
147  // Remove the list as a child because manual transform updates aren't properly checked in the linear stategies.
148  this->RemoveChild( this->SelectionList );
149 
150  // Update the children
151  this->LayoutStrat->Layout(OldSelfRect,NewSelfRect,this->ChildWidgets);
152 
153  // Updates for the other buttons are done, so now we can re-add the list
154  this->AddChild( this->SelectionList );
155 
156  // Update our width to the appropriate size
157  this->SelectionList->SetScrollbarWidth( UnifiedDim(this->ListToggle->GetActualSize().X / NewSelfRect.Size.X,0.0) );
158  // Next prepare the new rect for the selection list
159  const Rect OldListRect = this->SelectionList->GetRect();
160  Rect NewListRect;
161  NewListRect.Size = this->LayoutStrat->HandleChildSizing(OldSelfRect,NewSelfRect,this->SelectionList);
162  NewListRect.Position = this->LayoutStrat->HandleChildHorizontalPositioning(OldSelfRect,NewSelfRect,NewListRect.Size,this->SelectionList);
163 
164  // Finally update the scroller
165  this->SelectionList->UpdateDimensions(OldListRect,NewListRect);
166 
167  // We done got icky
168  this->_MarkAllLayersDirty();
169  }

280  void DropDownList::_NotifyEvent(const EventArguments& Args)
281  {
282  const WidgetEventArguments& WidArgs = static_cast<const WidgetEventArguments&>(Args);
283  Widget* EventWidget = this->ParentScreen->GetWidget(WidArgs.WidgetName);
284 
285  if( EventWidget == this->ListToggle ) {
286  if( WidArgs.EventName == CheckBox::EventSelected ) {
287  this->SelectionList->Show();
288  }else if( WidArgs.EventName == CheckBox::EventDeselected ) {
289  this->SelectionList->Hide();
290  }
291  }
292 
293  if( EventWidget == this->SelectionList->GetListContainer() ) {
294  if( WidArgs.EventName == PagedContainer::EventChildFocusGained ) {
295  this->UpdateCurrentSelection( this->SelectionList->GetListContainer()->GetLastFocusedWidget() );
296  this->ListToggle->ManualSelect(false);
297  }
298  }
299  }
