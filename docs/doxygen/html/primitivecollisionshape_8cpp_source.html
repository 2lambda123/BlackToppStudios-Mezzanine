70  void PrimitiveCollisionShape::ProtoSerialize(XML::Node& CurrentRoot) const
71  {
72  XML::Node CollisionNode = CurrentRoot.AppendChild(this->PrimitiveCollisionShape::GetSerializableName());
73  if (!CollisionNode) { SerializeError("create CollisionNode",this->PrimitiveCollisionShape::GetSerializableName());}
74 
75  XML::Attribute Version = CollisionNode.AppendAttribute("Version");
76  if (!Version)
77  { SerializeError("Create Version Attribute", GetSerializableName()); }
78  Version.SetValue(1);
79 
80  XML::Node ImplicitShape = CollisionNode.AppendChild("ImplicitShape");
81  if (!ImplicitShape)
82  { SerializeError("Create ImplicitShape Node", GetSerializableName()); }
83  Vector3(GetBulletConvexShape()->getImplicitShapeDimensions()).ProtoSerialize(ImplicitShape);
84 
85  this->CollisionShape::ProtoSerialize(CollisionNode);
86 
87  }

89  void PrimitiveCollisionShape::ProtoDeSerialize(const XML::Node& OneNode)
90  {
91  if ( Mezzanine::String(OneNode.Name())==this->PrimitiveCollisionShape::GetSerializableName() )
92  {
93  if(OneNode.GetAttribute("Version").AsInt() == 1)
94  {
95  XML::Node CollisionNode = OneNode.GetChild("CollisionShape");
96  if(!CollisionNode)
97  { DeSerializeError("locate CollisionShape node",GetSerializableName()); }
98  this->CollisionShape::ProtoDeSerialize(CollisionNode);
99 
100  XML::Node ImplicitShape = OneNode.GetChild("ImplicitShape").GetFirstChild();
101  if(!ImplicitShape)
102  { DeSerializeError("locate ImplicitShape node",GetSerializableName()); }
103  GetBulletConvexShape()->setImplicitShapeDimensions(Vector3(ImplicitShape).GetBulletVector3());
104  }else{
105  DeSerializeError("find usable serialization version",GetSerializableName());
106  }
107  }else{
108  DeSerializeError(String("find correct class to deserialize, found a ")+OneNode.Name(),GetSerializableName());
109  }
110  }
