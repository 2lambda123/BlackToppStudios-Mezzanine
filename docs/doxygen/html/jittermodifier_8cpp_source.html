91  void JitterModifier::Modify(TextureBuffer& Buffer)
92  {
93  TextureBuffer TempBuffer(Buffer);
94  MathTools::MersenneTwisterGenerator32 NumGen(this->GeneratorSeed);
95  Integer TargetWidth = static_cast<Integer>( Buffer.GetWidth() );
96  Integer TargetHeight = static_cast<Integer>( Buffer.GetHeight() );
97  Integer Radius = static_cast<Integer>( 1.0 + ( 9.0 / 255.0 ) * ( static_cast<Real>(this->JitterRadius) - 1.0 ) );
98  //Integer MaxDist = Radius * 2 + 1;
99  for( Integer Y = 0; Y < TargetHeight ; ++Y )
100  {
101  for( Integer X = 0; X < TargetWidth ; ++X )
102  {
103  Integer NewX = X + ( NumGen.GenerateUInt() % ( Radius * 2 + 1 ) ) - Radius;
104  Integer NewY = Y + ( NumGen.GenerateUInt() % ( Radius * 2 + 1 ) ) - Radius;
105 
106  if( NewX >= 0 && NewX < TargetWidth && NewY >= 0 && NewY < TargetHeight ) {
107  TempBuffer.SetPixel(static_cast<Whole>(NewX),static_cast<Whole>(NewY), Buffer.GetPixel(static_cast<Whole>(X),static_cast<Whole>(Y)) );
108  }
109  }
110  }
111  Buffer.SetData(TempBuffer);
112  }
