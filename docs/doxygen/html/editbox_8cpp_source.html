125  Boole EditBox::HandleInputImpl(const Input::MetaCode& Code)
126  {
127  // Verify we can consume inputs.
128  if( this->IsCurrentlyEditing() ) {
129  // Get the layer and get the cursor.
130  TextLayer* EditLayer = this->GetEditLayer();
131  TextCursor* EditCursor = EditLayer->GetCursor();
132  // Get our highlight indexes.
133  const Integer HighlightStart = EditLayer->GetHighlightStart();
134  const Integer HighlightEnd = EditLayer->GetHighlightEnd();
135 
136  // Start our actual checks.
137  if( Code.GetCode() == Input::OSTEXTINPUT &&
138  ( this->InputFilter != NULL ? this->InputFilter( Code.GetMetaValue() ) : true ) )
139  {
140  // If there is a selection range we need to clear it before inserting.
141  if( HighlightStart > -1 && HighlightEnd > -1 ) {
142  // Set our post erase position.
143  EditCursor->SetCursorIndex(HighlightStart);
144  EditLayer->RemoveCharacterRange(HighlightStart,HighlightEnd);
145  }
146  // Finally perform our insert.
147  EditLayer->GetCursor()->InsertCharacterAtCursor( Code.GetMetaValue() );
148  this->_OnTextUpdated();
149  return true;
150  }else if( Code.GetCode() == Input::KEY_BACKSPACE && Code.GetMetaValue() == Input::BUTTON_PRESSING ) {
151  /// @todo Enable repeat backspace somehow.
152  if( HighlightStart > -1 && HighlightEnd > -1 ) {
153  EditCursor->SetCursorIndex(HighlightStart);
154  EditLayer->RemoveCharacterRange(HighlightStart,HighlightEnd);
155  }else{
156  EditCursor->RemoveLeftCharacter();
157  }
158  this->_OnTextUpdated();
159  return true;
160  }else if( Code.GetCode() == Input::KEY_LEFT && Code.GetMetaValue() == Input::BUTTON_PRESSING ) {
161  /// @todo Enable repeat move somehow.
162  if( HighlightStart > -1 && HighlightEnd > -1 ) {
163  EditCursor->SetCursorIndex(HighlightStart);
164  EditLayer->ClearHighlights();
165  }else{
166  EditCursor->MoveCursorLeft();
167  }
168  return true;
169  }else if( Code.GetCode() == Input::KEY_RIGHT && Code.GetMetaValue() == Input::BUTTON_PRESSING ) {
170  /// @todo Enable repeat move somehow.
171  if( HighlightStart > -1 && HighlightEnd > -1 ) {
172  EditCursor->SetCursorIndex(HighlightEnd);
173  EditLayer->ClearHighlights();
174  }else{
175  EditCursor->MoveCursorRight();
176  }
177  return true;
178  }else if( !this->IsDragged() && Code.GetCode() == Input::MOUSEBUTTON_1 && Code.GetMetaValue() == Input::BUTTON_PRESSING ) {
179  if( HighlightStart > -1 && HighlightEnd > -1 ) {
180  EditLayer->ClearHighlights();
181  }
182  // Find the offset position of the mouse and use that to update our cursor position.
183  TextLayer::CharIndexPair Result = EditLayer->GetIndexAtOffset( this->ParentScreen->GetMouseHitPosition() - this->ActDims.Position );
184  if( !Result.first )
185  return false;
186  // Perform the update.
187  EditCursor->SetCursorIndex( Result.second );
188  return true;
189  }else if( this->IsDragged() && ( Code.GetCode() == Input::MOUSEABSOLUTEHORIZONTAL || Code.GetCode() == Input::MOUSEABSOLUTEVERTICAL ) ) {
190  // Assign the horizontal or vertical value to the target, then check if we have a complete target.
191  if( Code.GetCode() == Input::MOUSEABSOLUTEHORIZONTAL ) {
192  this->EditHighlightTarget.X = Code.GetMetaValue();
193  }else if( Code.GetCode() == Input::MOUSEABSOLUTEVERTICAL ) {
194  this->EditHighlightTarget.Y = Code.GetMetaValue();
195  }
196  if( this->EditHighlightTarget.X > -1 && this->EditHighlightTarget.Y > -1 ) {
197  // When the mouse is updated, the absolute position for both axes is send out regardless of change.
198  // So we'd only get to this point if we've processed both events. Convert our target to a usable index.
199  TextLayer::CharIndexPair Result = EditLayer->GetIndexAtOffset( this->EditHighlightTarget - this->ActDims.Position );
200  if( !Result.first )
201  return false;
202  // The mouse can move to earlier characters in the squence, so sort out which index is the actual start in this set.
203  Integer IndexMin = std::min(this->EditHighlightOrigin,Result.second);
204  Integer IndexMax = std::max(this->EditHighlightOrigin,Result.second);
205  if( IndexMax == Result.second && IndexMin != Result.second ) {
206  // If we're dragging to the right, we want the mouse cursor to be on the edge of the highlights, so reduce the index to be used for highlighting.
207  --IndexMax;
208  }
209  // If there was a change, update it.
210  if( IndexMin != HighlightStart || IndexMax != HighlightEnd ) {
211  EditLayer->Highlight(IndexMin,IndexMax);
212  }
213  // Update our cursor position
214  EditCursor->SetCursorIndex(Result.second);
215  }
216  return true;
217  }
218  }
219  return false;
220  }

222  void EditBox::ConstructEditBox(const RenderLayerType EditLayerType, FontData* EditFont)
223  {
224  // Create our events.
225  this->AddEvent(EditBox::EventTextUpdated);
226 
227  // Create our edit layer.
228  TextLayer* EditLayer = NULL;
229  if( EditLayerType == UI::RLT_SingleLineText ) {
230  EditLayer = this->CreateSingleLineTextLayer();
231  }else if( EditLayerType == UI::RLT_MultiLineText ) {
232  EditLayer = this->CreateMultiLineTextLayer();
233  }else{
234  MEZZ_EXCEPTION(ExceptionBase::PARAMETERS_EXCEPTION,"Invalid RenderLayer type passed in to EditBox constructor.");
235  }
236 
237  // Set the font separately since setting the font on construction can only be done via string. todo?
238  EditLayer->SetDefaultFont(EditFont);
239 
240  // Add the edit layer to the default groups.
241  this->AddLayerToGroup(EditLayer,5,Widget::WG_Normal);
242  this->AddLayerToGroup(EditLayer,5,Widget::WG_Hovered);
243  }
