00072     std::vector<String> StringTool::Split(const String& Source, const String& Delims, const Whole& MaxSplits)
00073     {
00074         std::vector<String> Ret;
00075         Ret.reserve( MaxSplits ? MaxSplits+1 : 10 );
00076         Whole Splits = 0;
00077 
00078         size_t Start = 0;
00079         size_t Pos = 0;
00080 
00081         do
00082         {
00083             Pos = Source.find_first_of(Delims,Start);
00084             if(Pos == Start)
00085             {
00086                 Start = Pos + 1;
00087             }
00088             else if(Pos == String::npos || (MaxSplits && Splits == MaxSplits))
00089             {
00090                 Ret.push_back(Source.substr(Start));
00091                 break;
00092             }
00093             else
00094             {
00095                 Ret.push_back(Source.substr(Start,Pos - Start));
00096                 Start = Pos + 1;
00097             }
00098             Start = Source.find_first_not_of(Delims,Start);
00099             ++Splits;
00100         }while(Pos != String::npos);
00101 
00102         return Ret;
00103     }
