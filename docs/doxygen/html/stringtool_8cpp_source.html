55  const Mezzanine::Real HexConversionMultiplier = Mezzanine::Real(1.0 / 255.0);

57  /// @internal
58  /// @brief Converts a string containing hex to a ColourValue channel.
59  /// @param Hex The Hex value to be converted.
60  /// @return Returns a Real representing the converted Hex string that can be applied to a ColourValue channel.
61  Mezzanine::Real ConvertHexToColourChannel(const Mezzanine::String& Hex)
62  {
63  if( Hex.size() != 2 ) {
64  MEZZ_EXCEPTION(Mezzanine::ExceptionBase::PARAMETERS_EXCEPTION,"Hex code requires 2 characters to express a ColourValue channel.");
65  }
66 
67  Mezzanine::Real Ret = 0;
68  Mezzanine::StringStream Converter;
69  Converter << std::hex << Hex;
70  Converter >> Ret;
71  return std::min(Ret *= HexConversionMultiplier,Mezzanine::Real(1.0));
72  }
73  /// @internal
74  /// @brief Converts a ColourValue channel to Hex.
75  /// @param Channel The value to be converted to Hex.
76  /// @return Returns a two character string containing the hex expression for the provided channel value.
77  Mezzanine::String ConvertColourChannelToHex(const Mezzanine::Real Channel)
78  {
79  Mezzanine::String Ret;
80  Mezzanine::StringStream Converter;
81  Converter << std::hex << static_cast<Mezzanine::UInt8>( Channel * 255.0 );
82  Converter >> Ret;
83 
84  if( Ret.size() == 1 ) {
85  Ret.insert(0,1,'0');
86  }
87  return Ret;
88  }
89 }

137  StringVector Split(const String& Source, const String& Delims, const Whole MaxSplits)
138  {
139  StringVector Ret;
140  Ret.reserve( MaxSplits ? MaxSplits + 1 : 10 );
141  Whole Splits = 0;
142 
143  size_t Start = 0;
144  size_t Pos = 0;
145 
146  do{
147  Pos = Source.find_first_of(Delims,Start);
148  if( Pos == Start ) {
149  Start = Pos + 1;
150  }else if( Pos == String::npos || ( MaxSplits && Splits == MaxSplits ) ) {
151  Ret.push_back(Source.substr(Start));
152  break;
153  }else{
154  Ret.push_back(Source.substr(Start,Pos - Start));
155  Start = Pos + 1;
156  }
157  Start = Source.find_first_not_of(Delims,Start);
158  ++Splits;
159  }while(Pos != String::npos);
160 
161  return Ret;
162  }

186  void ToCamelCase(String& Source)
187  {
188  Boole PrevCharIsLetter = false;
189  for( String::iterator CurrIt = Source.begin() ; CurrIt != Source.end() ; ++CurrIt )
190  {
191  Boole CurrCharIsLowerLetter = StringTools::IsLowerAlphaLetter( *CurrIt );
192  Boole CurrCharIsUpperLetter = StringTools::IsUpperAlphaLetter( *CurrIt );
193  if( !PrevCharIsLetter && CurrCharIsLowerLetter ) {
194  (*CurrIt) -= 32;
195  }else if( PrevCharIsLetter && CurrCharIsUpperLetter ) {
196  (*CurrIt) += 32;
197  }
198  PrevCharIsLetter = CurrCharIsLowerLetter || CurrCharIsUpperLetter;
199  ++CurrIt;
200  }
201  }

209  Boole StartsWith(const String& Str, const String& Pattern, const Boole CaseSensitive)
210  {
211  size_t StrLen = Str.length();
212  size_t PatternLen = Pattern.length();
213  String Start = Str.substr(0,PatternLen);
214 
215  if( CaseSensitive ) {
216  return ( Start == Pattern );
217  }
218 
219  if( PatternLen > StrLen || PatternLen == 0 ) {
220  return false;
221  }
222 
223  String LowerPattern = Pattern;
224  StringTools::ToLowerCase(Start);
225  StringTools::ToLowerCase(LowerPattern);
226  return (Start == LowerPattern);
227  }

230  {
231  size_t StrLen = Str.length();

233  String End = Str.substr(StrLen - PatternLen,PatternLen);
234

236  return ( End == Pattern );
237  }
238 
239  if( PatternLen > StrLen || PatternLen == 0 ) {
240  return false;
241  }
242 
243  String LowerPattern = Pattern;
244  StringTools::ToLowerCase(End);
245  StringTools::ToLowerCase(LowerPattern);
246  return ( End == LowerPattern );

249  void RemoveDuplicateWhitespaces(String& Source)
250  {
251  for( size_t CurrIndex = Source.find_first_of(" ") ; CurrIndex != String::npos ; CurrIndex = Source.find_first_of(" ",CurrIndex) )
252  {
253  size_t EndIndex = CurrIndex;

331  ColourValue ConvertHexToColourValue(const String& ToConvert)
332  {
333  if( ToConvert.size() < 6 ) {
334  MEZZ_EXCEPTION(ExceptionBase::PARAMETERS_EXCEPTION,"Hex code requires a minimum of 6 characters to express a ColourValue instance.");
335  }
336 
337  ColourValue Ret;
338  Ret.RedChannel = ConvertHexToColourChannel( ToConvert.substr(0,2) );
339  Ret.GreenChannel = ConvertHexToColourChannel( ToConvert.substr(2,2) );
340  Ret.BlueChannel = ConvertHexToColourChannel( ToConvert.substr(4,2) );
341 
342  if( ToConvert.size() == 6 ) Ret.AlphaChannel = 1.0;
343  else Ret.AlphaChannel = ConvertHexToColourChannel( ToConvert.substr(6,2) );
344 
345  return Ret;
346  }
