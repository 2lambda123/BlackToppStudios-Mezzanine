188  {
189  XML::Node BindingsNode = SelfRoot.AppendChild( "SubSetBindings" );

191  if( BindingsNode.AppendAttribute("Version").SetValue("1") ) {

194  XML::Node BindingNode = BindingsNode.AppendChild( "SubSetBinding" );

215  void TabSet::ProtoDeSerializeButtonBindings(const XML::Node& SelfRoot)
216  {
217  this->SubSetBindings.clear();
218 
219  XML::Attribute CurrAttrib;
220  XML::Node BindingsNode = SelfRoot.GetChild( "SubSetBindings" );
221 
222  if( !BindingsNode.Empty() ) {
223  if( BindingsNode.GetAttribute("Version").AsInt() == 1 ) {
224  for( XML::NodeIterator BindingNodeIt = BindingsNode.begin() ; BindingNodeIt != BindingsNode.end() ; ++BindingNodeIt )
225  {
226  if( (*BindingNodeIt).GetAttribute("Version").AsInt() == 1 ) {
227  UInt16 ConfigID = 0;
228  String ButtonName;
229 
230  CurrAttrib = (*BindingNodeIt).GetAttribute("ConfigID");
231  if( !CurrAttrib.Empty() )
232  ConfigID = static_cast<UInt16>( CurrAttrib.AsUint() );
233 
234  CurrAttrib = (*BindingNodeIt).GetAttribute("ButtonName");
235  if( !CurrAttrib.Empty() )
236  ButtonName = CurrAttrib.AsString();
237 
238  if( !ButtonName.empty() ) {
239  Widget* NamedButton = this->ParentScreen->GetWidget(ButtonName);
240  if( NamedButton != NULL && NamedButton->GetTypeName() == StackButton::TypeName ) {
241  this->SetButtonConfig(ConfigID,static_cast<StackButton*>(NamedButton));
242  }else{
243  StringStream ExceptionStream;
244  ExceptionStream << "Named StackButton \"" << ButtonName << "\" not found when deserializing Widget named \"" << this->GetName() << "\".";
245  MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,ExceptionStream.str());
246  }
247  }
248  }else{
249  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + String("SubSetBindings") + ": Not Version 1.");
250  }
251  }
252  }else{
253  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + String("SubSetBindings") + ": Not Version 1.");
254  }
255  }
