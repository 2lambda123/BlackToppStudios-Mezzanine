725  /// the Predicate returns true the Node it was just passed is returned. If Predicate never returns true, it is called
726  /// on every Node and a blank Node is returned. The Predicate is never called with a null value.

739  /// @brief Search for any Node descended from this Node using a function to check each Node individually.
740  /// @param pred a pointer to a function that accepts an Node, and returns bool.
741  /// @details This iterates through all children of this Node, and their children ( and so on), and calls the Predicate function
742  /// passing each Node to it. This iterates through all Nodes using a depth first algorithm. If the Predicate returns true the
743  /// Node it was just passed is returned. If Predicate never returns true, it is called on every Node and a blank Node is
744  /// returned. The Predicate is never called with a null value.
745  /// @return This returns the first Node that causes Predicate to return true.
746  template <typename Predicate> Node FindNode(Predicate pred) const
747  {
748  if (!_GetRoot) return Node();
749 
750  Node cur = GetFirstChild();
751 
752  while (cur._GetRoot && cur._GetRoot != _GetRoot)
753  {
754  if (pred(cur)) return cur;
755 
756  if (cur.GetFirstChild()) cur = cur.GetFirstChild();
757  else if (cur.GetNextSibling()) cur = cur.GetNextSibling();
758  else
759  {
760  while (!cur.GetNextSibling() && cur._GetRoot != _GetRoot) cur = cur.GetParent();
761 
762  if (cur._GetRoot != _GetRoot) cur = cur.GetNextSibling();
763  }
764  }
765 
766  return Node();
767  }
