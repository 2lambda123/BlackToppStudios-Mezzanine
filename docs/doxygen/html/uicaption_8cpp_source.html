00097                 GorillaRectangle = Parent->GetGorillaLayer()->createRectangle((Rect.Position * WinDim).GetOgreVector2(),(Rect.Size * WinDim).GetOgreVector2());
00098             }else{
00099                 RelPosition = Rect.Position / Parent->GetParent()->GetViewportDimensions();
00100                 RelSize = Rect.Size / Parent->GetParent()->GetViewportDimensions();
00101 
00102                 GorillaCaption = Parent->GetGorillaLayer()->createCaption(Glyph, Rect.Position.X, Rect.Position.Y, Text, Parent->GetParent()->GetPrimaryAtlas());
00103                 GorillaCaption->size(Rect.Size.X, Rect.Size.Y);
00104                 GorillaRectangle = Parent->GetGorillaLayer()->createRectangle(Rect.Position.GetOgreVector2(),Rect.Size.GetOgreVector2());
00105             }
00106             GorillaCaption->background(Ogre::ColourValue(0,0,0,0));
00107             GorillaCaption->align(Gorilla::TextAlign_Centre);
