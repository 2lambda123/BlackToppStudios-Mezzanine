70  void Mouse::UpdateImpl(std::vector<MetaCode>& Codes)
71  {
72  /// @todo Getting the mouse focus doesn't do what I originally thought it would. When a mouse leaves a window, the focus isn't set to NULL
73  /// as expected. If you have only one window, then it just stays pointing to that window. If you have two then it will stay on the first
74  /// window until you mouse over the second, even if there is a space between the windows. This should be updated somehow so that we can set
75  /// "HoveredWindow" to NULL when none of our windows are being hovered.
76  // First do some setup. Get the window and save our position.
77  SDL_Window* Focus = SDL_GetMouseFocus();
78  Vector2 OldPosition = this->Position;
79  // Now do the actual updating.
80  for( Whole X = 0 ; X < Codes.size() ; ++X )
81  {
82  MetaCode& CurrCode = Codes[X];
83  if( CurrCode.GetCode() >= Input::MOUSEBUTTON_FIRST && CurrCode.GetCode() <= Input::MOUSEBUTTON_LAST )
84  {
85  TransitioningIndexes.push_back( CurrCode.GetCode() - Input::MOUSEBUTTON_FIRST );
86  Buttons.at( CurrCode.GetCode() - Input::MOUSEBUTTON_FIRST ) = static_cast<Input::ButtonState>(CurrCode.GetMetaValue());
87  }
88  else if( Input::MOUSEWHEELVERTICAL == CurrCode.GetCode() )
89  {
90  VerticalWheelState = static_cast<Input::DirectionalMotionState>(CurrCode.GetMetaValue());
91  }
92  else if( Input::MOUSEWHEELHORIZONTAL == CurrCode.GetCode() )
93  {
94  HorizontalWheelState = static_cast<Input::DirectionalMotionState>(CurrCode.GetMetaValue());
95  }
96  // Only if we're on a window
97  if( Focus )
98  {
99  if( Input::MOUSEABSOLUTEVERTICAL == CurrCode.GetCode() )
100  {
101  Position.Y = (Real)(CurrCode.GetMetaValue());
102  }
103  else if( Input::MOUSEABSOLUTEHORIZONTAL == CurrCode.GetCode() )
104  {
105  Position.X = (Real)(CurrCode.GetMetaValue());
106  }
107  /*else if( Input::MOUSEVERTICAL == CurrCode.GetCode() )
108  {
109  Delta.Y = (Real)(CurrCode.GetMetaValue());
110  }
111  else if( Input::MOUSEHORIZONTAL == CurrCode.GetCode() )
112  {
113  Delta.X = (Real)(CurrCode.GetMetaValue());
114  }*/
115  }
116  }
117  if( NULL != Focus )
118  {
119  Graphics::GameWindow* Win = static_cast<Graphics::GameWindow*>(Focus->data->data);
120  for( Graphics::GameWindow::ViewportIterator ViewIt = Win->BeginViewport() ; ViewIt != Win->EndViewport() ; ++ViewIt )
121  {
122  if( (Position.X >= (Real)((*ViewIt)->GetActualLeft()) && Position.X <= (Real)((*ViewIt)->GetActualLeft() + (*ViewIt)->GetActualWidth())) &&
123  (Position.Y >= (Real)((*ViewIt)->GetActualTop()) && Position.Y <= (Real)((*ViewIt)->GetActualTop() + (*ViewIt)->GetActualHeight()) ) )
124  {
125  CurrentViewport = (*ViewIt);
126  break;
127  }
128  }
129  }else{
130  CurrentViewport = NULL;
131  Position.SetIdentity();
132  Delta.SetIdentity();
133  }
134  this->Delta = this->Position - OldPosition;
135  }
