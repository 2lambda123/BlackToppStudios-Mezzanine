00164         void ListBox::DrawList()
00165         {
00166             if(Selections.empty())
00167                 return;
00168             if(!Visible)
00169                 return;
00170             ScrollHideCheck();
00171             VisibleSelections.clear();
00172             Real ToBeRounded = VertScroll->GetScrollerValue() * (Real)Selections.size();
00173             Real Remainder = fmod(ToBeRounded,(Real)1.0);
00174             Whole FirstCaption = (Whole)(Remainder >= 0.5 ? ToBeRounded + (1.0 - Remainder) : ToBeRounded - Remainder);
00175             Vector2 SelectionPos = GetActualPosition();
00176             Real ActualInc = SelectionTemplate.Size.Y * Parent->GetParent()->GetViewportDimensions().Y;
00177 
00178             for( Whole w = 0 ; w < FirstCaption ; w++ )
00179             {
00180                 Selections[w]->SetPosition(GetPosition());
00181                 Selections[w]->Hide();
00182                 SelectionSizeCheck(Selections[w]);
00183             }
00184             Whole Displayed = FirstCaption+MaxDisplay > Selections.size() ? Selections.size() : FirstCaption+MaxDisplay;
00185             for( Whole x = FirstCaption ; x < Displayed ; x++ )
00186             {
00187                 VisibleSelections.push_back(Selections[x]);
00188                 Selections[x]->Show();
00189             }
00190             for( Whole y = FirstCaption+MaxDisplay ; y < Selections.size() ; y++ )
00191             {
00192                 Selections[y]->SetPosition(GetPosition());
00193                 Selections[y]->Hide();
00194                 SelectionSizeCheck(Selections[y]);
00195             }
00196             for( Whole z = 0 ; z < VisibleSelections.size() ; z++ )
00197             {
00198                 VisibleSelections[z]->SetActualPosition(SelectionPos);
00199                 SelectionPos.Y+=ActualInc;
00200                 SelectionSizeCheck(Selections[z]);
00201             }
00202         }
