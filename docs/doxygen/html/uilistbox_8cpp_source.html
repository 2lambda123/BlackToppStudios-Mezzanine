00137         {
00138             if(Selections.empty())
00139                 return;
00140             if(!Visible)
00141                 return;
00142             Vector2 NewSize;
00143             if(VertScroll->IsVisible())
00144             {
00145                 Real MaxWidth = RelSize.X - VertScroll->GetSize().X;
00146                 if(SelectionTemplate.Size.X > MaxWidth)
00147                 {
00148                     NewSize.X = MaxWidth;

00152                 if(SelectionTemplate.Size.X > Selections[0]->GetSize().X)
00153                 {
00154                     NewSize = SelectionTemplate.Size;

00156             }
00157             VisibleSelections.clear();
00158             Real ToBeRounded = VertScroll->GetScrollerValue() * (Real)Selections.size();
00159             Real Remainder = fmod(ToBeRounded,(Real)1.0);
00160             Whole FirstCaption = (Whole)(Remainder >= 0.5 ? ToBeRounded + (1.0 - Remainder) : ToBeRounded - Remainder);
00161             Vector2 SelectionPos = GetActualPosition();
00162             Real ActualInc = SelectionTemplate.Size.Y * Parent->GetParent()->GetViewportDimensions().Y;
00163 
00164             for( Whole w = 0 ; w < FirstCaption ; w++ )
00165             {
00166                 Selections[w]->SetPosition(GetPosition());
00167                 Selections[w]->Hide();
00168                 if(0 != NewSize.X && 0 != NewSize.Y)
00169                     Selections[w]->SetSize(NewSize);
00170             }
00171             Whole Displayed = FirstCaption+NumVisible > Selections.size() ? Selections.size() : FirstCaption+NumVisible;
00172             for( Whole x = FirstCaption ; x < Displayed ; x++ )
00173             {
00174                 VisibleSelections.push_back(Selections[x]);
00175                 Selections[x]->Show();
00176             }
00177             for( Whole y = FirstCaption+NumVisible ; y < Selections.size() ; y++ )
00178             {
00179                 Selections[y]->SetPosition(GetPosition());
00180                 Selections[y]->Hide();
00181                 if(0 != NewSize.X && 0 != NewSize.Y)
00182                     Selections[y]->SetSize(NewSize);
00183             }
00184             for( Whole z = 0 ; z < VisibleSelections.size() ; z++ )
00185             {
00186                 VisibleSelections[z]->SetActualPosition(SelectionPos);
00187                 SelectionPos.Y+=ActualInc;
