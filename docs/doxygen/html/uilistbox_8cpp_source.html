00179         void ListBox::DrawList()
00180         {
00181             if(Selections.empty())
00182                 return;
00183             ScrollHideCheck();
00184             VisibleSelections.clear();
00185             Whole FirstCaption = 0;
00186             if(MaxDisplay < Selections.size())
00187             {
00188                 Real ToBeRounded = VertScroll->GetScrollerValue() * (Real)(Selections.size() - MaxDisplay);
00189                 FirstCaption = (Whole)(ToBeRounded + 0.5);
00190             }
00191             Vector2 SelectionPos = GetActualPosition();
00192             Real ActualInc = SelectionTemplate.Size.Y;
00193 
00194             for( Whole w = 0 ; w < FirstCaption ; ++w )
00195             {
00196                 Selections[w]->SetPosition(GetPosition());
00197                 Selections[w]->Hide();
00198                 SelectionSizeCheck(Selections[w]);
00199             }
00200             Whole Displayed = FirstCaption+MaxDisplay > Selections.size() ? Selections.size() : FirstCaption+MaxDisplay;
00201             for( Whole x = FirstCaption ; x < Displayed ; ++x )
00202             {
00203                 VisibleSelections.push_back(Selections[x]);
00204                 Selections[x]->SetVisible(this->IsVisible());
00205             }
00206             for( Whole y = Displayed ; y < Selections.size() ; ++y )
00207             {
00208                 Selections[y]->SetPosition(GetPosition());
00209                 Selections[y]->Hide();
00210                 SelectionSizeCheck(Selections[y]);
00211             }
00212             for( Whole z = 0 ; z < VisibleSelections.size() ; ++z )
00213             {
00214                 VisibleSelections[z]->SetActualPosition(SelectionPos);
00215                 SelectionPos.Y+=ActualInc;
00216                 SelectionSizeCheck(VisibleSelections[z]);
00217             }
00218         }
