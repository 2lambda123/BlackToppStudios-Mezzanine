200  size_t IOStream::ReadLine(Char8* Buffer, size_t MaxCount, const String& Delim)
201  {
202  Boole TrimCR = false;
203  if( Delim.find_first_of('\n') != String::npos ) {
204  TrimCR = true;
205  }
206 
207  char Temp[TEMP_STREAM_SIZE];
208  size_t ChunkSize = std::min(MaxCount,(size_t)TEMP_STREAM_SIZE - 1);
209  size_t TotalCount = 0;
210  size_t ReadCount = 0;
211 
212  while( ChunkSize && ( ReadCount = Read(Temp,ChunkSize) ) != 0 )
213  {
214  Temp[ReadCount] = '\0';
215  size_t Pos = std::strcspn(Temp,Delim.c_str());
216 
217  if( Pos < ReadCount ) {
218  this->Advance((long)(Pos + 1 - ReadCount));
219  }
220 
221  if( Buffer ) {
222  std::memcpy(Buffer + TotalCount,Temp,Pos);
223  }
224  TotalCount += Pos;
225 
226  if( Pos < ReadCount ) {
227  if( TrimCR && TotalCount && Buffer[TotalCount - 1] == '\r' ) {
228  --TotalCount;
229  }
230  break;
231  }
232 
233  ChunkSize = std::min(MaxCount - TotalCount,(size_t)TEMP_STREAM_SIZE - 1);
234  }
235  Buffer[TotalCount] = '\0';
236  return TotalCount;
237  }
