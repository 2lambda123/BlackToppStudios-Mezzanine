90  void MetaCode::Construct(const RawEvent& RawEvent_)
91  {
92  #ifdef MEZZDEBUG
93  Entresol::GetSingletonPtr()->Log("Entering: MetaCode::Construct(RawEvent);");
94  #endif
95  switch(RawEvent_.type)
96  {
97  case SDL_KEYDOWN:
98  Construct(Input::BUTTON_PRESSING, GetInputCodeFromSDL_KEY(RawEvent_));
99  break;
100  case SDL_KEYUP:
101  Construct(Input::BUTTON_LIFTING, GetInputCodeFromSDL_KEY(RawEvent_));
102  break;
103  case SDL_MOUSEBUTTONDOWN:
104  Construct(Input::BUTTON_PRESSING, GetInputCodeFromSDL_MOUSE(RawEvent_));
105  break;
106  case SDL_MOUSEBUTTONUP:
107  Construct(Input::BUTTON_LIFTING, GetInputCodeFromSDL_MOUSE(RawEvent_));
108  break;
109  case SDL_JOYBUTTONDOWN:
110  Construct(Input::BUTTON_PRESSING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
111  break;
112  case SDL_JOYBUTTONUP:
113  Construct(Input::BUTTON_LIFTING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
114  break;
115 
116  // Fail when incorrectly constructed
117  case SDL_JOYBALLMOTION: case SDL_JOYHATMOTION:
118  case SDL_JOYAXISMOTION: case SDL_MOUSEMOTION:
119  Construct(Input::BUTTON_UP, Input::KEY_FIRST); //create a safe but gibberish default
120  throw ("RawEvent which creates Multiple Metacodes inserted into Metacode");
121  break;
122 
123  default:
124  Construct(Input::BUTTON_UP, Input::KEY_FIRST); //create a safe but gibberish default
125  throw ("Unknown User Input Inserted into Metacode");
126  break;
127  }
128  }

140  //This function assumes the RawEvent is a valid SDL Keyevent
141  Input::InputCode MetaCode::GetInputCodeFromSDL_KEY(const RawEvent &RawEvent_)
142  {
143  //This Whole thing will only work with SDL Keyboard events. If we switch out event subsystems this is one of those that must change it.
144  Input::InputCode To;
145 
146  assert( sizeof(To)==sizeof(RawEvent_.key.keysym.sym) );
147  memcpy( &To, &(RawEvent_.key.keysym.scancode), sizeof(RawEvent_.key.keysym.scancode));
148 
149  return To;
150  }

292  void MetaCode::ProtoSerialize(XML::Node& CurrentRoot) const
293  {
294  Mezzanine::XML::Node MetaNode = CurrentRoot.AppendChild(SerializableName());
295  MetaNode.SetName(SerializableName());
296 
297  Mezzanine::XML::Attribute VersionAttr = MetaNode.AppendAttribute("Version");
298  Mezzanine::XML::Attribute MetaValue_ = MetaNode.AppendAttribute("MetaValue");
299  Mezzanine::XML::Attribute Code_ = MetaNode.AppendAttribute("Code");
300 
301  if( VersionAttr && MetaValue_ && Code_ )
302  {
303  if( VersionAttr.SetValue("1") && Code_.SetValue(Code) && MetaValue_.SetValue(MetaValue) )
304  {
305  return;
306  }else{
307  SerializeError("Create XML Attribute Values", SerializableName(),true);
308  }
309  }else{
310  SerializeError("Create XML Attributes", SerializableName(),true);
311  }
312  }

315  void MetaCode::ProtoDeSerialize(const XML::Node& OneNode)
316  {
317  if ( Mezzanine::String(OneNode.Name())==Mezzanine::String(SerializableName()) )
318  {
319  if(OneNode.GetAttribute("Version").AsInt() == 1)

322  MetaValue=OneNode.GetAttribute("MetaValue").AsInt();
323  }else{
324  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + SerializableName() + ": Not Version 1.");
325  }
326  }else{
327  MEZZ_EXCEPTION(Exception::II_IDENTITY_INVALID_EXCEPTION,"Attempting to deserialize a " + SerializableName() + ", found a " + String(OneNode.Name()) + ".");
328  }
