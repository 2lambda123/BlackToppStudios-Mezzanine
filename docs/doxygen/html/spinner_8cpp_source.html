260  void Spinner::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
261  {
262  this->Widget::ProtoDeSerializeProperties(SelfRoot);
263 
264  XML::Attribute CurrAttrib;
265  XML::Node PropertiesNode = SelfRoot.GetChild( Spinner::GetSerializableName() + "Properties" );
266 
267  if( !PropertiesNode.Empty() ) {
268  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
269  CurrAttrib = PropertiesNode.GetAttribute("SpinValue");
270  if( !CurrAttrib.Empty() )
271  this->SetSpinValue( CurrAttrib.AsReal() );
272 
273  CurrAttrib = PropertiesNode.GetAttribute("MinValue");
274  if( !CurrAttrib.Empty() )
275  this->SetMinSpinValue( CurrAttrib.AsReal() );
276 
277  CurrAttrib = PropertiesNode.GetAttribute("MaxValue");
278  if( !CurrAttrib.Empty() )
279  this->SetMaxSpinValue( CurrAttrib.AsReal() );
280  }else{
281  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (Spinner::GetSerializableName() + "Properties") + ": Not Version 1.");
282  }
283  }else{
284  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,Spinner::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
285  }
286  }
