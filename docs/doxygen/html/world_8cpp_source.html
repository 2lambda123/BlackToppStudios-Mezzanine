00422     void World::MainLoop()
00423     {
00424         /// @todo create a lighting manager and put this in there
00425         //this->OgreSceneManager->setAmbientLight( Ogre::ColourValue( 1, 1, 1 ) );
00426 
00427         /*! @page mainloop1 Main Loop Structure and Flow
00428          The MainLoop is heart of most video games and simulations.
00429 
00430          @section mainloopoverview1 Main loop Overview
00431          The Main loop runs in World.MainLoop() which is called by default from @ref World.GameInit(). By default this Method also starts the render, the physics andthe input systems. It does very
00432          little on it's own. The main loop then calls the PreMainLoopItems(), DoMainLoopItems and PreMainLoopItems(), for each manager in the order of their priority from Lowest to Highest.
00433          \n Here is a listing of  default priorities for each of the managers the a world intantiates by default:
00434             -50 User Input and events
00435             -40 Actors
00436             -30 Physics
00437             -20 Camera
00438             -10 Lighting (Not yet implemented)
00439               0 Graphics
00440              10 Sound
00441              20 Resources
00442         */
00443 
00444         Ogre::Timer* LoopTimer = new Ogre::Timer();
00445         this->OneTimeMainLoopInit();
00446         //As long as all the CallBacks return true the game continues
00447         bool DoNotBreak=true;
00448         while (DoNotBreak)
00449         {
00450             for (std::list< ManagerBase* >::iterator Iter=this->ManagerList.begin(); Iter!=this->ManagerList.end(); ++Iter )
00451             {
00452                 #ifdef MEZZDEBUG
00453                 this->LogStream << "Current Manager: " << (*Iter)->GetTypeName() << " - Priority: " << (*Iter)->GetPriority();
00454                 //this->DoMainLoopLogging();
00455                 #endif
00456 
00457                 #ifdef MEZZPROFILE
00458                 LoopTimer->reset();
00459                 #endif
00460 
00461                 //Actual main loop work
00462                 if( !(*Iter)->PreMainLoopItems() )
00463                     { DoNotBreak=false; }
00464 
00465                 (*Iter)->DoMainLoopItems();
00466 
00467                 if( !(*Iter)->PostMainLoopItems() )
00468                     { DoNotBreak=false; }
00469 
00470                 #ifdef MEZZPROFILE
00471                 this->LogStream << (*Iter)->GetTypeName() << " took " << LoopTimer->getMicroseconds() << " microseconds.";
00472                 this->Log();
00473                 #endif
00474 
00475                 LogCommitFunc();
00476             }
00477             if(ManualLoopBreak)
00478                     break;
00479         }//End of main loop
00480 
00481         //Some after loop cleanup
00482         //this->DestroyRenderWindow();
00483         ManualLoopBreak = false;
00484         delete LoopTimer;
00485     }
