98  const String& SceneType, const std::vector <WorldManager*>& ManagerToBeAdded )
99  {
100  //add each manager that was passed in to the manager list
101  for(std::vector<WorldManager*>::const_iterator iter = ManagerToBeAdded.begin(); iter!= ManagerToBeAdded.end(); iter++)
102  { this->AddManager(*iter); }
103 
104  if(this->GetActorManager()==0)
105  { this->AddManager(new ActorManager()); }
106  if(this->GetAreaEffectManager()==0)

111  { this->AddManager(new Physics::PhysicsManager(PhysicsInfo)); }
