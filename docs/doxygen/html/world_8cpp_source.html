00427     void World::MainLoop()
00428     {
00429         /// @todo create a lighting manager and put this in there
00430         //this->OgreSceneManager->setAmbientLight( Ogre::ColourValue( 1, 1, 1 ) );
00431 
00432         /*! @page mainloop1 Main Loop Structure and Flow
00433          The MainLoop is heart of most video games and simulations.
00434 
00435          @section mainloopoverview1 Main loop Overview
00436          The Main loop runs in World.MainLoop() which is called by default from @ref World.GameInit(). By default this Method also starts the render, the physics andthe input systems. It does very
00437          little on it's own. The main loop then calls the PreMainLoopItems(), DoMainLoopItems and PreMainLoopItems(), for each manager in the order of their priority from Lowest to Highest.
00438          \n Here is a listing of  default priorities for each of the managers the a world intantiates by default:
00439             -50 User Input and events
00440             -40 Actors
00441             -30 Physics
00442             -20 Camera
00443             -10 Lighting (Not yet implemented)
00444               0 Graphics
00445              10 Sound
00446              20 Resources
00447         */
00448 
00449         Ogre::Timer* LoopTimer = new Ogre::Timer();
00450         this->OneTimeMainLoopInit();
00451         //As long as all the CallBacks return true the game continues
00452         bool DoNotBreak=true;
00453         while (DoNotBreak)
00454         {
00455             for (std::list< ManagerBase* >::iterator Iter=this->ManagerList.begin(); Iter!=this->ManagerList.end(); ++Iter )
00456             {
00457                 #ifdef MEZZDEBUG
00458                 this->LogStream << "Current Manager: " << (*Iter)->GetTypeName() << " - Priority: " << (*Iter)->GetPriority();
00459                 //this->DoMainLoopLogging();
00460                 #endif
00461 
00462                 #ifdef MEZZPROFILE
00463                 LoopTimer->reset();
00464                 #endif
00465 
00466                 //Actual main loop work
00467                 if( !(*Iter)->PreMainLoopItems() )
00468                     { DoNotBreak=false; }
00469 
00470                 (*Iter)->DoMainLoopItems();
00471 
00472                 if( !(*Iter)->PostMainLoopItems() )
00473                     { DoNotBreak=false; }
00474 
00475                 #ifdef MEZZPROFILE
00476                 this->LogStream << (*Iter)->GetTypeName() << " took " << LoopTimer->getMicroseconds() << " microseconds.";
00477                 this->Log();
00478                 #endif
00479 
00480                 LogCommitFunc();
00481             }
00482             if(ManualLoopBreak)
00483                     break;
00484         }//End of main loop
00485 
00486         //Some after loop cleanup
00487         //this->DestroyRenderWindow();
00488         ManualLoopBreak = false;
00489         delete LoopTimer;
00490     }
