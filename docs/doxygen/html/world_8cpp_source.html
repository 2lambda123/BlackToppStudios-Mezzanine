110  void World::Construct(const Physics::ManagerConstructionInfo& PhysicsInfo,
111  const String& SceneType, const std::vector <WorldManager*>& ManagerToBeAdded )
112  {
113  //add each manager that was passed in to the manager list
114  for(std::vector<WorldManager*>::const_iterator iter = ManagerToBeAdded.begin(); iter!= ManagerToBeAdded.end(); iter++)
115  { this->AddManager(*iter); }
116 
117  //Dummy param list so we can use the auto-added manager types if needed
118  NameValuePairList Params;
119  if( this->GetActorManager() == 0 ) {
120  this->CreateManager("DefaultActorManager",Params,true);
121  }
122  if( this->GetAreaEffectManager() == 0 ) {
123  this->CreateManager("DefaultAreaEffectManager",Params,true);
124  }
125  if( this->GetDebrisManager() == 0 ) {
126  this->CreateManager("DefaultDebrisManager",Params,true);
127  }
128  if( this->GetSceneManager() == 0 ) {
129  Params.push_back( std::make_pair( String("InternalManagerTypeName"),SceneType ) );
130  this->CreateManager("DefaultSceneManager",Params,true);
131  Params.clear();
132  }
133  if( this->GetPhysicsManager() == 0 ) {
134  Params.push_back( std::make_pair( String("GeographyUpperBounds"),StringTools::ConvertToString( PhysicsInfo.GeographyUpperBounds ) ) );
135  Params.push_back( std::make_pair( String("GeogrpahyLowerBounds"),StringTools::ConvertToString( PhysicsInfo.GeographyLowerBounds ) ) );
136  Params.push_back( std::make_pair( String("MaxProxies"),StringTools::ConvertToString( PhysicsInfo.MaxProxies ) ) );
137  Params.push_back( std::make_pair( String("Gravity"),StringTools::ConvertToString( PhysicsInfo.Gravity ) ) );
138  Params.push_back( std::make_pair( String("SoftRigidWorld"),StringTools::ConvertToString( Boole(PhysicsInfo.PhysicsFlags & Physics::ManagerConstructionInfo::PCF_SoftRigidWorld) ) ) );
139  Params.push_back( std::make_pair( String("LimitlessWorld"),StringTools::ConvertToString( Boole(PhysicsInfo.PhysicsFlags & Physics::ManagerConstructionInfo::PCF_LimitlessWorld) ) ) );
140  Params.push_back( std::make_pair( String("MultiThreaded"),StringTools::ConvertToString( Boole(PhysicsInfo.PhysicsFlags & Physics::ManagerConstructionInfo::PCF_Multithreaded) ) ) );
141  this->CreateManager("DefaultPhysicsManager",Params,true);
142  Params.clear();
143  }
144 
145  #ifdef ENABLE_OALS_AUDIO_IMPLEMENTATION
146  if(this->GetSoundScapeManager()==0) {
147  this->CreateManager("OALSSoundScapeManager",Params,true);
148  }
149  #endif //ENABLE_OALS_AUDIO_IMPLEMENTATION
150  }

163  void World::Clear()
164  {
165  Physics::PhysicsManager* PhysMan = this->GetPhysicsManager();
166  // Start with constraints and anything else that is linking the objects.
167  // Nuke the metadata while we're at it.
168  if( PhysMan != NULL ) {
169  PhysMan->DestroyAllConstraints();
170  PhysMan->DestroyAllWorldTriggers();
171  PhysMan->ClearPhysicsMetaData();
172  }
173 
174  ActorManager* ActorMan = this->GetActorManager();
175  AreaEffectManager* AreaEffectMan = this->GetAreaEffectManager();
176  DebrisManager* DebrisMan = this->GetDebrisManager();
177  // Now get the higher level world objects. They'll take out their bound proxies as they go.
178  if( ActorMan != NULL ) {
179  ActorMan->DestroyAllActors();
180  }
181  if( AreaEffectMan != NULL ) {
182  AreaEffectMan->DestroyAllAreaEffects();
183  }
184  if( DebrisMan != NULL ) {
185  DebrisMan->DestroyAllDebris();
186  }
187 
188  Audio::SoundScapeManager* SoundScapeMan = this->GetSoundScapeManager();
189  Graphics::SceneManager* SceneMan = this->GetSceneManager();
190  // Now deal with any dangling proxies.
191  // Any additional minor cleanup related to these managers can be done after proxy cleanup as well.
192  if( PhysMan != NULL ) {
193  PhysMan->DestroyAllProxies();
194  }
195  if( SceneMan != NULL ) {
196  SceneMan->DestroyAllProxies();
197  SceneMan->DisableSky();
198  }
199  if( SoundScapeMan != NULL ) {
200  SoundScapeMan->DestroyAllProxies();
201  }
202  }

232  WorldManager* World::CreateManager(const String& ManagerImplName, NameValuePairList& Params, Boole AddToWorld)
233  {
234  ManagerFactoryIterator FactoryIt = World::ManagerFactories.find(ManagerImplName);
235  if( FactoryIt == World::ManagerFactories.end() ) {
236  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,"Attempting to create manager of type \"" + ManagerImplName + "\", which has no factory registered.");
237  }
238  WorldManager* NewMan = (*FactoryIt).second->CreateManager(this,Params);
239  if(AddToWorld) {
240  this->AddManager(NewMan);
241  }
242  return NewMan;
