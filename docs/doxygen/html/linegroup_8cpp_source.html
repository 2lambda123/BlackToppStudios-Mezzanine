172  Line3D::Line3D()
173  {
174  mRenderOp.vertexData = new Ogre::VertexData();
175  this->SelfNode = Entresol::GetSingletonPtr()->GetSceneManager()->_GetGraphicsWorldPointer()->getRootSceneNode()->createChildSceneNode();
176 
177  // Initialization stuff
178  mRenderOp.indexData = 0;
179  mRenderOp.vertexData->vertexCount = 1024;
180  mRenderOp.vertexData->vertexStart = 0;
181  mRenderOp.operationType = Ogre::RenderOperation::OT_LINE_LIST;
182  mRenderOp.useIndexes = false;
183 
184  Ogre::VertexDeclaration* VDecl = mRenderOp.vertexData->vertexDeclaration;
185  Ogre::VertexBufferBinding* VBind = mRenderOp.vertexData->vertexBufferBinding;
186 
187  size_t Offset = 0;
188  // Position.
189  VDecl->addElement(0,0,Ogre::VET_FLOAT3,Ogre::VES_POSITION);
190  Offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
191 
192  // Colour
193  VDecl->addElement(0,Offset,Ogre::VET_FLOAT4,Ogre::VES_DIFFUSE);
194  Offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT4);
195 
196  this->VertexBuffer = Ogre::HardwareBufferManager::getSingleton().createVertexBuffer(
197  VDecl->getVertexSize(0),
198  mRenderOp.vertexData->vertexCount,
199  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY_DISCARDABLE,
200  false);
201 
202  VBind->setBinding(0,this->VertexBuffer);
203 
204  this->setMaterial("BaseWhiteNoLighting");
205  }

219  void Line3D::ResizeVertexBuffer(const Whole RequestedSize)
220  {
221  if(RequestedSize > this->VertexBuffer->getNumVertices() ) {
222  // Update only by powers of 2
223  Whole NewVertexBufferSize = 1;
224  while(NewVertexBufferSize < RequestedSize)
225  NewVertexBufferSize <<= 1;
226 
227  this->VertexBuffer = Ogre::HardwareBufferManager::getSingletonPtr()->createVertexBuffer(
228  this->mRenderOp.vertexData->vertexDeclaration->getVertexSize(0),
229  NewVertexBufferSize,
230  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY_DISCARDABLE,
231  false);
232 
233  this->mRenderOp.vertexData->vertexStart = 0;
234  this->mRenderOp.vertexData->vertexBufferBinding->setBinding(0,this->VertexBuffer);
235  }

273  void Line3D::DrawLines()
274  {
275  // Drawing stuff
276  Whole NumPoints = this->Points.size();
277  if( NumPoints > 0 ) {
278  this->ResizeVertexBuffer(NumPoints);
279  /*if( mRenderOp.vertexData->vertexCount < NumPoints ) {
280  Whole NewVertexBufferSize = 1;
281  while(NewVertexBufferSize < NumPoints)
282  NewVertexBufferSize <<= 1;
283 
284  this->VertexBuffer = Ogre::HardwareBufferManager::getSingleton().createVertexBuffer(
285  mRenderOp.vertexData->vertexDeclaration->getVertexSize(0),
286  NewVertexBufferSize,
287  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY_DISCARDABLE,
288  false);
289 
290  mRenderOp.vertexData->vertexStart = 0;
291  mRenderOp.vertexData->vertexBufferBinding->setBinding(0,this->VertexBuffer);
292  }//*/
293 
294  Vector3 vaabMin = this->Points[0].Position;
295  Vector3 vaabMax = this->Points[0].Position;
296 
297  Real* prPos = static_cast<Real*>(this->VertexBuffer->lock(Ogre::HardwareBuffer::HBL_DISCARD));
298  for( Whole Index = 0 ; Index < NumPoints ; ++Index )
299  {
300  const LineVertex& CurrVertex = this->Points.at(Index);
301 
302  // Position assignment
303  *prPos++ = CurrVertex.Position.X;
304  *prPos++ = CurrVertex.Position.Y;
305  *prPos++ = CurrVertex.Position.Z;
306 
307  // Colour assignment
308  *prPos++ = CurrVertex.Colour.RedChannel;
309  *prPos++ = CurrVertex.Colour.GreenChannel;
310  *prPos++ = CurrVertex.Colour.BlueChannel;
311  *prPos++ = CurrVertex.Colour.AlphaChannel;
312 
313  // Bounds checking
314  if(CurrVertex.Position.X < vaabMin.X)
315  vaabMin.X = CurrVertex.Position.X;
316  if(CurrVertex.Position.Y < vaabMin.Y)
317  vaabMin.Y = CurrVertex.Position.Y;
318  if(CurrVertex.Position.Z < vaabMin.Z)
319  vaabMin.Z = CurrVertex.Position.Z;
320 
321  if(CurrVertex.Position.X > vaabMax.X)
322  vaabMax.X = CurrVertex.Position.X;
323  if(CurrVertex.Position.Y > vaabMax.Y)
324  vaabMax.Y = CurrVertex.Position.Y;
325  if(CurrVertex.Position.Z > vaabMax.Z)
326  vaabMax.Z = CurrVertex.Position.Z;
327  }
328  this->VertexBuffer->unlock();
329  this->mRenderOp.vertexData->vertexCount = NumPoints;
330 
331  mBox.setExtents(vaabMin.GetOgreVector3(), vaabMax.GetOgreVector3());
332  }

335  Real Line3D::getSquaredViewDepth(const Ogre::Camera* Cam) const
336  {
337  Ogre::Vector3 vMin, vMax, vMid, vDist;
338  vMin = mBox.getMinimum();
339  vMax = mBox.getMaximum();
340  vMid = ((vMin - vMax) * 0.5) + vMin;
341  vDist = Cam->getDerivedPosition() - vMid;
342 
343  return vDist.squaredLength();
344  }
