00082     void MetaCode::Construct(const RawEvent &RawEvent_)
00083     {
00084         #ifdef PHYSDEBUG
00085         World::GetWorldPointer()->Log("Entering: MetaCode::Construct(RawEvent);");
00086         #endif
00087         switch(RawEvent_.type)
00088         {
00089             case SDL_KEYDOWN:
00090                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_KEY(RawEvent_));
00091                 break;
00092             case SDL_KEYUP:
00093                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_KEY(RawEvent_));
00094                 break;
00095             case SDL_MOUSEBUTTONDOWN:
00096                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_MOUSE(RawEvent_));
00097                 break;
00098             case SDL_MOUSEBUTTONUP:
00099                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_MOUSE(RawEvent_));
00100                 break;
00101             case SDL_JOYBUTTONDOWN:
00102                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
00103                 break;
00104             case SDL_JOYBUTTONUP:
00105                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
00106                 break;
00107 
00108             // Fail when incorrectly constructed
00109             case SDL_JOYBALLMOTION:     case SDL_JOYHATMOTION:
00110             case SDL_JOYAXISMOTION:     case SDL_MOUSEMOTION:
00111                 Construct(BUTTON_UP, KEY_FIRST);            //create a safe but gibberish default
00112                 throw ("RawEvent which creates Multiple Metacodes inserted into Metacode");
00113                 break;
00114 
00115             default:
00116                 Construct(BUTTON_UP, KEY_FIRST);            //create a safe but gibberish default
00117                 throw ("Unknown User Input Inserted into Metacode");
00118                 break;
00119         }
