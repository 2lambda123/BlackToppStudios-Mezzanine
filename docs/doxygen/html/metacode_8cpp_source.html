00084     void MetaCode::Construct(const RawEvent &RawEvent_)
00085     {
00086         #ifdef MEZZDEBUG
00087         World::GetWorldPointer()->Log("Entering: MetaCode::Construct(RawEvent);");
00088         #endif
00089         switch(RawEvent_.type)
00090         {
00091             case SDL_KEYDOWN:
00092                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_KEY(RawEvent_));
00093                 break;
00094             case SDL_KEYUP:
00095                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_KEY(RawEvent_));
00096                 break;
00097             case SDL_MOUSEBUTTONDOWN:
00098                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_MOUSE(RawEvent_));
00099                 break;
00100             case SDL_MOUSEBUTTONUP:
00101                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_MOUSE(RawEvent_));
00102                 break;
00103             case SDL_JOYBUTTONDOWN:
00104                 Construct(BUTTON_PRESSING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
00105                 break;
00106             case SDL_JOYBUTTONUP:
00107                 Construct(BUTTON_LIFTING, GetInputCodeFromSDL_JOYSTICK(RawEvent_));
00108                 break;
00109 
00110             // Fail when incorrectly constructed
00111             case SDL_JOYBALLMOTION:     case SDL_JOYHATMOTION:
00112             case SDL_JOYAXISMOTION:     case SDL_MOUSEMOTION:
00113                 Construct(BUTTON_UP, KEY_FIRST);            //create a safe but gibberish default
00114                 throw ("RawEvent which creates Multiple Metacodes inserted into Metacode");
00115                 break;
00116 
00117             default:
00118                 Construct(BUTTON_UP, KEY_FIRST);            //create a safe but gibberish default
00119                 throw ("Unknown User Input Inserted into Metacode");
00120                 break;
00121         }
00122     }
