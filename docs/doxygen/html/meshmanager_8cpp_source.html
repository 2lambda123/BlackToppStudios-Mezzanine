00323     Mesh* MeshManager::CreateSphereMesh(const String& MeshName, const String& MaterialName, const Real& Radius, const Real& Rings, const Real& Segments)
00324     {
00325         Ogre::MaterialPtr TheMaterial = Ogre::MaterialManager::getSingleton().getByName(MaterialName);
00326         String GroupName = TheMaterial->getGroup();
00327 
00328         Real RingAngle = (Ogre::Math::PI / Rings);
00329         Real SegAngle = (2 * Ogre::Math::PI / Segments);
00330         Whole VertIndex = 0;
00331 
00332         Ogre::ManualObject* sphere = new Ogre::ManualObject("TempMan");
00333         sphere->begin(MaterialName);
00334 
00335         for( Whole ring = 0 ; ring <= Rings ; ring++ ) {
00336         Real r0 = Radius * sinf (ring * RingAngle);
00337         Real y0 = Radius * cosf (ring * RingAngle);
00338 
00339             // Generate the group of segments for the current ring
00340             for( Whole seg = 0 ; seg <= Segments ; seg++ ) {
00341                 Real x0 = r0 * sinf(seg * SegAngle);
00342                 Real z0 = r0 * cosf(seg * SegAngle);
00343 
00344                 // Add one vertex to the strip which makes up the sphere
00345                 sphere->position( x0, y0, z0);
00346                 sphere->normal(Ogre::Vector3(x0, y0, z0).normalisedCopy());
00347                 sphere->textureCoord((Real) seg / (Real) Segments, (Real) ring / (Real) Rings);
00348 
00349                 if (ring != Rings) {
00350                     // each vertex (except the last) has six indicies pointing to it
00351                     sphere->index(VertIndex + Segments + 1);
00352                     sphere->index(VertIndex);
00353                     sphere->index(VertIndex + Segments);
00354                     sphere->index(VertIndex + Segments + 1);
00355                     sphere->index(VertIndex + 1);
00356                     sphere->index(VertIndex);
00357                     VertIndex++;
00358                 }
00359             }
00360         }
