00326     {

00328         String GroupName = TheMaterial->getGroup();
00329 
00330         Real RingAngle = (Ogre::Math::PI / Rings);
00331         Real SegAngle = (2 * Ogre::Math::PI / Segments);
00332         Whole VertIndex = 0;
00333 
00334         Ogre::ManualObject* sphere = new Ogre::ManualObject("TempMan");
00335         sphere->begin(MaterialName);
00336 
00337         for( Whole ring = 0 ; ring <= Rings ; ring++ ) {
00338         Real r0 = Radius * sinf (ring * RingAngle);
00339         Real y0 = Radius * cosf (ring * RingAngle);
00340 
00341             // Generate the group of segments for the current ring
00342             for( Whole seg = 0 ; seg <= Segments ; seg++ ) {
00343                 Real x0 = r0 * sinf(seg * SegAngle);
00344                 Real z0 = r0 * cosf(seg * SegAngle);
00345 
00346                 // Add one vertex to the strip which makes up the sphere
00347                 sphere->position( x0, y0, z0);
00348                 sphere->normal(Ogre::Vector3(x0, y0, z0).normalisedCopy());
00349                 sphere->textureCoord((Real) seg / (Real) Segments, (Real) ring / (Real) Rings);
00350 
00351                 if (ring != Rings) {
00352                     // each vertex (except the last) has six indicies pointing to it
00353                     sphere->index(VertIndex + Segments + 1);
00354                     sphere->index(VertIndex);
00355                     sphere->index(VertIndex + Segments);
00356                     sphere->index(VertIndex + Segments + 1);
00357                     sphere->index(VertIndex + 1);
