00093     void ActorBasePhysicsSettings::SetCCDParams(const Real& MotionThreshold, const Real& SweptSphereRadius)
00094     {
00095         if(0==MotionThreshold)
00096         {
00097             ActorCO->setCcdMotionThreshold(MotionThreshold);
00098             ActorCO->setCcdSweptSphereRadius(MotionThreshold);
00099         }else{
00100             ActorCO->setCcdMotionThreshold(MotionThreshold);
00101             if(0==SweptSphereRadius)
00102             {
00103                 if(ActorShape)
00104                 {
00105                     btTransform Trans;
00106                     btVector3 AABBmin, AABBmax, AABBsize;
00107                     ActorShape->GetBulletShape()->getAabb(Trans,AABBmin,AABBmax);
00108                     AABBsize = AABBmax - AABBmin;
00109                     Real ResultRadius = AABBsize.getX() < AABBsize.getY() ? (AABBsize.getX() < AABBsize.getZ() ? AABBsize.getX() : AABBsize.getZ()) : (AABBsize.getY() < AABBsize.getZ() ? AABBsize.getY() : AABBsize.getZ());
00110                     ActorCO->setCcdSweptSphereRadius(ResultRadius*0.2);
00111                 }else{
00112                     ActorCO->setCcdSweptSphereRadius(1);
00113                 }
00114             }else{
00115                 ActorCO->setCcdSweptSphereRadius(SweptSphereRadius);
00116             }
