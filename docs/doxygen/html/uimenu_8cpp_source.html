00076         {
00077             MetaCode::ButtonState State = InputQueryTool::GetMouseButtonState(1);
00078             if(HoveredSubWidget)

00080                 Button* button = HoveredSubWidget->GetHoveredButton();
00081                 if(button && State == MetaCode::BUTTON_LIFTING)
00082                 {
00083                     UI::MenuWindow* MenWin = static_cast<UI::MenuWindow*>(HoveredSubWidget);
00084                     UI::MenuWindow* ChildMenWin = MenWin->GetWindowOfAccessButton(button);
00085                     if(ChildMenWin)
00086                     {
00087                         if(!ChildMenWin->IsVisible())
00088                         {
00089                             if(MenuStack.back()->GetAutoHide())
00090                                 MenuStack.back()->Hide();
00091                             MenuStack.push_back(ChildMenWin);
00092                             ChildMenWin->Show();
00093                             return;
00094                         }else{
00095                             RollMenuBackToWindow(MenWin);
00096                         }
00097                     }
00098                     else if(button == MenWin->GetBackButton())
00099                     {
00100                         MenWin->Hide();
00101                         MenuStack.pop_back();
00102                         if(!MenuStack.back()->IsVisible())
00103                             MenuStack.back()->Show();
00104                         return;
00105                     }
00106                 }
00107             }
00108         }
