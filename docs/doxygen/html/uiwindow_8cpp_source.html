00211         void Window::BorderAreaCheck(const Vector2& ScreenLoc)
00212         {
00213             if(!BorderWidth)
00214                 return;
00215             Vector2 Size = WindowBack->GetActualSize();
00216             Vector2 Position = WindowBack->GetActualPosition();
00217             bool Left = ScreenLoc.X >= Position.X && ScreenLoc.X <= Position.X + BorderWidth;
00218             bool Right = ScreenLoc.X >= Position.X + Size.X && ScreenLoc.X <= (Position.X + Size.X) - BorderWidth;
00219             bool Top = ScreenLoc.Y >= Position.Y && ScreenLoc.Y <= Position.Y + BorderWidth;
00220             bool Bottom = ScreenLoc.Y >= Position.Y + Size.Y && ScreenLoc.Y <= (Position.Y + Size.Y) - BorderWidth;
00221             if( Top && Left ) CurrentRM = Window::RM_TopLeft;
00222             else if( Top && Right ) CurrentRM = Window::RM_TopRight;
00223             else if( Bottom && Left ) CurrentRM = Window::RM_BottomLeft;
00224             else if( Bottom && Right ) CurrentRM = Window::RM_BottomRight;
00225             else if( Left ) CurrentRM = Window::RM_Left;
00226             else if( Right ) CurrentRM = Window::RM_BottomRight;
00227             else if( Top ) CurrentRM = Window::RM_BottomRight;
00228             else if( Bottom ) CurrentRM = Window::RM_BottomRight;
00229         }

00231         Vector2 Window::CalculateOffset(const Vector2 NewSize, const Vector2 OldSize, const Vector2 EleOffset, UI::ResizeableAnchor Anchor)
00232         {
00233             Vector2 NewOffset;
00234             switch (Anchor)
00235             {
00236                 case RA_AnchorTopLeft:
00237                 {
00238                     NewOffset = EleOffset;
00239                     break;
00240                 }
00241                 case RA_AnchorTop:
00242                 {
00243                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00244                     NewOffset.Y = EleOffset.Y;
00245                     break;
00246                 }
00247                 case RA_AnchorTopRight:
00248                 {
00249                     NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
00250                     NewOffset.Y = EleOffset.Y;
00251                     break;
00252                 }
00253                 case RA_AnchorLeft:
00254                 {
00255                     NewOffset.X = EleOffset.X;
00256                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00257                     break;
00258                 }
00259                 case RA_AnchorMiddle:
00260                 {
00261                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00262                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00263                     break;
00264                 }
00265                 case RA_AnchorRight:
00266                 {
00267                     NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
00268                     NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
00269                     break;
00270                 }
00271                 case RA_AnchorBottomLeft:
00272                 {
00273                     NewOffset.X = EleOffset.X;
00274                     NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
00275                     break;
00276                 }
00277                 case RA_AnchorBottom:
00278                 {
00279                     NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
00280                     NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
00281                     break;
00282                 }
00283                 case RA_AnchorBottomRight:
00284                 {
00285                     NewOffset = EleOffset + (NewSize - OldSize);
00286                     break;
00287                 }
00288                 default:
00289                     return NewOffset;
00290             }
00291             return NewOffset;
00292         }

00316                 {
00317                     NewEleSize.X = EleSize.X;
