00068         void Window::BorderAreaCheck(const Vector2& ScreenLoc)
00069         {
00070             if(!BorderWidth)
00071                 return;
00072             Vector2 Size = WidgetBack->GetActualSize();
00073             Vector2 Position = WidgetBack->GetActualPosition();
00074             bool Left = ScreenLoc.X >= Position.X && ScreenLoc.X <= Position.X + BorderWidth;
00075             bool Right = ScreenLoc.X >= Position.X + Size.X && ScreenLoc.X <= (Position.X + Size.X) - BorderWidth;
00076             bool Top = ScreenLoc.Y >= Position.Y && ScreenLoc.Y <= Position.Y + BorderWidth;
00077             bool Bottom = ScreenLoc.Y >= Position.Y + Size.Y && ScreenLoc.Y <= (Position.Y + Size.Y) - BorderWidth;
00078             if( Top && Left ) CurrentRM = Window::RM_TopLeft;
00079             else if( Top && Right ) CurrentRM = Window::RM_TopRight;
00080             else if( Bottom && Left ) CurrentRM = Window::RM_BottomLeft;
00081             else if( Bottom && Right ) CurrentRM = Window::RM_BottomRight;
00082             else if( Left ) CurrentRM = Window::RM_Left;
00083             else if( Right ) CurrentRM = Window::RM_BottomRight;
00084             else if( Top ) CurrentRM = Window::RM_BottomRight;
00085             else if( Bottom ) CurrentRM = Window::RM_BottomRight;
00086         }
