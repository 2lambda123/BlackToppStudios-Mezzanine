59  Circle::Circle(const Vector2& Point1, const Vector2& Point2, const Vector2& Point3)
60  {
61  Vector2 c1 = ( Point1 + Point2 ) * 0.5;
62  Vector2 d1 = ( Point2 - Point1 ).Perpendicular();
63  Real a1 = d1.Y;
64  Real b1 = -d1.X;
65  Real g1 = d1.X * c1.Y - d1.Y * c1.X;
66 
67  Vector2 c3 = ( Point2 + Point3 ) * 0.5;
68  Vector2 d3 = ( Point3 - Point2 ).Perpendicular();
69  Real a2 = d3.Y;
70  Real b2 = -d3.X;
71  Real g2 = d3.X * c3.Y - d3.Y * c3.X;
72 
73  Real IntersectX = ( b2 * g1 - b1 * g2 ) / ( b1 * a2 - b2 * a1 );
74  Real IntersectY = ( a2 * g1 - a1 * g2 ) / ( a1 * b2 - a2 * b1 );
75 
76  Vector2 Intersect(IntersectX,IntersectY);
77 
78  this->Center = Intersect;
79  this->Radius = ( Intersect - Point1 ).Length();
80  }
