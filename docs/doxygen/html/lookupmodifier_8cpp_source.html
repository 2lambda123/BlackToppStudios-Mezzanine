90  void LookupModifier::Modify(TextureBuffer& Buffer)
91  {
92  if( this->LookupTexture != NULL ) {
93  Whole LookupWidth = this->LookupTexture->GetWidth();
94  Whole LookupHeight = this->LookupTexture->GetHeight();
95  Whole TargetWidth = Buffer.GetWidth();
96  Whole TargetHeight = Buffer.GetHeight();
97 
98  if( LookupWidth != TargetWidth && LookupHeight != TargetHeight ) {
99  MEZZ_EXCEPTION(ExceptionBase::INVALID_STATE_EXCEPTION,"The LookupTexture size and the target texture size must be the same.");
100  }
101 
102  TextureBuffer TempBuffer( Buffer );
103 
104  for( Whole Y = 0 ; Y < LookupHeight ; ++Y )
105  {
106  for( Whole X = 0 ; X < LookupWidth ; ++X )
107  {
108  ColourValue PixelColour = this->LookupTexture->GetPixel(X,Y);
109  Whole U = static_cast<Whole>( PixelColour.RedChannel * static_cast<Real>( TargetWidth ) );
110  Whole V = static_cast<Whole>( PixelColour.GreenChannel * static_cast<Real>( TargetHeight ) );
111  U = MathTools::Clamp( U, Whole(0), TargetWidth - 1 );
112  V = MathTools::Clamp( V, Whole(0), TargetHeight - 1 );
113  TempBuffer.SetPixel( X, Y, Buffer.GetPixel(V,U) );
114  }
115  }
116 
117  Buffer.SetData(TempBuffer);
118  }
119  }
