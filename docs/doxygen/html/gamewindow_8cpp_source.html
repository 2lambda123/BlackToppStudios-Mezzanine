00277         {

00314     void GameWindow::setRenderResolution(const Whole &Width, const Whole &Height)
00315     {
00316         if(Settings.RenderWidth == Width && Settings.RenderHeight == Height)
00317             return;
00318         if(Settings.Fullscreen)
00319         {
00320             SDL_DisplayMode CurrentDisplay;
00321             SDL_GetWindowDisplayMode(SDLWindow,&CurrentDisplay);
00322             CurrentDisplay.w = Width;
00323             CurrentDisplay.h = Height;
00324             //CurrentDisplay.refresh_rate = 60;
00325             if(SDL_SetWindowDisplayMode(SDLWindow,&CurrentDisplay) == 0)
00326             {
00327                 OgreWindow->setFullscreen(true,Width,Height);
00328                 for(Whole X=0;X<Viewports.size();X++)
00329                     CorrectViewportAndCamera(X);
00330                 Settings.RenderWidth = Width;
00331                 Settings.RenderHeight = Height;
00332                 return;
00333             }
00334         }else{
00335             int Result = IsLargerThenDesktop(Width,Height);
00336             if(Result == 0)
00337             {
00338                 Whole ResultWidth, ResultHeight;

00340                 SDL_SetWindowSize(SDLWindow,ResultWidth,ResultHeight);
00341                 OgreWindow->setFullscreen(false,ResultWidth,ResultHeight);
00342             }else if(Result == 1){
00343                 World::GetWorldPointer()->Log("Cannot create a window larger then the desktop resolution.");
00344                 return;
