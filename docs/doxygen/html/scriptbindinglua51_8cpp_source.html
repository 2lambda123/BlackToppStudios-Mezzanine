00616                      const char *name) {
00617   /* STEP 1: Search the name field using binary search */
00618   swig_type_info *ret = SWIG_MangledTypeQueryModule(start, end, name);
00619   if (ret) {

00621   } else {
00622     /* STEP 2: If the type hasn't been found, do a complete search
00623        of the str field (the human readable name) */

00625     do {
00626       register size_t i = 0;
00627       for (; i < iter->size; ++i) {
00628         if (iter->types[i]->str && (SWIG_TypeEquiv(iter->types[i]->str, name)))
00629           return iter->types[i];
00630       }
00631       iter = iter->next;
00632     } while (iter != end);
00633   }
00634   
00635   /* neither found a match */

00641 */
00642 SWIGRUNTIME char *
00643 SWIG_PackData(char *c, void *ptr, size_t sz) {
00644   static const char hex[17] = "0123456789abcdef";
00645   register const unsigned char *u = (unsigned char *) ptr;
00646   register const unsigned char *eu =  u + sz;
00647   for (; u != eu; ++u) {
00648     register unsigned char uu = *u;
00649     *(c++) = hex[(uu & 0xf0) >> 4];
00650     *(c++) = hex[uu & 0xf];
00651   }
00652   return c;
00653 }

00655 /* 
00656    Unpack binary data from a string
00657 */
00658 SWIGRUNTIME const char *
00659 SWIG_UnpackData(const char *c, void *ptr, size_t sz) {
00660   register unsigned char *u = (unsigned char *) ptr;
00661   register const unsigned char *eu = u + sz;
00662   for (; u != eu; ++u) {
00663     register char d = *(c++);
00664     register unsigned char uu;
00665     if ((d >= '0') && (d <= '9'))
00666       uu = ((d - '0') << 4);
00667     else if ((d >= 'a') && (d <= 'f'))
00668       uu = ((d - ('a'-10)) << 4);
00669     else 
00670       return (char *) 0;
00671     d = *(c++);
00672     if ((d >= '0') && (d <= '9'))
00673       uu |= (d - '0');
00674     else if ((d >= 'a') && (d <= 'f'))
00675       uu |= (d - ('a'-10));
00676     else 
00677       return (char *) 0;
00678     *u = uu;
00679   }
00680   return c;
00681 }

04734 /* -----------------------------------------------------------------------------
04735  * Type initialization:
04736  * This problem is tough by the requirement that no dynamic 
04737  * memory is used. Also, since swig_type_info structures store pointers to 
04738  * swig_cast_info structures and swig_cast_info structures store pointers back
04739  * to swig_type_info structures, we need some lookup code at initialization. 
04740  * The idea is that swig generates all the structures that are needed. 
04741  * The runtime then collects these partially filled structures. 
04742  * The SWIG_InitializeModule function takes these initial arrays out of 
04743  * swig_module, and does all the lookup, filling in the swig_module.types
04744  * array with the correct data and linking the correct swig_cast_info
04745  * structures together.
04746  *
04747  * The generated swig_type_info structures are assigned staticly to an initial 
04748  * array. We just loop through that array, and handle each type individually.
04749  * First we lookup if this type has been already loaded, and if so, use the
04750  * loaded structure instead of the generated one. Then we have to fill in the
04751  * cast linked list. The cast data is initially stored in something like a
04752  * two-dimensional array. Each row corresponds to a type (there are the same
04753  * number of rows as there are in the swig_type_initial array). Each entry in
04754  * a column is one of the swig_cast_info structures for that type.
04755  * The cast_initial array is actually an array of arrays, because each row has
04756  * a variable number of columns. So to actually build the cast linked list,
04757  * we find the array of casts associated with the type, and loop through it 
04758  * adding the casts to the list. The one last trick we need to do is making
04759  * sure the type pointer in the swig_cast_info struct is correct.
04760  *
04761  * First off, we lookup the cast->type name to see if it is already loaded. 
04762  * There are three cases to handle:
04763  *  1) If the cast->type has already been loaded AND the type we are adding
04764  *     casting info to has not been loaded (it is in this module), THEN we
04765  *     replace the cast->type pointer with the type pointer that has already
04766  *     been loaded.
04767  *  2) If BOTH types (the one we are adding casting info to, and the 
04768  *     cast->type) are loaded, THEN the cast info has already been loaded by
04769  *     the previous module so we just ignore it.
04770  *  3) Finally, if cast->type has not already been loaded, then we add that
04771  *     swig_cast_info to the linked list (because the cast->type) pointer will
04772  *     be correct.
04773  * ----------------------------------------------------------------------------- */

04791   int found, init;
04792 
04793   clientdata = clientdata;
04794 
04795   /* check to see if the circular list has been setup, if not, set it up */
04796   if (swig_module.next==0) {
04797     /* Initialize the swig_module */
04798     swig_module.type_initial = swig_type_initial;
04799     swig_module.cast_initial = swig_cast_initial;
04800     swig_module.next = &swig_module;
04801     init = 1;
04802   } else {
04803     init = 0;
04804   }
04805 
04806   /* Try and load any already created modules */
04807   module_head = SWIG_GetModule(clientdata);
04808   if (!module_head) {
04809     /* This is the first module loaded for this interpreter */
04810     /* so set the swig module into the interpreter */
04811     SWIG_SetModule(clientdata, &swig_module);
04812     module_head = &swig_module;
04813   } else {
04814     /* the interpreter has loaded a SWIG module, but has it loaded this one? */
04815     found=0;
04816     iter=module_head;
04817     do {
04818       if (iter==&swig_module) {
04819         found=1;
04820         break;
04821       }
04822       iter=iter->next;
04823     } while (iter!= module_head);
04824 
04825     /* if the is found in the list, then all is done and we may leave */
04826     if (found) return;
04827     /* otherwise we must add out module into the list */
04828     swig_module.next = module_head->next;
04829     module_head->next = &swig_module;
04830   }
04831 
04832   /* When multiple interpeters are used, a module could have already been initialized in
04833      a different interpreter, but not yet have a pointer in this interpreter.
04834      In this case, we do not want to continue adding types... everything should be
04835      set up already */
04836   if (init == 0) return;
04837 
04838   /* Now work on filling in swig_module.types */
04839 #ifdef SWIGRUNTIME_DEBUG
04840   printf("SWIG_InitializeModule: size %d\n", swig_module.size);
04841 #endif
04842   for (i = 0; i < swig_module.size; ++i) {
04843     swig_type_info *type = 0;
04844     swig_type_info *ret;
04845     swig_cast_info *cast;
04846   
04847 #ifdef SWIGRUNTIME_DEBUG
04848     printf("SWIG_InitializeModule: type %d %s\n", i, swig_module.type_initial[i]->name);
04849 #endif
04850 
04851     /* if there is another module already loaded */
04852     if (swig_module.next != &swig_module) {
04853       type = SWIG_MangledTypeQueryModule(swig_module.next, &swig_module, swig_module.type_initial[i]->name);
04854     }
04855     if (type) {
04856       /* Overwrite clientdata field */
04857 #ifdef SWIGRUNTIME_DEBUG
04858       printf("SWIG_InitializeModule: found type %s\n", type->name);
04859 #endif
04860       if (swig_module.type_initial[i]->clientdata) {
04861         type->clientdata = swig_module.type_initial[i]->clientdata;
04862 #ifdef SWIGRUNTIME_DEBUG
04863       printf("SWIG_InitializeModule: found and overwrite type %s \n", type->name);
04864 #endif
04865       }
04866     } else {
04867       type = swig_module.type_initial[i];
04868     }
04869 
04870     /* Insert casting types */
04871     cast = swig_module.cast_initial[i];
04872     while (cast->type) {
04873     
04874       /* Don't need to add information already in the list */
04875       ret = 0;
04876 #ifdef SWIGRUNTIME_DEBUG
04877       printf("SWIG_InitializeModule: look cast %s\n", cast->type->name);
04878 #endif
04879       if (swig_module.next != &swig_module) {
04880         ret = SWIG_MangledTypeQueryModule(swig_module.next, &swig_module, cast->type->name);
04881 #ifdef SWIGRUNTIME_DEBUG
04882         if (ret) printf("SWIG_InitializeModule: found cast %s\n", ret->name);
04883 #endif
04884       }
04885       if (ret) {
04886         if (type == swig_module.type_initial[i]) {
04887 #ifdef SWIGRUNTIME_DEBUG
04888           printf("SWIG_InitializeModule: skip old type %s\n", ret->name);
04889 #endif
04890           cast->type = ret;
04891           ret = 0;
04892         } else {
04893           /* Check for casting already in the list */
04894           swig_cast_info *ocast = SWIG_TypeCheck(ret->name, type);
04895 #ifdef SWIGRUNTIME_DEBUG
04896           if (ocast) printf("SWIG_InitializeModule: skip old cast %s\n", ret->name);
04897 #endif
04898           if (!ocast) ret = 0;
04899         }
04900       }
04901 
04902       if (!ret) {
04903 #ifdef SWIGRUNTIME_DEBUG
04904         printf("SWIG_InitializeModule: adding cast %s\n", cast->type->name);
04905 #endif
04906         if (type->cast) {
04907           type->cast->prev = cast;
04908           cast->next = type->cast;
04909         }
04910         type->cast = cast;
04911       }
04912       cast++;
04913     }
04914     /* Set entry in modules->types array equal to the type */
04915     swig_module.types[i] = type;
04916   }
04917   swig_module.types[i] = 0;
