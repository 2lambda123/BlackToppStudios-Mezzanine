97  void TimerManager::DestroyTimer(Timer* ToBeDestroyed)
98  {
99  if(!ToBeDestroyed)
100  return;
101  if(Timers.empty())
102  return;
103  for( std::vector<Timer*>::iterator it = Timers.begin() ; it != Timers.end() ; it++ )
104  {
105  if( ToBeDestroyed == (*it) )

107  delete ToBeDestroyed;
108  Timers.erase(it);
109  return;
110  }
