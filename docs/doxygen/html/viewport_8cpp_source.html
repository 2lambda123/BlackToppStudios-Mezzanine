157  void Viewport::ProtoSerialize(XML::Node& ParentNode) const
158  {
159  XML::Node ViewNode = ParentNode.AppendChild( Viewport::GetSerializableName() );
160 
161  if( ViewNode.AppendAttribute("Version").SetValue("1") &&
162  ViewNode.AppendAttribute("ZOrder").SetValue( this->GetZOrder() ) )
163  //ViewNode.AppendAttribute("CameraName").SetValue( this->GetCamera()->GetName() ) )
164  {
165  XML::Node PositionNode = ViewNode.AppendChild("Position");
166  this->GetPosition().ProtoSerialize(PositionNode);
167 
168  XML::Node SizeNode = ViewNode.AppendChild("Size");
169  this->GetSize().ProtoSerialize(SizeNode);
170 
171  return;
172  }else{
173  SerializeError("Create XML Attribute Values",Viewport::GetSerializableName(),true);
174  }
175  }

177  void Viewport::ProtoDeSerialize(const XML::Node& SelfRoot)
178  {
179  // We need to perform some cleanup first, because Ogre doesn't let us change the viewport ZOrder.
180  // Start with the camera
181  CameraProxy* CurrCam = this->ViewportCam;
182  this->ViewportCam = NULL;
183  // Now the viewport itself
184  if( this->OgreViewport ) {
185  this->Parent->_GetOgreWindowPointer()->removeViewport( this->GetZOrder() );
186  this->OgreViewport = NULL;
187  }
188 
189  // Now do some deserializing
190  XML::Attribute CurrAttrib;
191  if( SelfRoot.Name() == Viewport::GetSerializableName() ) {
192  if(SelfRoot.GetAttribute("Version").AsInt() == 1) {
193  Integer ViewZOrder = 0;
194  Vector2 TempPos, TempSize;
195 
196  CurrAttrib = SelfRoot.GetAttribute("ZOrder");
197  if( !CurrAttrib.Empty() )
198  ViewZOrder = CurrAttrib.AsInteger();
199 
200  /*CurrAttrib = SelfRoot.GetAttribute("CameraName");
201  if( !CurrAttrib.Empty() )
202  this->SetCamera( CurrAttrib.AsString() );//*/
203 
204  XML::Node PositionNode = SelfRoot.GetChild("Position").GetFirstChild();
205  if( !PositionNode.Empty() ) {
206  TempPos.ProtoDeSerialize(PositionNode);
207  }
208 
209  XML::Node SizeNode = SelfRoot.GetChild("Size").GetFirstChild();
210  if( !SizeNode.Empty() ) {
211  TempSize.ProtoDeSerialize(SizeNode);
212  }
213 
214  this->OgreViewport = this->Parent->_GetOgreWindowPointer()->addViewport(NULL,ViewZOrder);
215  this->SetDimensions(TempPos,TempSize);
216  this->SetCamera(CurrCam);
217  }else{
218  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + Viewport::GetSerializableName() + ": Not Version 1.");
219  }
220  }else{
221  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,Viewport::GetSerializableName() + " was not found in the provided XML node, which was expected.");
222  }
223  }
