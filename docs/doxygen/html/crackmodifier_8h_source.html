79  /// @brief A modifier that will generate random coloured lines on a texture.
80  /// @details Crack starting points are generated randomly, and can be configured to have varying or non-varying lengths.
81  /// Cracks will bend based on normals generated from the parameter image. Normals are calculated from the red and green
82  /// channels exclusively. Values on the Blue channel have no effect. @n @n
83  /// Additionally the Alpha channel in the parameter image can be used to filter which pixels are allowed to generate
84  /// lines. This is done internally by comparing the Alpha channel value to the normal length.
85  ///////////////////////////////////////
86  class MEZZ_LIB CrackModifier : public TextureModifier
87  {
88  public:
89  /// @enum LengthMode
90  /// @brief An enum used to set how the length of each generated crack is to be determined.
91  enum LengthMode
92  {
93  LM_Constant = 1, ///< The modifier will use the configured length unaltered.
94  LM_Random = 2, ///< The modifier will randomize the pixel length of the cracks between zero and double the configured length.
95  LM_NormalBased = 3 ///< The modifier will generate the pixel length of a crack between the configured value and a value larger based on the normal length set in the parameter texture.
96  };
97  /// @enum QualityMode
98  /// @brief An enum used to set the quality
99  enum QualityMode
100  {
101  QM_LowQuality = 1, ///< Use the configured colour for cracks without further modification.
102  QM_Blend = 2, ///< Combines the colours from the image being modified and the configured crack colour.
103  QM_SubPixel = 3 ///< Attempts to blend the colours of all surrounding pixels for a more smooth appearance.
104  };
105  protected:
106  /// @internal
107  /// @brief The base colour to give to the pixels forming the cracks.
108  ColourValue CrackColour;
109  /// @internal
110  /// @brief The parameter image to pull normals the cracks are to move along from. Alpha channel and colours may also be used.
111  TextureBuffer* NormalsTexture;
112  /// @internal
113  /// @brief The number of cracks to generate.
114  Whole CrackCount;
115  /// @internal
116  /// @brief The seed for the random number generator used to place the cracks.
117  Whole GeneratorSeed;
118  /// @internal
119  /// @brief The mode in which the final length of each crack is determined. See LengthMode enum for more details.
120  LengthMode CrackLengthMode;
121  /// @internal
122  /// @brief The mode in which the colours at or near the crack are to be determined. See QualityMode enum for more details.
123  QualityMode CrackQualityMode;
124  /// @internal
125  /// @brief The pixel length of the cracks to be generated.
126  UInt8 CrackLength;
127  /// @internal
128  /// @brief Determines the amount of "stutter" to the cracks normal curve (or non-curve depending on the parameter image).
129  UInt8 CrackVariation;
130  public:
131  /// @brief Blank constructor.
132  CrackModifier();
133  /// @brief Class destructor.
134  virtual ~CrackModifier();
135

147  /// @brief Sets the normals texture to use for generating cracks.
148  /// @param Normals A pointer to the Texture to use for reading normals that cracks will be generated along. Initial Value: NULL.
149  /// @return Returns a reference to this.
150  CrackModifier& SetNormalsTexture(TextureBuffer* Normals);
151  /// @brief Sets the colour of the generated cracks.
152  /// @param Colour The base colour to give to the pixels forming the cracks. Initial Value: (1.0,1.0,1.0,1.0).
153  /// @return Returns a reference to this.
154  CrackModifier& SetColour(const ColourValue& Colour);
155  /// @brief Sets the colour of the generated cracks.
156  /// @param Red The amount of red to give to the pixels forming the cracks. Initial Value: 1.0.
157  /// @param Green The amount of green to give to the pixels forming the cracks. Initial Value: 1.0.
158  /// @param Blue The amount of blue to give to the pixels forming the cracks. Initial Value: 1.0.
159  /// @param Alpha The amount of alpha to give to the pixels forming the cracks. Initial Value: 1.0.
160  /// @return Returns a reference to this.
161  CrackModifier& SetColour(const Real Red, const Real Green, const Real Blue, const Real Alpha = 1.0);
162  /// @brief Sets how many cracks to create.
163  /// @param Count The number of cracks to generate. Initial Value: 100.
164  /// @return Returns a reference to this.
165  CrackModifier& SetCrackCount(const Whole Count);
166  /// @brief Sets any additional configuration for the length of the cracks.
167  /// @param Length The mode in which the final length of each crack is determined. See LengthMode enum for more details. Initial Value: LM_Constant.
168  /// @return Returns a reference to this.
169  CrackModifier& SetCrackLengthMode(const LengthMode Length);
170  /// @brief Sets any additional configuration for the colour/sampling of the crack colour.
171  /// @param Quality The mode in which the colours at or near the crack are to be determined. See QualityMode enum for more details. Initial Value: QM_LowQuality.
172  /// @return Returns a reference to this.
173  CrackModifier& SetCrackQualityMode(const QualityMode Quality);
174  /// @brief Sets the length of each crack that is created. May not be used verbatim.
175  /// @param Length The pixel length of the cracks to be generated. Initial Value: 255.
176  /// @return Returns a reference to this.
177  CrackModifier& SetCrackLength(const UInt8 Length);
178  /// @brief Sets the amount of wobble or stutter to the cracks that are created.
179  /// @param Variation Determines the amount of "stutter" to the cracks normal curve (or non-curve depending on the parameter image). Initial Value: 64.
180  /// @return Returns a reference to this.
181  CrackModifier& SetCrackVariation(const UInt8 Variation);
182  /// @brief Sets the seed for the random number generator used to determine crack starting positions.
183  /// @param Seed The seed to be used for the randomly generatated positions to place the cracks. Initial Value: 5120.
184  /// @return Returns a reference to this.
185  CrackModifier& SetGeneratorSeed(const Whole Seed);
