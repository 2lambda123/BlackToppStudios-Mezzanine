67  Boole HTTPRequest::ParseHTTPHeader(StringIterator& CurrIt, const StringIterator EndIt)
68  {
69  // A request header has 3 components: A method, a URI, and a version.
70  Boole EoL = false;
71  String HeaderTemp;
72  // Method
73  EoL = this->GetMessageComponent(CurrIt,EndIt,HeaderTemp);
74  this->RequestMethod = this->ConvertRequestMethod(HeaderTemp);
75  if( EoL ) {
76  return false;
77  }
78  // URI
79  HeaderTemp.clear();
80  EoL = this->GetMessageComponent(CurrIt,EndIt,HeaderTemp);
81  this->RequestURI.ParseFromString(HeaderTemp);
82  if( EoL ) {
83  return false;
84  }
85  // Version
86  HeaderTemp.clear();
87  this->GetMessageComponent(CurrIt,EndIt,HeaderTemp);
88  this->ParseHTTPVersion(HeaderTemp);
89  // Handle extra trash
90  //this->AdvanceToNewline(CurrIt,EndIt);
91  return true;
92  }

144  String HTTPRequest::Compose() const
145  {
146  StringStream RequestStream;
147  // Header
148  RequestStream << HTTPRequest::ConvertRequestMethod( this->RequestMethod ) << " " << this->RequestURI.ConvertToString() << " ";
149  RequestStream << "HTTP/" << this->MessageVersion.Major << "." << this->MessageVersion.Minor << "\r\n";
150  // Fields
151  for( NameValuePairMap::const_iterator FieldIt = this->MessageFields.begin() ; FieldIt != this->MessageFields.end() ; ++FieldIt )
152  {
153  RequestStream << (*FieldIt).first << ": " << (*FieldIt).second << "\r\n";
154  }
155  RequestStream << "\r\n";
156  // Body
157  RequestStream << this->MessageBody;
158  return RequestStream.str();
159  }

161  Boole HTTPRequest::Decompose(const String& Request)
162  {
163  if( !Request.empty() ) {
164  // Set up the data
165  StringIterator CurrIt = Request.begin();
166  const StringIterator EndIt = Request.end();
167  // Start decomposing
168  if( !this->ParseHTTPHeader(CurrIt,EndIt) ) {
169  return false;
170  }
171  if( !this->ParseHTTPFields(CurrIt,EndIt) ) {
172  return false;
173  }
174  this->MessageBody.clear();
175  this->MessageBody.assign(CurrIt,EndIt);
176 
177  return true;
178  }
179  return false;
180  }
