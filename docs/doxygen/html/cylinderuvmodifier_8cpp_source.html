93  void CylinderUVModifier::Modify(TriangleBuffer& Buffer)
94  {
95  if( this->CylinderHeight <= 0 )
96  MEZZ_EXCEPTION(ExceptionBase::INVALID_STATE_EXCEPTION,"Cylinder height must be positive.");
97  if( this->CylinderRadius <= 0 )
98  MEZZ_EXCEPTION(ExceptionBase::INVALID_STATE_EXCEPTION,"Cylinder radius must be positive.");
99 
100  Real AngleThreshold = MathTools::ATan( this->CylinderHeight / this->CylinderRadius );
101  for( VertexIterator VertIt = Buffer.GetVertices().begin() ; VertIt != Buffer.GetVertices().end() ; ++VertIt )
102  {
103  Vector2 XZNorm( VertIt->Normal.X, VertIt->Normal.Z );
104  Real Alpha = ( MathTools::ATan( VertIt->Normal.Y / XZNorm.Length() ) + MathTools::GetHalfPi() );
105  if( MathTools::Abs(Alpha) > AngleThreshold ) {
106  Vector2 XZVec( VertIt->Position.X, VertIt->Position.Z );
107  VertIt->UV = XZVec / this->CylinderRadius;
108  }else{
109  Vector2 XZVec( VertIt->Position.X, VertIt->Position.Z );
110  VertIt->UV.X = Vector2::Unit_X().AngleTo(XZVec) / MathTools::GetTwoPi();
111  VertIt->UV.Y = VertIt->Position.Y / this->CylinderHeight - 0.5;
112  }
113  }
114  }
