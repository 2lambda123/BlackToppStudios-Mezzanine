124  UInt16 InputManager::DetectControllers()
125  {
126  UInt16 Count;
127  for( Count = 0 ; Count < SDL_NumJoysticks() ; ++Count )
128  {
129  SDL_Joystick* InternalControl = SDL_JoystickOpen(Count);
130  Input::Controller* NewController = new Input::Controller( InternalControl, Count );
131  this->Controllers.push_back( NewController );
132  this->InternalControllers.push_back( InternalControl );
133  }
134  return Count;
135  }

137  void InputManager::ReleaseAllControllers()
138  {
139  if( Controllers.empty() )
140  return;
141 
142  for( ControllerIterator ContIt = Controllers.begin() ; ContIt != Controllers.end() ; ++ContIt )
143  {
144  delete (*ContIt);
145  }
146  for( std::vector<void*>::iterator ContIt = InternalControllers.begin() ; ContIt != InternalControllers.end() ; ++ContIt )
147  {
148  SDL_JoystickClose( (SDL_Joystick*)(*ContIt) );
149  }
150  this->Controllers.clear();
151  this->InternalControllers.clear();
152  }

160  void InputManager::DoMainLoopItems()
161  {
162  std::vector< MetaCode > MouseCodes;
163  std::vector< MetaCode > KeyboardCodes;
164  std::vector< std::vector< MetaCode > > ControllerCodes;
165  ControllerCodes.resize( GetNumControllers() );
166 
167  std::list<EventUserInput*>* UserInput = EventManager::GetSingletonPtr()->GetAllUserInputEvents();
168  EventManager::GetSingletonPtr()->RemoveAllSpecificEvents(EventBase::UserInput);
169 
170  for(std::list<EventUserInput*>::iterator Iter = UserInput->begin(); !UserInput->empty(); Iter = UserInput->begin())
171  {
172  EventUserInput* CurrEvent = (*Iter);
173  for( Whole X = 0 ; X < CurrEvent->size() ; ++X )
174  {
175  MetaCode& CurrCode = CurrEvent->at(X);
176  if( Input::MOUSE_FIRST <= CurrCode.GetCode() && Input::MOUSE_LAST >= CurrCode.GetCode() ){
177  MouseCodes.push_back( CurrCode );
178  }else if( Input::KEY_FIRST <= CurrCode.GetCode() && Input::KEY_LAST >= CurrCode.GetCode() ){
179  KeyboardCodes.push_back( CurrCode );
180  }else if( Input::CONTROLLER_FIRST <= CurrCode.GetCode() && Input::CONTROLLER_LAST >= CurrCode.GetCode() ){
181  ControllerCodes[ CurrCode.GetDeviceIndex() ].push_back( CurrCode );
182  }
183  }
184 
185  delete CurrEvent;
186  UserInput->remove(*Iter);
187  }
188 
189  SystemMouse->_Update(MouseCodes);
190  SystemKeyboard->_Update(KeyboardCodes);
191  for( Whole X = 0 ; X < GetNumControllers() ; ++X )
192  {
193  Controllers.at(X)->_Update( ControllerCodes.at(X) );
194  }
195  }
