00125         void ButtonListBox::DrawList()
00126         {
00127             if(Selections.empty())
00128                 return;
00129             if(!Visible)
00130                 return;
00131             Vector2 NewSize;
00132             if(VertScroll->IsVisible())
00133             {
00134                 Real MaxWidth = RelSize.X - (VertScroll->GetSize().X + SelectionDist);
00135                 if(TSize.X > MaxWidth)
00136                 {
00137                     NewSize.X = MaxWidth;
00138                     NewSize.Y = TSize.Y;
00139                 }
00140             }else{
00141                 if(TSize.X > Selections[0]->GetSize().X)
00142                 {
00143                     NewSize = TSize;
00144                 }
00145             }
00146             VisibleSelections.clear();
00147             Real One = 1;
00148             Real ToBeRounded = VertScroll->GetScrollerValue() * (Real)Selections.size();
00149             Real Remainder = fmod(ToBeRounded,One);
00150             Whole FirstButton = (Whole)(Remainder >= 0.5 ? ToBeRounded + (One - Remainder) : ToBeRounded - Remainder);
00151             Vector2 SelectionPos = GetActualPosition();
00152             Real ActualDist = SelectionDist * Parent->GetParent()->GetViewportDimensions().Y;
00153             Real ActualInc = ActualDist + (TSize.Y * Parent->GetParent()->GetViewportDimensions().Y);
00154             SelectionPos.X+=ActualDist;
00155             SelectionPos.Y+=ActualDist;
00156             for( Whole w = 0 ; w < FirstButton ; w++ )
00157             {
00158                 Selections[w]->SetPosition(GetPosition());
00159                 Selections[w]->Hide();
00160                 if(0 != NewSize.X && 0 != NewSize.Y)
00161                     Selections[w]->SetSize(NewSize);
00162             }
00163             Whole Displayed = FirstButton+NumVisible > Selections.size() ? Selections.size() : FirstButton+NumVisible;
00164             for( Whole x = FirstButton ; x < Displayed ; x++ )
00165             {
00166                 VisibleSelections.push_back(Selections[x]);
00167                 Selections[x]->Show();
00168             }
00169             for( Whole y = FirstButton+NumVisible ; y < Selections.size() ; y++ )
00170             {
00171                 Selections[y]->SetPosition(GetPosition());
00172                 Selections[y]->Hide();
00173                 if(0 != NewSize.X && 0 != NewSize.Y)
00174                     Selections[y]->SetSize(NewSize);
00175             }
00176             for( Whole z = 0 ; z < VisibleSelections.size() ; z++ )
00177             {
00178                 VisibleSelections[z]->SetActualPosition(SelectionPos);
00179                 SelectionPos.Y+=ActualInc;
00180                 if(0 != NewSize.X && 0 != NewSize.Y)
00181                     VisibleSelections[z]->SetSize(NewSize);
00182             }

00221             if(MetaCode::BUTTON_LIFTING == State)
