00398     Matrix4x4 Matrix4x4::Inverse() const
00399     {

00401         Real m10 = Matrix[1][0], m11 = Matrix[1][1], m12 = Matrix[1][2], m13 = Matrix[1][3];
00402         Real m20 = Matrix[2][0], m21 = Matrix[2][1], m22 = Matrix[2][2], m23 = Matrix[2][3];
00403         Real m30 = Matrix[3][0], m31 = Matrix[3][1], m32 = Matrix[3][2], m33 = Matrix[3][3];
00404 
00405         Real v0 = m20 * m31 - m21 * m30;
00406         Real v1 = m20 * m32 - m22 * m30;
00407         Real v2 = m20 * m33 - m23 * m30;
00408         Real v3 = m21 * m32 - m22 * m31;
00409         Real v4 = m21 * m33 - m23 * m31;
00410         Real v5 = m22 * m33 - m23 * m32;
00411 
00412         Real t00 = + (v5 * m11 - v4 * m12 + v3 * m13);
00413         Real t10 = - (v5 * m10 - v2 * m12 + v1 * m13);
00414         Real t20 = + (v4 * m10 - v2 * m11 + v0 * m13);
00415         Real t30 = - (v3 * m10 - v1 * m11 + v0 * m12);
00416 
00417         Real invDet = 1 / (t00 * m00 + t10 * m01 + t20 * m02 + t30 * m03);
00418 
00419         Real d00 = t00 * invDet;
00420         Real d10 = t10 * invDet;
00421         Real d20 = t20 * invDet;
00422         Real d30 = t30 * invDet;
00423 
00424         Real d01 = - (v5 * m01 - v4 * m02 + v3 * m03) * invDet;
00425         Real d11 = + (v5 * m00 - v2 * m02 + v1 * m03) * invDet;
00426         Real d21 = - (v4 * m00 - v2 * m01 + v0 * m03) * invDet;
00427         Real d31 = + (v3 * m00 - v1 * m01 + v0 * m02) * invDet;
00428 
00429         v0 = m10 * m31 - m11 * m30;
00430         v1 = m10 * m32 - m12 * m30;
00431         v2 = m10 * m33 - m13 * m30;
00432         v3 = m11 * m32 - m12 * m31;
00433         v4 = m11 * m33 - m13 * m31;
00434         v5 = m12 * m33 - m13 * m32;
00435 
00436         Real d02 = + (v5 * m01 - v4 * m02 + v3 * m03) * invDet;
00437         Real d12 = - (v5 * m00 - v2 * m02 + v1 * m03) * invDet;
00438         Real d22 = + (v4 * m00 - v2 * m01 + v0 * m03) * invDet;
00439         Real d32 = - (v3 * m00 - v1 * m01 + v0 * m02) * invDet;
00440 
00441         v0 = m21 * m10 - m20 * m11;
00442         v1 = m22 * m10 - m20 * m12;
00443         v2 = m23 * m10 - m20 * m13;
00444         v3 = m22 * m11 - m21 * m12;
00445         v4 = m23 * m11 - m21 * m13;
00446         v5 = m23 * m12 - m22 * m13;
00447
