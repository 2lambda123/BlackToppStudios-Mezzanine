00476         {
00477             Real Diff = (Real)Height > (*it).second->mLineHeight ? (Real)Height - (*it).second->mLineHeight : (*it).second->mLineHeight - (Real)Height;
00478             if(Diff < BestMatchDiff)
00479             {
00480                 BestMatch = (*it).first;
00481                 BestHeight = (*it).second->mLineHeight;
00482                 BestMatchDiff = Diff;
00483             }
00484         }
00485
