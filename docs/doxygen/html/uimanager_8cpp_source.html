00507             {
00508                 Diff = (*it).second->LineHeight - RequestedHeight;
00509                 if(Diff < LargerMatchDiff)
00510                 {
00511                     LargerMatch = (*it).first;
00512                     LargerHeight = (*it).second->LineHeight;
00513                     LargerMatchDiff = Diff;
00514                     continue;
00515                 }
00516             }else{
00517                 Diff = RequestedHeight - (*it).second->LineHeight;

00521                     SmallerHeight = (*it).second->LineHeight;
00522                     SmallerMatchDiff = Diff;
00523                     continue;
00524                 }
00525             }
00526         }
00527 
00528         Real Scale = 1;
00529         if(LargerMatch != 0)
00530         {
00531             Scale = RequestedHeight / LargerHeight;
00532             return std::pair<Whole,Real>(LargerMatch,Scale);
