190  void UIManager::HandlePreFocusMouseInput(const Input::MetaCode& Code)
191  {
192  // Perform our simple checks for the focus switching/locking
193  if( Code.IsMouseButton() ) {
194  if( Input::BUTTON_PRESSING == Code.GetMetaValue() ) {
195  this->SwitchFocus(HoveredWidget);
196 
197  // If we got a valid focus, there is more work to be done.
198  if( this->WidgetFocus ) {
199  // Regardless of it has changed, lock the focus.
200  if( !this->FocusIsLocked() ) {
201  this->WidgetFocus->_OnFocusLocked();
202  this->FocusLockCode = Code;
203  }
204  }
205  }else if( Input::BUTTON_LIFTING == Code.GetMetaValue() ) {
206  if( this->WidgetFocus ) {
207  // Check the code to see if we're releasing the focus lock.
208  if( this->FocusLockCode.GetCode() == Code.GetCode() && this->FocusLockCode.GetDeviceIndex() == Code.GetDeviceIndex() ) {
209  this->WidgetFocus->_OnFocusUnlocked();
210  this->WidgetFocus->_OnMouseDragEnd();
211  this->FocusLockCode.SetNullValues();
212  }
213  }
214  }
215  }
216 
217  // Perform the check for dragging
218  if( this->FocusIsLocked() && this->MouseMoved ) {
219  if( !this->WidgetFocus->IsDragged() ) {
220  this->WidgetFocus->_OnMouseDragStart();
221  }
222  }
223  }

250  Boole UIManager::SwitchFocus(Widget* NewFocus)
251  {
252  // Check if the focus is changing at all
253  if( this->WidgetFocus != NewFocus && !this->FocusIsLocked() )
254  {
255  // If we already have a focus, drop it
256  if( this->WidgetFocus )
257  this->WidgetFocus->_OnFocusLost();
258 
259  // Assign the new focus
260  this->WidgetFocus = NewFocus;
261 
262  // If the new focus is valid, notify it of it's new status
263  if( this->WidgetFocus )
264  this->WidgetFocus->_OnFocusGained();
265 
266  // Acknowledge the focus was changed
267  return true;
268  }
269  // Nothing changed
270  return false;
271  }

573  UIManager::FontResult UIManager::SuggestGlyphIndex(const Whole& Height, const String& Atlas)
574  {
575  TextureAtlas* TheAtlas = AtlasHandler->GetAtlas(Atlas);
576  if(NULL == TheAtlas)
577  {
578  MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,"Attempting to access TextureAtlas \"" + Atlas + "\", which does not exist or is not loaded.");
579  }
580  TextureAtlas::FontDataContainer& Fonts = TheAtlas->GetFonts();
581  String LargerMatch, SmallerMatch;
582  Real LargerHeight = 0;
583  Real SmallerHeight = 0;
584  Real LargerMatchDiff = 1000000.0;
585  Real SmallerMatchDiff = 1000000.0;
586  Real RequestedHeight = (Real)Height;
587 
588  for( TextureAtlas::FontDataIterator it = Fonts.begin() ; it != Fonts.end() ; it++ )
589  {
590  Real Diff = 0.0;
591  if((*it).second->GetLineHeight() > RequestedHeight)
592  {
593  Diff = (*it).second->GetLineHeight() - RequestedHeight;
594  if(Diff < LargerMatchDiff)
595  {
596  LargerMatch = (*it).first;
597  LargerHeight = (*it).second->GetLineHeight();
598  LargerMatchDiff = Diff;
599  continue;
600  }
601  }else{
602  Diff = RequestedHeight - (*it).second->GetLineHeight();
603  if(Diff < SmallerMatchDiff)
604  {
605  SmallerMatch = (*it).first;
606  SmallerHeight = (*it).second->GetLineHeight();
607  SmallerMatchDiff = Diff;
608  continue;
609  }
610  }
611  }
612 
613  Real Scale = 1;
614  if(!LargerMatch.empty())
615  {
616  Scale = RequestedHeight / LargerHeight;
617  return FontResult(LargerMatch,Scale);
618  }else{
619  Scale = RequestedHeight / SmallerHeight;
620  return FontResult(SmallerMatch,Scale);
621  }
622  }

624  void UIManager::UpdateScreens()
625  {
626  Input::Mouse* SysMouse = Input::InputManager::GetSingletonPtr()->GetSystemMouse();
627 
628  // Tell all screens to check if the veiwport dimensions have changed
629  for( ScreenIterator ScIt = this->Screens.begin() ; ScIt != this->Screens.end() ; ++ScIt )
630  (*ScIt)->CheckViewportSize();
631 
632  // Inject input delta's from the input system
633  const MetaCodeContainer& Inputs = Input::InputManager::GetSingletonPtr()->GetInputDeltas();
634  // InjectedInputs.insert(InjectedInputs.end(),Inputs.begin(),Inputs.end()); // Commented because we want to use "InjectInput()" so we can filter the codes.
635  for( ConstMetaCodeIterator MCIt = Inputs.begin() ; MCIt != Inputs.end() ; ++MCIt )
636  this->InjectInput( (*MCIt) );
637 
638  // Get the hovered quad based on mouse position
639  // We don't care what is hovered so much if the mouse hasn't moved, or is hidden due to relative mode.
640  if( this->MouseMoved && !SysMouse->GetRelativeMode() ) {
641  Widget* TempWidget = this->CheckWidgetUnderPoint(SysMouse->GetHoveredViewport(),SysMouse->GetViewportPosition());
642  if( TempWidget != this->HoveredWidget )
643  {
644  if( this->HoveredWidget ) {
645  this->HoveredWidget->_OnMouseExit();
646  }
647 
648  this->HoveredWidget = TempWidget;
649 
650  if( this->HoveredWidget ) {
651  this->HoveredWidget->_OnMouseEnter();
652  }
653  }
654  }
655 
656  // Check the focus to see if it's still visible. If it isn't, attempt to pass focus to the next available parent.
657  if( this->WidgetFocus && !this->WidgetFocus->GetVisible() ) {
658  // Clear the focus lock
659  if( this->FocusIsLocked() ) {
660  this->WidgetFocus->_OnFocusUnlocked();
661  this->FocusLockCode.SetNullValues();
662  }
663 
664  // Find the next available parent
665  Widget* PassedFocus = this->WidgetFocus;
666  do{
667  PassedFocus = ( PassedFocus->GetParent()->IsWidget() ? static_cast<Widget*>(PassedFocus->GetParent()) : NULL );
668  }while( PassedFocus != NULL && !PassedFocus->GetVisible() );
669 
670  // Switch the focus, even if it is NULL
671  this->SwitchFocus(PassedFocus);
672  }
673 
674  // Look at and process all injected inputs accordingly
675  for( InputIterator InIt = this->InjectedInputs.begin() ; InIt != this->InjectedInputs.end() ; ++InIt )
676  {
677  // Allow our pre processes to get first dibs.
678  this->HandlePreFocusInput( (*InIt) );
679 
680  // Allow the focus to consume the input if it needs/wants to.
681  if( this->WidgetFocus && this->WidgetFocus->_HandleInput( (*InIt) ) )
682  continue;
683
