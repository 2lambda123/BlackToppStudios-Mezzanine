00206         void Spinner::CalculateOffsets(const Vector2& Size)
00207         {
00208             switch (SpinLayout)
00209             {
00210                 case UI::Spn_Separate:
00211                 {
00212                     if(Size.X > Size.Y * 2)
00213                     {
00214                         IncrementOffset = Vector2(Size.X - Size.Y,0);
00215                         DecrementOffset = Vector2(0,0);
00216                         ValueDisplayOffset = Vector2(Size.Y,0);
00217                     }
00218                     else if(Size.Y > Size.X * 2)
00219                     {
00220                         IncrementOffset = Vector2(0,0);
00221                         DecrementOffset = Vector2(0,Size.Y - Size.X);
00222                         ValueDisplayOffset = Vector2(0,Size.X);
00223                     }
00224                     break;
00225                 }
00226                 case UI::Spn_Together_Left:
00227                 {
00228                     IncrementOffset = Vector2(0,0);
00229                     DecrementOffset = Vector2(0,Size.Y * 0.5);
00230                     ValueDisplayOffset = Vector2(Size.Y * 0.5,0);
00231                     break;
00232                 }
00233                 case UI::Spn_Together_Right:
00234                 {
00235                     IncrementOffset = Vector2(Size.X - (Size.Y * 0.5),0);
00236                     DecrementOffset = Vector2(Size.X - (Size.Y * 0.5),Size.Y * 0.5);

00305         void Spinner::UpdateImpl(bool Force)
00306         {
00307             ProcessCapturedInputs();
00308             if(HoveredButton)
00309             {
00310                 MetaCode::ButtonState State = InputQueryTool::GetMouseButtonState(1);
00311                 if(MetaCode::BUTTON_PRESSING == State)
00312                 {
00313                     if(HoveredButton==Increment)
00314                         IncrementLock = false;
00315                     else if(HoveredButton==Decrement)
00316                         DecrementLock = false;
00317                 }
00318                 else if(MetaCode::BUTTON_LIFTING && (!IncrementLock || !DecrementLock))
00319                 {
00320                     if(HoveredButton==Increment && !IncrementLock)
00321                     {
00322                         Value++;
00323                         CheckValueLimits();
00324                         ValueDisplay->SetText(GetValueAsText());
00325                         IncrementLock = true;
00326                     }
00327                     else if(HoveredButton==Decrement && !DecrementLock)
00328                     {
00329                         Value--;
00330                         CheckValueLimits();
00331                         ValueDisplay->SetText(GetValueAsText());
00332                         DecrementLock = true;
00333                     }
00334                 }
00335             }

00365         void Spinner::ProcessCapturedInputs()
00366         {
00367             if(!Editable)
00368                 return;
00369             if(0 == CaptureData->GetNumCapturedInputs())
00370                 return;
00371             std::vector<MetaCode::InputCode>* CCodes = CaptureData->GetCapturedInputs();
00372             bool EnterPressed = false;
00373             for( Whole X = 0 ; X < CCodes->size() ; X++ )
00374             {
00375                 MetaCode::InputCode Current = CCodes->at(X);
00376                 if(MetaCode::KEY_RETURN == Current || MetaCode::KEY_KP_ENTER == Current)
00377                 {
00378                     EnterPressed = true;
00379                     continue;
00380                 }else{
00381                     EditCache+=StringTool::ConvertToString(Current,false);
00382                 }
00383             }
00384             if(EnterPressed)
00385             {
00386                 std::stringstream TempStream;
00387                 if(DecimalDisplay)
00388                     EditCache.erase(EditCache.find("."),1);
00389                 TempStream << EditCache;
00390                 TempStream >> Value;
00391                 CheckValueLimits();
00392                 ValueDisplay->SetText(GetValueAsText());
00393                 EditCache.clear();
00394             }
00395         }
