00295         void Spinner::Update(bool Force)
00296         {
00297             ProcessCapturedInputs();
00298             if(HoveredButton || Force)
00299             {
00300                 MetaCode::ButtonState State = InputQueryTool::GetMouseButtonState(1);
00301                 if(MetaCode::BUTTON_PRESSING == State)
00302                 {
00303                     if(HoveredButton==Increment)
00304                         IncrementLock = false;
00305                     else if(HoveredButton==Decrement)
00306                         DecrementLock = false;
00307                 }
00308                 else if(MetaCode::BUTTON_LIFTING && (!IncrementLock || !DecrementLock))
00309                 {
00310                     if(HoveredButton==Increment && !IncrementLock)
00311                     {
00312                         Value++;
00313                         CheckValueLimits();
00314                         ValueDisplay->SetText(GetValueAsText());
00315                         IncrementLock = true;
00316                     }
00317                     else if(HoveredButton==Decrement && !DecrementLock)
00318                     {
00319                         Value--;
00320                         CheckValueLimits();
00321                         ValueDisplay->SetText(GetValueAsText());
00322                         DecrementLock = true;
00323                     }
00324                 }
00325             }

00328         void Spinner::ProcessCapturedInputs()
00329         {
00330             if(!Editable)
00331                 return;
00332             if(0 == CaptureData->GetNumCapturedInputs())
00333                 return;
00334             std::vector<MetaCode::InputCode>* CCodes = CaptureData->GetCapturedInputs();
00335             bool EnterPressed = false;
00336             for( Whole X = 0 ; X < CCodes->size() ; X++ )
00337             {
00338                 MetaCode::InputCode Current = CCodes->at(X);
00339                 if(MetaCode::KEY_RETURN == Current || MetaCode::KEY_KP_ENTER == Current)
00340                 {
00341                     EnterPressed = true;
00342                     continue;
00343                 }else{
00344                     EditCache+=StringTool::ConvertInputCodeToString(Current,false);
00345                 }
00346             }
00347             if(EnterPressed)
00348             {
00349                 std::stringstream TempStream;
00350                 if(DecimalDisplay)
00351                     EditCache.erase(EditCache.find("."),1);
00352                 TempStream << EditCache;
00353                 TempStream >> Value;
00354                 CheckValueLimits();
00355                 ValueDisplay->SetText(GetValueAsText());
00356                 EditCache.clear();
00357             }
00358         }
