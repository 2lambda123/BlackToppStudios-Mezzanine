88  void ChannelModifier::Modify(TextureBuffer& Buffer)
89  {
90  for( Whole Y = 0 ; Y < Buffer.GetHeight() ; ++Y )
91  {
92  for( Whole X = 0 ; X < Buffer.GetWidth() ; ++X )
93  {
94  if( this->ChannelOp & ChannelModifier::CS_Gray ) {
95  ColourValue Pixel = Buffer.GetPixel(X,Y);
96  Real Gray = ( Pixel.RedChannel + Pixel.GreenChannel + Pixel.BlueChannel ) / 3.0;
97  Buffer.SetPixelReal(X,Y,Gray,Gray,Gray,Pixel.AlphaChannel);
98  }else{
99  if( this->ChannelOp & ChannelModifier::CS_Red ) Buffer.SetRedReal(X,Y,0.0);
100  if( this->ChannelOp & ChannelModifier::CS_Green ) Buffer.SetGreenReal(X,Y,0.0);
101  if( this->ChannelOp & ChannelModifier::CS_Blue ) Buffer.SetBlueReal(X,Y,0.0);
102  if( this->ChannelOp & ChannelModifier::CS_Alpha ) Buffer.SetAlphaReal(X,Y,0.0);
103  }
104  }
105  }
106  }
