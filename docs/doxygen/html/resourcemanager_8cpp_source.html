112  AssetGroup* ResourceManager::CreateAssetGroup(const String& GroupName)
113  {
114  AssetGroupIterator GroupIt = this->AssetGroups.find(GroupName);

183  String ResourceManager::GetAssetPath(const String& FileName, const String& Group)
184  {
185  Ogre::FileInfoListPtr FileList = this->OgreResource->listResourceFileInfo(Group);
186  for( Whole X = 0 ; X < FileList->size() ; ++X )
187  {
188  if( FileName == FileList->at(X).filename ) {
189  //return FileList->at(X).path;
190  return FileList->at(X).archive->getName() + "/" + FileList->at(X).path;
191  }
192  }
193  return "";
194  }

270  ManagerBase* DefaultResourceManagerFactory::CreateManager(NameValuePairList& Params)
271  {
272  if(ResourceManager::SingletonValid())
273  {
274  /// @todo Add something to log a warning that the manager exists and was requested to be constructed when we have a logging manager set up.
275  return ResourceManager::GetSingletonPtr();
276  }else{
277  if( Params.empty() ) {
278  return new ResourceManager();
279  }else{
280  String EngineDataPath;
281  ArchiveType ArchiveType_;
282  for( NameValuePairList::iterator ParIt = Params.begin() ; ParIt != Params.end() ; ++ParIt )
283  {
284  String Lower = (*ParIt).first;
285  StringTools::ToLowerCase(Lower);
286  if( "enginedatapath" == Lower ) {
287  EngineDataPath = (*ParIt).second;
288  }else if( "archivetype" == Lower ) {
289  ArchiveType_ = ResourceManager::GetArchiveTypeFromString((*ParIt).second);
290  }
291  }
292  return new ResourceManager(EngineDataPath,ArchiveType_);
293  }
294  }
295  }
