00064         String GetOneTag(std::istream& stream )
00065         {
00066             Character ReadOne = 0;
00067             Character PrevOne = 0;
00068             Integer TagCount = 0;
00069             bool HaveSlash = false;
00070             bool InText = false;
00071 
00072             String OneTag;
00073             while (!stream.get(ReadOne).fail() && !stream.eof())        //Read one character and if you didn't fail continue the loop
00074             {
00075                 OneTag.push_back(ReadOne);
00076 
00077                 if ('<' == ReadOne && !InText)                          //If a new tag starts
00078                 {
00079                     ++TagCount;
00080                 }
00081 
00082                 if ( '"' == ReadOne && '\\' != PrevOne )                //If we find an unescaped " the we are either leaving or entering a quote
00083                     { InText = !InText; }
00084 
00085                 if ( !InText && '/' == ReadOne )                        //If we are not In a quote then a / means the next > will end a tag
00086                     { HaveSlash=true; }
00087 
00088                 if ( '>' == ReadOne )                                   // if are at a > then the amount of nested tags we are in changes
00089                 {
00090                     if (HaveSlash)                                      // if have passed an unescaped slash then we  are ending a tage
00091                     {
00092                         --TagCount;
00093                         if ( 0>=TagCount )                              // if we are ending the last tage then we are done
00094                         {
00095                             break;
00096                         }
00097                     }else{
00098                         //++TagCount;                                     // if the tag does not have a slash then we are going one deeper
00099                     }
00100                     HaveSlash=false;
00101                 }
00102 
00103                 PrevOne=ReadOne;                                        //Store the Previous Character to check for escapes
00104             }
00105 
00106             return OneTag;
