91  void RigidDebris::CreateRigidDebris(const Real Mass)
92  {
93  Graphics::SceneManager* SceneMan = Entresol::GetSingletonPtr()->GetSceneManager();
94  if( SceneMan ) {
95  this->EntProx = SceneMan->CreateEntityProxy(false);
96  this->EntProx->_Bind( this );
97  }
98 
99  Physics::PhysicsManager* PhysMan = Entresol::GetSingletonPtr()->GetPhysicsManager();
100  if( PhysMan ) {
101  this->RigProx = PhysMan->CreateRigidProxy(Mass,false);
102  this->RigProx->_Bind( this );
103  }
104 
105  if( this->EntProx && this->RigProx ) {
106  this->RigProx->AddSyncObject( this->EntProx );
107  }
108  }

323  void RigidDebris::ProtoSerializeProxies(XML::Node& SelfRoot) const
324  {
325  // No base implementations to call
326  XML::Node ProxiesNode = SelfRoot.AppendChild( RigidDebris::GetSerializableName() + "Proxies" );
327 
328  if( ProxiesNode.AppendAttribute("Version").SetValue("1") )
329  {
330  XML::Node EntProxNode = ProxiesNode.AppendChild("EntProx");
331  this->EntProx->ProtoSerialize( EntProxNode );
332  XML::Node RigProxNode = ProxiesNode.AppendChild("RigProx");
333  this->RigProx->ProtoSerialize( RigProxNode );
334 
335  return;
336  }else{
337  SerializeError("Create XML Attribute Values",RigidDebris::GetSerializableName() + "Proxies",true);
338  }
339  }

346  void RigidDebris::ProtoDeSerializeProxies(const XML::Node& SelfRoot)
347  {
348  this->DestroyRigidDebris();
349  // No base implementations to call
350  //XML::Attribute CurrAttrib;
351  XML::Node ProxiesNode = SelfRoot.GetChild( RigidDebris::GetSerializableName() + "Proxies" );
352 
353  if( !ProxiesNode.Empty() ) {
354  if(ProxiesNode.GetAttribute("Version").AsInt() == 1) {
355  /// @todo I don't think an exception is appropriate for the failure of the worldmanager validity checks,
356  /// however a warning should be written to the log if that happens. This should be updated to do that once
357  /// logging refactors are done.
358 
359  XML::Node EntProxNode = ProxiesNode.GetChild("EntProx").GetFirstChild();
360  if( !EntProxNode.Empty() ) {
361  Graphics::SceneManager* SceneMan = Entresol::GetSingletonPtr()->GetSceneManager();
362  if( SceneMan ) {
363  this->EntProx = SceneMan->CreateEntityProxy( EntProxNode );
364  this->EntProx->_Bind( this );
365  }
366  }
367 
368  XML::Node RigProxNode = ProxiesNode.GetChild("RigProx").GetFirstChild();
369  if( !RigProxNode.Empty() ) {
370  Physics::PhysicsManager* PhysMan = Entresol::GetSingletonPtr()->GetPhysicsManager();
371  if( PhysMan ) {
372  this->RigProx = PhysMan->CreateRigidProxy(RigProxNode);
373  this->RigProx->_Bind( this );
374  }
375  }
376 
377  if( this->EntProx && this->RigProx ) {
378  this->RigProx->AddSyncObject( this->EntProx );
379  }
380  }else{
381  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (RigidDebris::GetSerializableName() + "Proxies" ) + ": Not Version 1.");
382  }
383  }else{
384  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,RigidDebris::GetSerializableName() + "Proxies" + " was not found in the provided XML node, which was expected.");
385  }
386  }
