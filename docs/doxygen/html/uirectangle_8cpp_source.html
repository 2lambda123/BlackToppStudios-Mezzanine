00151         void Rectangle::RotationTransform(Vector2& TopLeft, Vector2& TopRight, Vector2& BottomLeft, Vector2& BottomRight)
00152         {
00153             if(0 != RotAngle)
00154             {
00155                 Real RotCos = MathTool::Cos(RotAngle);
00156                 Real RotSin = MathTool::Sin(RotAngle);
00157 
00158                 if (!CustomCenter)
00159                 {
00160                     RotCenter.X = TopLeft.X + (BottomRight.X - TopLeft.X) / 2;
00161                     RotCenter.Y = TopLeft.Y + (BottomRight.Y - TopLeft.Y) / 2;
00162                 }
00163 
00164                 TopLeft.X -= RotCenter.X;
00165                 TopLeft.Y -= RotCenter.Y;
00166 
00167                 TopRight.X -= RotCenter.X;
00168                 TopRight.Y -= RotCenter.Y;
00169 
00170                 BottomLeft.X -= RotCenter.X;
00171                 BottomLeft.Y -= RotCenter.Y;
00172 
00173                 BottomRight.X -= RotCenter.X;
00174                 BottomRight.Y -= RotCenter.Y;
00175 
00176                 Vector2 A2 = TopLeft, B2 = TopRight, C2 = BottomLeft, D2 = BottomRight;
00177                 TopLeft.X = A2.X * RotCos - A2.Y * RotSin;
00178                 TopLeft.Y = A2.X * RotSin + A2.Y * RotCos;
00179 
00180                 TopRight.X = B2.X * RotCos - B2.Y * RotSin;
00181                 TopRight.Y = B2.X * RotSin + B2.Y * RotCos;
00182 
00183                 BottomLeft.X = C2.X * RotCos - C2.Y * RotSin;
00184                 BottomLeft.Y = C2.X * RotSin + C2.Y * RotCos;
00185 
00186                 BottomRight.X = D2.X * RotCos - D2.Y * RotSin;
00187                 BottomRight.Y = D2.X * RotSin + D2.Y * RotCos;
00188 
00189                 TopLeft.X += RotCenter.X;
00190                 TopLeft.Y += RotCenter.Y;
00191 
00192                 TopRight.X += RotCenter.X;
00193                 TopRight.Y += RotCenter.Y;
00194 
00195                 BottomLeft.X += RotCenter.X;
00196                 BottomLeft.Y += RotCenter.Y;
00197 
00198                 BottomRight.X += RotCenter.X;
00199                 BottomRight.Y += RotCenter.Y;
00200             }
00201         }
