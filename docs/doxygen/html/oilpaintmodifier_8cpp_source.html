91  void OilPaintModifier::Modify(TextureBuffer& Buffer)
92  {
93  TextureBuffer TempBuffer( Buffer );
94 
95  Whole IntensitiesArray[256];
96  Whole RedArray[256];
97  Whole GreenArray[256];
98  Whole BlueArray[256];
99 
100  for( Integer Y = this->Radius ; Y < static_cast<Integer>( Buffer.GetHeight() - this->Radius ) ; ++Y )
101  {
102  for( Integer X = this->Radius ; X < static_cast<Integer>( Buffer.GetWidth() - this->Radius ) ; ++X )
103  {
104  memset(IntensitiesArray, 0, sizeof(IntensitiesArray));
105  memset(RedArray, 0, sizeof(RedArray));
106  memset(GreenArray, 0, sizeof(GreenArray));
107  memset(BlueArray, 0, sizeof(BlueArray));
108 
109  for( Integer j = -this->Radius ; j <= this->Radius ; ++j )
110  {
111  for( Integer i = -this->Radius ; i <= this->Radius ; ++i )
112  {
113  UInt16 Red = Buffer.GetRedByte( static_cast<Whole>(X + i), static_cast<Whole>(Y + j) );
114  UInt16 Green = Buffer.GetGreenByte( static_cast<Whole>(X + i), static_cast<Whole>(Y + j) );
115  UInt16 Blue = Buffer.GetBlueByte( static_cast<Whole>(X + i), static_cast<Whole>(Y + j) );
116 
117  Whole CurrIntensity = static_cast<Whole>( ( ( static_cast<Real>(Red + Green + Blue) / 3.0 ) * this->IntensityLevels ) / 255.0 );
118  if( CurrIntensity > 255 ) {
119  CurrIntensity = 255;
120  }
121  IntensitiesArray[CurrIntensity]++;
122 
123  RedArray[CurrIntensity] += Red;
124  GreenArray[CurrIntensity] += Green;
125  BlueArray[CurrIntensity] += Blue;
126  }
127  }
128 
129  Whole MaxIntensity = 0;
130  Whole MaxIndex = 0;
131  for( Whole i = 0 ; i < 256 ; ++i )
132  {
133  if( IntensitiesArray[i] > MaxIntensity ) {
134  MaxIntensity = IntensitiesArray[i];
135  MaxIndex = i;
136  }
137  }
138 
139  TempBuffer.SetPixelByte( static_cast<Whole>(X), static_cast<Whole>(Y),
140  static_cast<UInt8>(RedArray[MaxIndex] / MaxIntensity),
141  static_cast<UInt8>(GreenArray[MaxIndex] / MaxIntensity),
142  static_cast<UInt8>(BlueArray[MaxIndex] / MaxIntensity) );
143  }
144  }
145 
146  Buffer.SetData(TempBuffer);
147  }
