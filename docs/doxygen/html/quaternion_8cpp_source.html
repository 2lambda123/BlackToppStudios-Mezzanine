00119     void Quaternion::SetFromMatrix3x3(const Matrix3x3& Mat)
00120     {
00121         Real Trace = Mat.Matrix[0][0]+Mat.Matrix[1][1]+Mat.Matrix[2][2];
00122         Real Root;
00123 
00124         if(Trace > 0.0)
00125         {
00126             Root = MathTool::Sqrt(Trace + 1.0);
00127             this->W = Root * 0.5;
00128             Root = 0.5/Root;
00129             this->X = (Mat.Matrix[2][1] - Mat.Matrix[1][2]) * Root;
00130             this->Y = (Mat.Matrix[0][2] - Mat.Matrix[2][0]) * Root;
00131             this->Z = (Mat.Matrix[1][0] - Mat.Matrix[0][1]) * Root;
00132         }else{
00133             static size_t s_iNext[3] = { 1, 2, 0 };
00134             size_t I = 0;
00135             if ( Mat.Matrix[1][1] > Mat.Matrix[0][0] )
00136                 I = 1;
00137             if ( Mat.Matrix[2][2] > Mat.Matrix[I][I] )
00138                 I = 2;
00139             size_t J = s_iNext[I];
00140             size_t K = s_iNext[J];
00141 
00142             Root = MathTool::Sqrt(Mat.Matrix[I][I] - Mat.Matrix[J][J] - Mat.Matrix[K][K] + 1.0);
00143             Real* QuatMem[3] = { &X, &Y, &Z };
00144             *QuatMem[I] = 0.5 * Root;
00145             Root = 0.5 / Root;
00146             this->W = (Mat.Matrix[K][J] - Mat.Matrix[J][K]) * Root;
00147             *QuatMem[J] = (Mat.Matrix[J][I] + Mat.Matrix[I][J]) * Root;
00148             *QuatMem[K] = (Mat.Matrix[K][I] + Mat.Matrix[I][K]) * Root;
00149         }
00150     }
