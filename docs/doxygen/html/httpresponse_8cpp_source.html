77  Boole HTTPResponse::ParseHTTPHeader(StringIterator& CurrIt, const StringIterator EndIt)
78  {
79  // A response header has 3 components: A version, a response code, and that codes' description.
80  Boole EoL = false;
81  String HeaderTemp;
82  // Version
83  EoL = this->GetMessageComponent(CurrIt,EndIt,HeaderTemp);
84  this->ParseHTTPVersion(HeaderTemp);
85  if( EoL ) {
86  return false;
87  }
88  // Response Code
89  HeaderTemp.clear();
90  EoL = this->GetMessageComponent(CurrIt,EndIt,HeaderTemp);
91  this->ResponseCode = StringTools::ConvertToWhole(HeaderTemp);
92  if( EoL ) {
93  return false;
94  }
95  // Response Description
96  this->ResponseDescription.clear();
97  this->GetMessageComponent(CurrIt,EndIt,this->ResponseDescription);
98  // Handle extra trash
99  //this->AdvanceToNewline(CurrIt,EndIt);
100  return true;
101  }

106  String HTTPResponse::Compose() const
107  {
108  StringStream ResponseStream;
109  // Header
110  ResponseStream << "HTTP/" << this->MessageVersion.Major << "." << this->MessageVersion.Minor << " ";
111  ResponseStream << StringTools::ConvertToString( this->ResponseCode ) << " " << this->ResponseDescription << "\r\n";
112  // Fields
113  for( NameValuePairMap::const_iterator FieldIt = this->MessageFields.begin() ; FieldIt != this->MessageFields.end() ; ++FieldIt )
114  {
115  ResponseStream << (*FieldIt).first << ": " << (*FieldIt).second << "\r\n";
116  }
117  ResponseStream << "\r\n";
118  // Body
119  ResponseStream << this->MessageBody;
120  return ResponseStream.str();
121  }

123  Boole HTTPResponse::Decompose(const String& Response)
124  {
125  if( !Response.empty() ) {
126  // Set up the data
127  StringIterator CurrIt = Response.begin();
128  const StringIterator EndIt = Response.end();
129  // Start decomposing
130  if( !this->ParseHTTPHeader(CurrIt,EndIt) ) {
131  return false;
132  }
133  if( !this->ParseHTTPFields(CurrIt,EndIt) ) {
134  return false;
135  }
136 
137  // The message body may or may not be chunked
138  this->MessageBody.clear();
139  if( this->GetField("transfer-encoding") != "chunked" ) {
140  NameValuePairMap::iterator FieldIt = this->MessageFields.find("content-length");
141  if( FieldIt == this->MessageFields.end() ) {
142  return false;
143  //MEZZ_EXCEPTION(ExceptionBase::INVALID_STATE_EXCEPTION,"HTTP Response does not contain a \"Content-Length\" nor \"Transfer-Encoding\" header. Must contain one of these to be valid.");
144  }
145  this->MessageBody.reserve( StringTools::ConvertToWhole( (*FieldIt).second ) );
146  this->MessageBody.assign(CurrIt,EndIt);
147  }else{
148  String ChunkSizeStr;
149  this->GetMessageComponent(CurrIt,EndIt,ChunkSizeStr);
150  size_t ChunkLength = ConvertToWholeHex(ChunkSizeStr);
151  while( ChunkLength != 0 )
152  {
153  this->MessageBody.reserve( this->MessageBody.size() + ChunkLength );
154  while( (*CurrIt) != '\r' && CurrIt != EndIt )
155  {
156  this->MessageBody.push_back(*CurrIt);
157  ++CurrIt;
158  }
159 
160  ChunkSizeStr.clear();
161  this->GetMessageComponent(CurrIt,EndIt,ChunkSizeStr);
162  ChunkLength = ConvertToWholeHex(ChunkSizeStr);
163  }
164  // When the body is chunked, there may be optional trailing headers
165  if( !this->ParseHTTPFields(CurrIt,EndIt) ) {
166  return false;
167  }
168  }
169  return true;
170  }
171 
172  return false;
173  }
