00064     /// @enum AttenuationStyle
00065     /// @brief These values represent the kind of attenuation applied to the field strength
00066     /// over a distance.
00067     /// @details None is the default, where the force is constant in all area's of the
00068     /// field.  Linear is where the force applied drops by the attenuation value times the distance
00069     /// (strength - (attenuation amount * distance to AE center)).  Quadratic is where the force
00070     /// applied drops by the attenuation value times the distance squared (strength -
00071     /// (attenuation amount * distance to AE center * distance to AE center)).
00072     enum AttenuationStyle
00073     {
00074         Att_None,        ///< No Attentuation, Equal strength through.
00075         Att_Linear,      ///< Linear attentuation, Strength weaker farther from center.
00076         Att_Quadratic    ///< Quadratic/Exponential Attentuation, similar to real gravity, it tapers of more the further from the center you get.
00077     };
