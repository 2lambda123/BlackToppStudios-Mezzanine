00279     {
00280         Ogre::MaterialPtr TheMaterial = Ogre::MaterialManager::getSingleton().getByName(MaterialName);
00281         String GroupName = TheMaterial->getGroup();
00282 
00283         Real RingAngle = (Ogre::Math::PI / Rings);
00284         Real SegAngle = (2 * Ogre::Math::PI / Segments);
00285         Whole VertIndex = 0;
00286 
00287         Ogre::ManualObject* sphere = new Ogre::ManualObject("TempMan");
00288         sphere->begin(MaterialName);
00289 
00290         for( Whole ring = 0 ; ring <= Rings ; ring++ ) {
00291         Real r0 = Radius * sinf (ring * RingAngle);
00292         Real y0 = Radius * cosf (ring * RingAngle);
00293 
00294             // Generate the group of segments for the current ring
00295             for( Whole seg = 0 ; seg <= Segments ; seg++ ) {
00296                 Real x0 = r0 * sinf(seg * SegAngle);
00297                 Real z0 = r0 * cosf(seg * SegAngle);
00298 
00299                 // Add one vertex to the strip which makes up the sphere
00300                 sphere->position( x0, y0, z0);
00301                 sphere->normal(Ogre::Vector3(x0, y0, z0).normalisedCopy());
00302                 sphere->textureCoord((Real) seg / (Real) Segments, (Real) ring / (Real) Rings);
00303 
00304                 if (ring != Rings) {
00305                     // each vertex (except the last) has six indicies pointing to it
00306                     sphere->index(VertIndex + Segments + 1);
00307                     sphere->index(VertIndex);
00308                     sphere->index(VertIndex + Segments);
00309                     sphere->index(VertIndex + Segments + 1);
00310                     sphere->index(VertIndex + 1);
00311                     sphere->index(VertIndex);
00312                     VertIndex++;
00313                 }
00314             }
00315         }
00316 
00317         sphere->end();

00319         spheremesh->_setBounds(Ogre::AxisAlignedBox(-Radius,-Radius,-Radius,Radius,Radius,Radius));
00320         spheremesh->_setBoundingSphereRadius(Radius);
